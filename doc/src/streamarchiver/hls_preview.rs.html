1//! Live HLS playlist generation over growing SABR fragmented-MP4 files.
2//!
3//! A SABR live-edge preview download produces two separate growing fMP4 files
4//! (video + audio). No mpv-native mechanism follows them reliably at the live
5//! edge: `appending://` retries reads at EOF for only ~2 s, which loses the
6//! race against SABR's segment cadence once and then the demuxer latches EOF
7//! permanently (seek nudges land in the demuxer cache and never touch the
8//! file). ffmpeg's HLS demuxer, however, is *designed* to poll a live
9//! playlist forever — so we hand the player a hand-rolled local HLS master:
10//! each media playlist maps the file's `ftyp+moov` init via `EXT-X-MAP` and
11//! addresses runs of `moof+mdat` fragments as `EXT-X-BYTERANGE` segments into
12//! the growing file. An updater rewrites the playlists every couple of
13//! seconds as new fragments land. Verified against a live capture: mpv plays
14//! 1:1 with wall clock at the edge with ~40 s of readahead and never EOFs.
15//!
16//! SABR writes a moof/mdat pair every ~16 ms, far too granular for playlist
17//! entries — fragments are coalesced into ~[`SEG_SECS`]-second segments using
18//! the `tfdt` timestamp deltas.
19
20use std::fs::File;
21use std::io::{Read, Seek, SeekFrom};
22use std::path::{Path, PathBuf};
23
24/// Target coalesced HLS segment duration, in media time.
25const SEG_SECS: f64 = 3.0;
26
27/// One coalesced playlist segment: a byte range covering whole moof+mdat
28/// fragments, starting at `tfdt` (in the track's timescale).
29#[derive(Clone, Copy, Debug)]
30struct Segment {
31    offset: u64,
32    len: u64,
33    tfdt: Option<u64>,
34}
35
36/// Incremental top-level ISOBMFF box walker over a growing fMP4.
37struct Scanner {
38    file: File,
39    /// Next unparsed top-level box offset.
40    off: u64,
41    /// Byte length of the leading `ftyp+moov` init segment, once seen.
42    init_end: Option<u64>,
43    /// Media timescale from the first `moov > trak > mdia > mdhd`.
44    timescale: u32,
45    /// Coalesced segments, append-only.
46    segments: Vec<Segment>,
47    /// Start of the currently-open (not yet emitted) group.
48    group: Option<(u64, Option<u64>)>,
49    /// `moof` awaiting its `mdat`: (offset, tfdt).
50    pending_moof: Option<(u64, Option<u64>)>,
51}
52
53/// Read a box header at `off`: (total box size, fourcc). `None` on a short or
54/// nonsense read (box still being written, or not ISOBMFF).
55fn box_header(f: &mut File, off: u64) -> Option<(u64, [u8; 4])> {
56    let mut hdr = [0u8; 8];
57    f.seek(SeekFrom::Start(off)).ok()?;
58    f.read_exact(&mut hdr).ok()?;
59    let mut size = u32::from_be_bytes([hdr[0], hdr[1], hdr[2], hdr[3]]) as u64;
60    let typ = [hdr[4], hdr[5], hdr[6], hdr[7]];
61    if size == 1 {
62        let mut big = [0u8; 8];
63        f.read_exact(&mut big).ok()?;
64        size = u64::from_be_bytes(big);
65    }
66    (size >= 8).then_some((size, typ))
67}
68
69/// Find the first direct child box named `name` within `[off, end)`.
70fn find_child(f: &mut File, mut off: u64, end: u64, name: &[u8; 4]) -> Option<(u64, u64)> {
71    while off + 8 <= end {
72        let (size, typ) = box_header(f, off)?;
73        if &typ == name {
74            return Some((off, size));
75        }
76        off += size;
77    }
78    None
79}

81/// Timescale from the first `trak > mdia > mdhd` inside `moov` (one track per
82/// SABR per-format file).
83fn mdhd_timescale(f: &mut File, moov_off: u64, moov_size: u64) -> Option<u32> {
84    let end = moov_off + moov_size;
85    let (trak, tsz) = find_child(f, moov_off + 8, end, b"trak")?;
86    let (mdia, msz) = find_child(f, trak + 8, trak + tsz, b"mdia")?;
87    let (mdhd, _) = find_child(f, mdia + 8, mdia + msz, b"mdhd")?;
88    let mut ver = [0u8; 1];
89    f.seek(SeekFrom::Start(mdhd + 8)).ok()?;
90    f.read_exact(&mut ver).ok()?;
91    // v0: 4B flags-rest + 2×4B times; v1: + 2×8B times.
92    let ts_off = mdhd + 8 + 4 + if ver[0] == 1 { 16 } else { 8 };
93    let mut ts = [0u8; 4];
94    f.seek(SeekFrom::Start(ts_off)).ok()?;
95    f.read_exact(&mut ts).ok()?;
96    Some(u32::from_be_bytes(ts))
97}

99/// `baseMediaDecodeTime` from `moof > traf > tfdt`.
100fn tfdt_time(f: &mut File, moof_off: u64, moof_size: u64) -> Option<u64> {
101    let end = moof_off + moof_size;
102    let (traf, tsz) = find_child(f, moof_off + 8, end, b"traf")?;
103    let (tfdt, _) = find_child(f, traf + 8, traf + tsz, b"tfdt")?;
104    let mut ver = [0u8; 1];
105    f.seek(SeekFrom::Start(tfdt + 8)).ok()?;
106    f.read_exact(&mut ver).ok()?;
107    f.seek(SeekFrom::Start(tfdt + 12)).ok()?;
108    if ver[0] == 1 {
109        let mut t = [0u8; 8];
110        f.read_exact(&mut t).ok()?;
111        Some(u64::from_be_bytes(t))
112    } else {
113        let mut t = [0u8; 4];
114        f.read_exact(&mut t).ok()?;
115        Some(u32::from_be_bytes(t) as u64)
116    }
117}

119impl Scanner {
120    /// Open a growing file, verifying it starts with an ISOBMFF `ftyp`.
121    fn open(path: &Path) -> Option<Scanner> {
122        let mut file = File::open(path).ok()?;
123        let (_, typ) = box_header(&mut file, 0)?;
124        if &typ != b"ftyp" {
125            return None;
126        }
127        Some(Scanner {
128            file,
129            off: 0,
130            init_end: None,
131            timescale: 90_000,
132            segments: Vec::new(),
133            group: None,
134            pending_moof: None,
135        })
136    }
137
138    /// True current size via the open handle (directory-entry sizes are stale
139    /// for files another process is writing).
140    fn size(&mut self) -> u64 {
141        self.file.seek(SeekFrom::End(0)).unwrap_or(0)
142    }
143
144    /// Parse any newly-landed complete top-level boxes, coalescing fragments.
145    fn scan(&mut self) {
146        let end = self.size();
147        while self.off + 8 <= end {
148            let Some((size, typ)) = box_header(&mut self.file, self.off) else { break };
149            if self.off + size > end {
150                break; // box still being written
151            }
152            match &typ {
153                b"moov" if self.init_end.is_none() => {
154                    // SABR re-appends a fresh ftyp+moov after reconnects; the
155                    // EXT-X-MAP must reference only the FIRST init.
156                    if let Some(ts) = mdhd_timescale(&mut self.file, self.off, size) {
157                        self.timescale = ts.max(1);
158                    }
159                    self.init_end = Some(self.off + size);
160                }
161                b"moof" => {
162                    self.pending_moof =
163                        Some((self.off, tfdt_time(&mut self.file, self.off, size)));
164                }
165                b"mdat" => {
166                    if let Some((moof_off, t)) = self.pending_moof.take() {
167                        match self.group {
168                            None => self.group = Some((moof_off, t)),
169                            Some((g_off, Some(g_t))) => {
170                                if let Some(t) = t {
171                                    let secs =
172                                        (t.saturating_sub(g_t)) as f64 / self.timescale as f64;
173                                    if secs >= SEG_SECS {
174                                        self.segments.push(Segment {
175                                            offset: g_off,
176                                            len: moof_off - g_off,
177                                            tfdt: Some(g_t),
178                                        });
179                                        self.group = Some((moof_off, Some(t)));
180                                    }
181                                }
182                            }
183                            Some((_, None)) => {
184                                // No timestamp on the group start; restart it.
185                                self.group = Some((moof_off, t));
186                            }
187                        }
188                    }
189                }
190                _ => {}
191            }
192            self.off += size;
193        }
194    }
195
196    /// Segment durations in seconds. Each segment ends where the next one
197    /// starts; the last emitted segment ends at the open group's start.
198    fn durations(&self) -> Vec<f64> {
199        let mut durs = Vec::with_capacity(self.segments.len());
200        for (i, s) in self.segments.iter().enumerate() {
201            let next_t = match self.segments.get(i + 1) {
202                Some(n) => n.tfdt,
203                None => self.group.and_then(|(_, t)| t),
204            };
205            let d = match (s.tfdt, next_t) {
206                (Some(a), Some(b)) if b > a => (b - a) as f64 / self.timescale as f64,
207                _ => durs.last().copied().unwrap_or(SEG_SECS),
208            };
209            durs.push(f64::max(d, 0.04));
210        }
211        durs
212    }
213}

215/// Atomically replace `dst` with `content`, retrying briefly: the player may
216/// hold the playlist open at reload time, and on Windows a rename onto an
217/// open file fails with a sharing violation. A skipped cycle just means the
218/// playlist lags one update.
219fn replace_playlist(dst: &Path, content: &str) {
220    let tmp = dst.with_extension("m3u8.tmp");
221    if std::fs::write(&tmp, content).is_err() {
222        return;
223    }
224    for _ in 0..10 {
225        if std::fs::rename(&tmp, dst).is_ok() {
226            return;
227        }
228        std::thread::sleep(std::time::Duration::from_millis(30));
229    }
230    let _ = std::fs::remove_file(&tmp);
231}

233/// Render a media playlist over the scanner's segments. `media_uri` is the
234/// URI of the growing file relative to the playlist. `ended` appends
235/// `EXT-X-ENDLIST` (the preview download finished — lets the player end
236/// cleanly instead of polling forever).
237fn media_playlist(scan: &Scanner, media_uri: &str, ended: bool) -> Option<String> {
238    let init_end = scan.init_end?;
239    if scan.segments.len() < 2 {
240        return None;
241    }
242    let durs = scan.durations();
243    let target = durs.iter().fold(0.0f64, |a, &b| a.max(b));
244    let mut s = String::with_capacity(scan.segments.len() * 96 + 256);
245    s.push_str("#EXTM3U\n#EXT-X-VERSION:7\n#EXT-X-MEDIA-SEQUENCE:0\n");
246    s.push_str(&format!("#EXT-X-TARGETDURATION:{}\n", target.ceil() as u64 + 1));
247    s.push_str(&format!("#EXT-X-MAP:URI=\"{media_uri}\",BYTERANGE=\"{init_end}@0\"\n"));
248    for (seg, d) in scan.segments.iter().zip(&durs) {
249        s.push_str(&format!(
250            "#EXTINF:{d:.3},\n#EXT-X-BYTERANGE:{}@{}\n{media_uri}\n",
251            seg.len, seg.offset
252        ));
253    }
254    if ended {
255        s.push_str("#EXT-X-ENDLIST\n");
256    }
257    Some(s)
258}

260/// Live HLS generator for a SABR split-A/V preview: playlists live in
261/// `out_dir`, addressing the two growing files. Call [`Self::tick`]
262/// periodically; it returns `true` once `master.m3u8` is ready to play.
263pub(crate) struct HlsPreview {
264    video: Scanner,
265    audio: Scanner,
266    video_uri: String,
267    audio_uri: String,
268    out_dir: PathBuf,
269    master_written: bool,
270}
271
272impl HlsPreview {
273    /// `video`/`audio` are the growing fMP4s (largest-first order from the
274    /// stream-target probe). Returns `None` unless both parse as ISOBMFF —
275    /// the caller falls back to plain `appending://` playback (e.g. for the
276    /// VP9-in-MKV SABR variant).
277    pub(crate) fn open(video: &Path, audio: &Path, out_dir: &Path) -> Option<HlsPreview> {
278        // Relative URI from the playlist dir to the media file (same dir or a
279        // subdir like `.cache/…`); forward slashes for URI use.
280        let rel_uri = |p: &Path| -> Option<String> {
281            let rel = p.strip_prefix(out_dir).ok()?;
282            Some(rel.to_string_lossy().replace('\\', "/"))
283        };
284        Some(HlsPreview {
285            video: Scanner::open(video)?,
286            audio: Scanner::open(audio)?,
287            video_uri: rel_uri(video)?,
288            audio_uri: rel_uri(audio)?,
289            out_dir: out_dir.to_path_buf(),
290            master_written: false,
291        })
292    }
293
294    pub(crate) fn master_path(&self) -> PathBuf {
295        self.out_dir.join("master.m3u8")
296    }
297
298    /// Scan for new fragments and rewrite the playlists. Returns `true` once
299    /// the master playlist exists (both media playlists have ≥2 segments).
300    pub(crate) fn tick(&mut self, ended: bool) -> bool {
301        self.video.scan();
302        self.audio.scan();
303        let v = media_playlist(&self.video, &self.video_uri, ended);
304        let a = media_playlist(&self.audio, &self.audio_uri, ended);
305        if let Some(v) = &v {
306            replace_playlist(&self.out_dir.join("video.m3u8"), v);
307        }
308        if let Some(a) = &a {
309            replace_playlist(&self.out_dir.join("audio.m3u8"), a);
310        }
311        if !self.master_written && v.is_some() && a.is_some() {
312            let master = "#EXTM3U\n#EXT-X-VERSION:7\n\
313                 #EXT-X-MEDIA:TYPE=AUDIO,GROUP-ID=\"a\",NAME=\"audio\",DEFAULT=YES,URI=\"audio.m3u8\"\n\
314                 #EXT-X-STREAM-INF:BANDWIDTH=6000000,AUDIO=\"a\"\nvideo.m3u8\n";
315            replace_playlist(&self.master_path(), master);
316            self.master_written = true;
317        }
318        self.master_written
319    }
320}

327    fn boxed(typ: &[u8; 4], payload: &[u8]) -> Vec<u8> {
328        let mut v = Vec::with_capacity(8 + payload.len());
329        v.extend_from_slice(&((payload.len() as u32 + 8).to_be_bytes()));
330        v.extend_from_slice(typ);
331        v.extend_from_slice(payload);
332        v
333    }
334
335    /// Minimal moov with trak>mdia>mdhd carrying `timescale` (v0 layout).
336    fn moov(timescale: u32) -> Vec<u8> {
337        let mut mdhd = vec![0u8; 4]; // version+flags
338        mdhd.extend_from_slice(&[0u8; 8]); // creation+modification (v0)
339        mdhd.extend_from_slice(&timescale.to_be_bytes());
340        mdhd.extend_from_slice(&[0u8; 4]); // duration
341        mdhd.extend_from_slice(&[0u8; 4]); // language+quality
342        let mdia = boxed(b"mdia", &boxed(b"mdhd", &mdhd));
343        let trak = boxed(b"trak", &mdia);
344        boxed(b"moov", &trak)
345    }
346
347    /// moof with traf>tfdt (v1) at `t`.
348    fn moof(t: u64) -> Vec<u8> {
349        let mut tfdt = vec![1u8, 0, 0, 0]; // version 1 + flags
350        tfdt.extend_from_slice(&t.to_be_bytes());
351        boxed(b"moof", &boxed(b"traf", &boxed(b"tfdt", &tfdt)))
352    }

354    fn write_fmp4(path: &std::path::Path, times_ms: &[u64]) {
355        let mut f = std::fs::File::create(path).unwrap();
356        f.write_all(&boxed(b"ftyp", b"isom")).unwrap();
357        f.write_all(&moov(1000)).unwrap(); // timescale 1000 = ms
358        for &t in times_ms {
359            f.write_all(&moof(t)).unwrap();
360            f.write_all(&boxed(b"mdat", &[0u8; 32])).unwrap();
361        }
362    }

364    #[test]
365    fn scanner_coalesces_fragments_into_segments() {
366        let dir = std::env::temp_dir().join(format!(
367            "sa_hls_{}_{}",
368            std::process::id(),
369            crate::models::now_unix()
370        ));
371        std::fs::create_dir_all(&dir).unwrap();
372        let media = dir.join("preview.f400.mp4.sq0.part");
373        // Fragments every second; groups close when the NEXT group's start is
374        // ≥3s later: expect boundaries at 0→3000 and 3000→6500.
375        write_fmp4(&media, &[0, 1000, 2000, 3000, 4000, 5000, 6500, 7000]);
376
377        let mut sc = Scanner::open(&media).unwrap();
378        sc.scan();
379        assert_eq!(sc.timescale, 1000);
380        assert!(sc.init_end.is_some());
381        assert_eq!(sc.segments.len(), 2);
382        assert_eq!(sc.segments[0].tfdt, Some(0));
383        assert_eq!(sc.segments[1].tfdt, Some(3000));
384        let durs = sc.durations();
385        assert!((durs[0] - 3.0).abs() < 1e-9);
386        assert!((durs[1] - 3.5).abs() < 1e-9);
387
388        // Growing file: appending more fragments extends the segment list.
389        {
390            let mut f = std::fs::OpenOptions::new().append(true).open(&media).unwrap();
391            f.write_all(&moof(10_000)).unwrap();
392            f.write_all(&boxed(b"mdat", &[0u8; 32])).unwrap();
393        }
394        sc.scan();
395        assert_eq!(sc.segments.len(), 3); // 6500→10000 closed
396
397        let _ = std::fs::remove_dir_all(&dir);
398    }
