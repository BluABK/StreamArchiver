1//! Bulk-import channels a user already follows/subscribes to, from each platform.
2//!
3//! * **Twitch** — the connected account's *followed* channels (Helix
4//!   `channels/followed`, needs the `user:read:follows` scope).
5//! * **YouTube** — the connected Google account's *subscriptions* (YouTube Data
6//!   API `subscriptions.list?mine=true`, needs a connected Google account via
7//!   [`crate::google_oauth`]).
8//!
9//! Each returns a flat list of [`ImportCandidate`]s; the UI shows a confirmation
10//! dialog (pick which, set Auto / Disabled) and then calls [`create_monitor`] per
11//! chosen entry, reusing the exact same per-platform defaults a manual "Add
12//! stream" would.
13
14use anyhow::{Result, bail};
15use reqwest::Client;
16use tracing::warn;

18use crate::models::{AuthKind, Monitor, MonitorDefaults, Platform};
19use crate::store::Store;

22/// Hard cap on imported pages, so a huge follow/subscription list can't loop
23/// unboundedly (100/page Twitch, 50/page YouTube → up to a few thousand).
24const MAX_PAGES: usize = 60;
25
26/// One channel offered for import, with enough detail for the confirm dialog.
27#[derive(Clone)]
28pub struct ImportCandidate {
29    pub platform: Platform,
30    /// Display name (Twitch display name / YouTube channel title).
31    pub name: String,
32    /// Stable identity for dedup against existing monitors (lowercased login /
33    /// channel id).
34    pub identity: String,
35    /// Platform id (Twitch broadcaster id / YouTube channel id) — shown in the UI.
36    pub id: String,
37    /// Canonical channel URL to store on the monitor.
38    pub url: String,
39    /// Extra one-line detail for the dialog (e.g. "followed 2021-05-03").
40    pub detail: String,
41}
42
43/// The connected Twitch account's followed channels, sorted by name.
44pub async fn twitch_followed(http: &Client, store: &Store) -> Result<Vec<ImportCandidate>> {
45    let client_id = store
46        .get_setting("twitch_client_id")
47        .ok()
48        .flatten()
49        .filter(|s| !s.is_empty())
50        .ok_or_else(|| anyhow::anyhow!("Set a Twitch Client ID in Settings first."))?;
51    let token = oauth::valid_user_token(http, store)
52        .await
53        .ok_or_else(|| anyhow::anyhow!("Connect your Twitch account in Settings first."))?;
54    // Older connections may predate user-id persistence — resolve it on demand
55    // (Get Users) and cache it, so the import works without a reconnect.
56    let user_id = match oauth::connected_user_id(store) {
57        Some(id) => id,
58        None => match oauth::fetch_user(http, &client_id, &token).await {
59            Ok((_login, id)) => {
60                let _ = store.set_setting(oauth::K_USER_ID, &id);
61                id
62            }
63            Err(_) => bail!("Reconnect your Twitch account in Settings to enable importing."),
64        },
65    };
66
67    let mut out = Vec::new();
68    let mut cursor: Option<String> = None;
69    let mut truncated = true;
70    for _ in 0..MAX_PAGES {
71        let mut query: Vec<(&str, &str)> = vec![("user_id", user_id.as_str()), ("first", "100")];
72        if let Some(c) = &cursor {
73            query.push(("after", c.as_str()));
74        }
75        let resp = http
76            .get("https://api.twitch.tv/helix/channels/followed")
77            .header("Client-Id", &client_id)
78            .bearer_auth(&token)
79            .query(&query)
80            .send()
81            .await?;
82        if !resp.status().is_success() {
83            let s = resp.status();
84            // 401 here usually means the older token lacks user:read:follows.
85            if s == reqwest::StatusCode::UNAUTHORIZED {
86                bail!(
87                    "Twitch rejected the request ({s}). Reconnect your Twitch account to grant \
88                     the 'follows' permission."
89                );
90            }
91            bail!("Twitch followed channels failed: {s}");
92        }
93        let v: serde_json::Value = resp.json().await?;
94        for item in v["data"].as_array().into_iter().flatten() {
95            let login = item["broadcaster_login"].as_str().unwrap_or("");
96            if login.is_empty() {
97                continue;
98            }
99            let name = item["broadcaster_name"].as_str().unwrap_or(login);
100            let id = item["broadcaster_id"].as_str().unwrap_or("");
101            let followed = item["followed_at"]
102                .as_str()
103                .unwrap_or("")
104                .split('T')
105                .next()
106                .unwrap_or("");
107            out.push(ImportCandidate {
108                platform: Platform::Twitch,
109                name: name.to_string(),
110                identity: login.to_lowercase(),
111                id: id.to_string(),
112                url: format!("https://twitch.tv/{login}"),
113                detail: if followed.is_empty() {
114                    String::new()
115                } else {
116                    format!("followed {followed}")
117                },
118            });
119        }
120        cursor = v["pagination"]["cursor"]
121            .as_str()
122            .filter(|s| !s.is_empty())
123            .map(str::to_string);
124        if cursor.is_none() {
125            truncated = false;
126            break;
127        }
128    }
129    if truncated {
130        warn!("Twitch followed list exceeded {MAX_PAGES} pages; import list may be incomplete.");
131    }
132    out.sort_by(|a, b| a.name.to_lowercase().cmp(&b.name.to_lowercase()));
133    Ok(out)
134}

136/// The connected Google account's YouTube subscriptions, sorted by name.
137pub async fn youtube_subscriptions(http: &Client, store: &Store) -> Result<Vec<ImportCandidate>> {
138    let token = google_oauth::valid_token(http, store)
139        .await
140        .ok_or_else(|| anyhow::anyhow!("Connect your YouTube (Google) account in Settings first."))?;
141
142    let mut out = Vec::new();
143    let mut page: Option<String> = None;
144    let mut truncated = true;
145    for _ in 0..MAX_PAGES {
146        let mut query: Vec<(&str, &str)> = vec![
147            ("part", "snippet"),
148            ("mine", "true"),
149            ("maxResults", "50"),
150            ("order", "alphabetical"),
151        ];
152        if let Some(p) = &page {
153            query.push(("pageToken", p.as_str()));
154        }
155        let resp = http
156            .get("https://www.googleapis.com/youtube/v3/subscriptions")
157            .bearer_auth(&token)
158            .query(&query)
159            .send()
160            .await?;
161        if !resp.status().is_success() {
162            let s = resp.status();
163            let body = resp.text().await.unwrap_or_default();
164            bail!("YouTube subscriptions failed: {s} — {body}");
165        }
166        let v: serde_json::Value = resp.json().await?;
167        for item in v["items"].as_array().into_iter().flatten() {
168            let snip = &item["snippet"];
169            let channel_id = snip["resourceId"]["channelId"].as_str().unwrap_or("");
170            if channel_id.is_empty() {
171                continue;
172            }
173            let title = snip["title"].as_str().unwrap_or(channel_id);
174            out.push(ImportCandidate {
175                platform: Platform::YouTube,
176                name: title.to_string(),
177                identity: channel_id.to_lowercase(),
178                id: channel_id.to_string(),
179                url: format!("https://www.youtube.com/channel/{channel_id}"),
180                detail: String::new(),
181            });
182        }
183        page = v["nextPageToken"]
184            .as_str()
185            .filter(|s| !s.is_empty())
186            .map(str::to_string);
187        if page.is_none() {
188            truncated = false;
189            break;
190        }
191    }
192    if truncated {
193        warn!("YouTube subscriptions exceeded {MAX_PAGES} pages; import list may be incomplete.");
194    }
195    out.sort_by(|a, b| a.name.to_lowercase().cmp(&b.name.to_lowercase()));
196    Ok(out)
197}

199/// Create one channel container + its first monitor for an imported URL, reusing
200/// the same per-platform defaults a manual "Add stream" would (max-archival
201/// booleans on). `monitor_enabled` is the "Auto" choice (whether the scheduler
202/// auto-records it); `channel_enabled` is the container on/off (a "Disabled"
203/// import sets it false). Returns the new channel id.
204pub fn create_monitor(
205    store: &Store,
206    defaults: &MonitorDefaults,
207    default_out: &str,
208    name: &str,
209    url: &str,
210    monitor_enabled: bool,
211    channel_enabled: bool,
212) -> Result<i64> {
213    let platform = Platform::detect(url);
214    let channel_id = store.create_container(name)?;
215    let monitor = Monitor {
216        id: 0,
217        channel_id,
218        url: url.to_string(),
219        enabled: monitor_enabled,
220        tool: defaults.resolve_tool(platform),
221        detection_method: defaults.resolve_detection(platform),
222        poll_interval_secs: defaults.resolve_poll_interval(platform).max(5),
223        quality: defaults.resolve_quality(platform),
224        output_dir: defaults.resolve_output_dir(platform, default_out),
225        filename_template: defaults.resolve_filename_template(platform),
226        container: defaults.resolve_container(platform),
227        capture_from_start: defaults.resolve_from_start(platform),
228        dual_capture: false,
229        ad_free: false,
230        auth_kind: AuthKind::Inherit,
231        auth_value: String::new(),
232        audio_tracks: "all".into(),
233        subtitle_tracks: "all".into(),
234        chat_log: true,
235        fetch_thumbnail: true,
236        thumbnail_in_toast: false,
237        fetch_chat_assets: true,
238        extra_args: String::new(),
239        max_concurrent: 1,
240        last_checked_at: None,
241        last_state: "idle".into(),
242    };
243    store.insert_monitor(&monitor)?;
244    if !channel_enabled {
245        store.set_channel_enabled(channel_id, false)?;
246    }
247    Ok(channel_id)
248}

250#[cfg(test)]
251mod tests {
252    use super::*;
253
254    #[test]
255    fn create_monitor_sets_flags_and_platform() {
256        let store = Store::open_in_memory().unwrap();
257        let defaults = MonitorDefaults::default();
258
259        // Auto off (monitor disabled), not "Disabled" (channel on) → info-only.
260        let cid = create_monitor(
261            &store,
262            &defaults,
263            "C:/out",
264            "Cool",
265            "https://twitch.tv/cool",
266            false,
267            true,
268        )
269        .unwrap();
270        let rows = store.list_monitors_with_channels().unwrap();
271        let row = rows.iter().find(|r| r.channel.id == cid).unwrap();
272        assert!(!row.monitor.enabled, "Auto off → monitor disabled");
273        assert!(row.channel.enabled, "not 'Disabled' → channel on");
274        assert_eq!(row.monitor.platform(), Platform::Twitch);
275        assert_eq!(row.monitor.audio_tracks, "all"); // max-archival defaults
276
277        // Auto on + "Disabled" → monitor.enabled true but channel.enabled false.
278        let cid2 = create_monitor(
279            &store,
280            &defaults,
281            "C:/out",
282            "Tuber",
283            "https://www.youtube.com/channel/UCabc",
284            true,
285            false,
286        )
287        .unwrap();
288        let rows = store.list_monitors_with_channels().unwrap();
289        let row2 = rows.iter().find(|r| r.channel.id == cid2).unwrap();
290        assert!(row2.monitor.enabled);
291        assert!(!row2.channel.enabled);
292        assert_eq!(row2.monitor.platform(), Platform::YouTube);
293    }
