1//! YouTube WebSub (PubSubHubbub) push detection via the external `yt-websub`
2//! VPS relay.
3//!
4//! streamarchiver runs at home and is not reachable from the internet, so it
5//! can't receive YouTube's hub callbacks directly. The `yt-websub` server (see
6//! `../yt-websub`) lives on a public VPS, subscribes to the hub for our channels,
7//! durably logs every notification, and exposes them over a token-authenticated
8//! HTTPS API we **poll** from home.
9//!
10//! A WebSub notification is *not* an authoritative "is live" signal — YouTube
11//! fires it for uploads, go-lives, and metadata edits, and it's neither perfectly
12//! reliable nor timely. So each event is treated as a **"check this channel now"
13//! trigger**: we send `ManualCommand::Start(monitor_id)`, which runs the existing
14//! liveness check and records *only if actually live* (idempotent). The scheduler
15//! keeps a slow scrape poll on the same monitors as a safety net (analogous to
16//! how `EventSubHelix` pairs push with polling on Twitch).
17//!
18//! Flow each cycle: load enabled WebSub YouTube monitors -> resolve each to its
19//! `UC…` channel id (so incoming events map back) -> reconcile the desired set on
20//! the VPS (`POST /api/channels`) when it changes -> pull new events
21//! (`GET /api/events?after=<cursor>`) -> trigger a liveness check per mapped
22//! event -> advance + persist the cursor and ack the VPS. Idles cheaply when
23//! there are no WebSub monitors or the VPS isn't configured.
24

41const K_URL: &str = "websub_vps_url";
42const K_TOKEN: &str = "websub_token";
43const K_CURSOR: &str = "websub_cursor";
44const K_POLL: &str = "websub_poll_secs";

46const UA: &str = "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/124.0 Safari/537.36";
47/// How long to wait between API polls by default (the VPS is our own; cheap).
48const DEFAULT_POLL_SECS: u64 = 15;
49/// Minimum time between liveness checks for the same (monitor, video_id) pair.
50/// YouTube fires repeated `updated` events for the same video after a stream
51/// ends (metadata edits, final processing), each with a new VPS seq number.
52/// Without a cooldown every one would re-trigger a `ManualCommand::Start`.
53/// A new video_id (genuine new stream or upload) always bypasses this gate.
54const VIDEO_COOLDOWN: Duration = Duration::from_secs(10 * 60);
55/// Events fetched per `/api/events` page; a full page means more are pending.
56const EVENTS_PAGE: usize = 500;
57/// How long to idle when there's nothing to do (no monitors / no VPS config).
58const IDLE: Duration = Duration::from_secs(30);
59
60/// One event from the VPS `/api/events` feed (only the fields we use).
61struct Event {
62    seq: u64,
63    kind: String,
64    channel_id: String,
65    video_id: String,
66}

68/// Run the WebSub poller until shutdown. Idles cheaply when unused.
69pub async fn run(
70    store: Arc<Store>,
71    manual_tx: UnboundedSender<ManualCommand>,
72    shutdown: Arc<AtomicBool>,
73    jobs: crate::events::JobRegistry,
74) {
75    let http = Client::builder()
76        .user_agent(UA)
77        .timeout(Duration::from_secs(20))
78        .build()
79        .expect("building reqwest client");
80
81    // State carried across cycles.
82    let mut uc_cache: HashMap<String, String> = HashMap::new();
83    let mut last_sig: Option<String> = None;
84    let mut cursor: Option<u64> = None;
85    let mut warned_no_config = false;
86    // Tracks when each (monitor_id, video_id) last triggered a liveness check,
87    // so repeated hub pings for the same ended-stream video don't re-check every
88    // poll cycle.  Pruned to a 1-hour horizon each cycle to prevent growth.
89    let mut video_last_checked: HashMap<(i64, String), Instant> = HashMap::new();
90
91    while !shutdown.load(Ordering::SeqCst) {
92        // Toggleable from the Background view; idle-check for re-enable when off.
93        if !store.job_enabled("job_websub_poll") {
94            sleep_cancellable(IDLE, &shutdown).await;
95            continue;
96        }
97        let monitors = load_websub_monitors(&store);
98
99        let Some((base, token)) = config(&store) else {
100            if !monitors.is_empty() && !warned_no_config {
101                debug!(
102                    "websub: have WebSub monitor(s) but no VPS URL/token set; idling \
103                     (set them in Settings -> YouTube WebSub to enable push)"
104                );
105                warned_no_config = true;
106            }
107            sleep_cancellable(IDLE, &shutdown).await;
108            continue;
109        };
110        warned_no_config = false;
111
112        // Resolve every monitored channel to its UC id so events map back.
113        let mut uc_to_monitors: HashMap<String, Vec<i64>> = HashMap::new();
114        let mut resolve_failed = false;
115        for (url, mid) in &monitors {
116            match resolve_uc(&http, url, &mut uc_cache).await {
117                Some(uc) => uc_to_monitors.entry(uc).or_default().push(*mid),
118                None => {
119                    warn!("websub: could not resolve YouTube channel id for {url}");
120                    resolve_failed = true;
121                }
122            }
123        }
124
125        // Reconcile the VPS subscription set only when it changes (incl. -> empty
126        // to unsubscribe when the last WebSub monitor is removed). Skip the
127        // reconcile entirely if any channel failed to resolve this cycle: pushing
128        // a partial set would unsubscribe channels on the VPS over a transient
129        // scrape failure. (Resolutions are cached, so once a channel resolves it
130        // keeps resolving; the VPS persists its current subs meanwhile.)
131        if !resolve_failed {
132            let mut ucs: Vec<String> = uc_to_monitors.keys().cloned().collect();
133            ucs.sort();
134            let sig = ucs.join(",");
135            if last_sig.as_deref() != Some(sig.as_str()) {
136                match post_channels(&http, &base, &token, &ucs).await {
137                    Ok((sub, unsub, active)) => {
138                        info!("websub: reconciled VPS ({active} active; +{sub} -{unsub})");
139                        last_sig = Some(sig);
140                    }
141                    Err(e) => warn!("websub: reconcile failed: {e:#}"),
142                }
143            }
144        }
145
146        if uc_to_monitors.is_empty() {
147            sleep_cancellable(IDLE, &shutdown).await;
148            continue;
149        }
150
151        // First run with no stored cursor: jump to the current max_seq so we don't
152        // replay the VPS's whole backlog (old uploads) on startup.
153        if cursor.is_none() {
154            cursor = match load_cursor(&store) {
155                Some(c) => Some(c),
156                None => match health(&http, &base, &token).await {
157                    Ok(max_seq) => {
158                        save_cursor(&store, max_seq);
159                        debug!("websub: starting at cursor {max_seq} (skipping backlog)");
160                        Some(max_seq)
161                    }
162                    Err(e) => {
163                        warn!("websub: health failed: {e:#}");
164                        None
165                    }
166                },
167            };
168            if cursor.is_none() {
169                sleep_cancellable(IDLE, &shutdown).await;
170                continue;
171            }
172        }
173
174        // Drain new events a page at a time until caught up, so a backlog larger
175        // than one page (e.g. after downtime) is processed promptly rather than
176        // left for later — and never skipped.
177        loop {
178            let after = cursor.unwrap_or(0);
179            let (events, max_seq) = match poll_events(&http, &base, &token, after).await {
180                Ok(r) => r,
181                Err(e) => {
182                    warn!("websub: poll failed: {e:#}");
183                    break;
184                }
185            };
186            let full_page = events.len() >= EVENTS_PAGE;
187            // Advance the cursor ONLY to the highest event actually received (never
188            // to the server's max_seq) so a >1-page backlog isn't skipped. Dedupe
189            // the monitors to check: YouTube fires new + updated and delivery is
190            // at-least-once, so one channel can recur within a page — a single
191            // liveness check per monitor is enough.
192            //
193            // Additionally gate on VIDEO_COOLDOWN: YouTube keeps firing `updated`
194            // events for the same video_id after a stream ends (metadata edits,
195            // final processing), each with a fresh VPS seq.  We suppress repeat
196            // checks for the same (monitor, video_id) within the cooldown window.
197            // A genuinely new video_id (new stream or upload) always passes through.
198            let now = Instant::now();
199            let mut to_check: HashSet<i64> = HashSet::new();
200            let mut newcur = after;
201            for e in &events {
202                newcur = newcur.max(e.seq);
203                if e.kind == "deleted" {
204                    continue;
205                }
206                if let Some(mons) = uc_to_monitors.get(&e.channel_id) {
207                    for &mid in mons {
208                        let key = (mid, e.video_id.clone());
209                        let cooled_down = video_last_checked
210                            .get(&key)
211                            .map_or(true, |&t| now.duration_since(t) >= VIDEO_COOLDOWN);
212                        if cooled_down {
213                            if to_check.insert(mid) {
214                                info!(
215                                    "websub: {} (video {}) -> check monitor {mid}",
216                                    e.kind, e.video_id
217                                );
218                            }
219                            video_last_checked.insert(key, now);
220                        } else {
221                            debug!(
222                                "websub: {} (video {}) suppressed for monitor {mid} (cooldown)",
223                                e.kind, e.video_id
224                            );
225                        }
226                    }
227                }
228            }
229            // Prune entries older than 1 hour to keep the map bounded.
230            video_last_checked
231                .retain(|_, &mut t| now.duration_since(t) < Duration::from_secs(3600));
232            for mid in &to_check {
233                let _ = manual_tx.send(ManualCommand::Start { id: *mid, notify_offline: false });
234            }
235            if newcur != after {
236                cursor = Some(newcur);
237                save_cursor(&store, newcur);
238                // Best-effort: let the VPS compact acked events.
239                if let Err(e) = ack(&http, &base, &token, newcur).await {
240                    debug!("websub: ack failed: {e:#}");
241                }
242            }
243            // Stop draining when caught up (short page), on no progress (defensive
244            // against a stuck cursor), or on shutdown; keep going while a full page
245            // signals more pending.
246            if !full_page || newcur == after || shutdown.load(Ordering::SeqCst) {
247                if max_seq > newcur {
248                    debug!(
249                        "websub: {} event(s) still pending (cursor {newcur}/{max_seq})",
250                        max_seq - newcur
251                    );
252                }
253                break;
254            }
255        }
256
257        let poll = poll_secs(&store);
258        crate::events::mark_job(&jobs, "YouTube WebSub poll", poll as i64);
259        sleep_cancellable(Duration::from_secs(poll), &shutdown).await;
260    }
261}

263/// YouTube monitors using the WebSub or WebSubOnly method, as `(channel_url, monitor_id)`.
264/// Includes disabled monitors so we still receive push notifications and can show live status
265/// in the UI even for channels with Auto off.
266fn load_websub_monitors(store: &Store) -> Vec<(String, i64)> {
267    let mut out = Vec::new();
268    if let Ok(rows) = store.list_monitors_with_channels() {
269        for row in rows {
270            if row.monitor.platform() == Platform::YouTube
271                && matches!(
272                    row.monitor.detection_method,
273                    DetectionMethod::WebSub | DetectionMethod::WebSubOnly
274                )
275            {
276                out.push((row.monitor.url.clone(), row.monitor.id));
277            }
278        }
279    }
280    out
281}

283/// The VPS base URL (trailing slash trimmed) + bearer token, if both are set.
284fn config(store: &Store) -> Option<(String, String)> {
285    let url = store.get_setting(K_URL).ok().flatten()?;
286    let token = store.get_setting(K_TOKEN).ok().flatten()?;
287    let url = url.trim().trim_end_matches('/').to_string();
288    let token = token.trim().to_string();
289    if url.is_empty() || token.is_empty() {
290        None
291    } else {
292        Some((url, token))
293    }
294}
295
296fn poll_secs(store: &Store) -> u64 {
297    store
298        .get_setting(K_POLL)
299        .ok()
300        .flatten()
301        .and_then(|s| s.trim().parse::<u64>().ok())
302        .unwrap_or(DEFAULT_POLL_SECS)
303        .max(5)
304}
305
306fn load_cursor(store: &Store) -> Option<u64> {
307    store
308        .get_setting(K_CURSOR)
309        .ok()
310        .flatten()
311        .and_then(|s| s.trim().parse::<u64>().ok())
312}
313
314fn save_cursor(store: &Store, cursor: u64) {
315    let _ = store.set_setting(K_CURSOR, &cursor.to_string());
316}

318/// `POST /api/channels` -> `(subscribed, unsubscribed, active)`.
319async fn post_channels(
320    http: &Client,
321    base: &str,
322    token: &str,
323    channels: &[String],
324) -> Result<(u64, u64, u64)> {
325    let resp = http
326        .post(format!("{base}/api/channels"))
327        .bearer_auth(token)
328        .json(&json!({ "channels": channels }))
329        .send()
330        .await?;
331    if !resp.status().is_success() {
332        bail!("channels {}", resp.status());
333    }
334    let v: Value = resp.json().await.unwrap_or_default();
335    Ok((
336        v["subscribed"].as_u64().unwrap_or(0),
337        v["unsubscribed"].as_u64().unwrap_or(0),
338        v["active"].as_u64().unwrap_or(0),
339    ))
340}

356/// `GET /api/events?after=<cursor>` -> `(events, max_seq)`.
357async fn poll_events(
358    http: &Client,
359    base: &str,
360    token: &str,
361    after: u64,
362) -> Result<(Vec<Event>, u64)> {
363    let resp = http
364        .get(format!("{base}/api/events"))
365        .query(&[("after", after.to_string()), ("max", EVENTS_PAGE.to_string())])
366        .bearer_auth(token)
367        .send()
368        .await?;
369    if !resp.status().is_success() {
370        bail!("events {}", resp.status());
371    }
372    let v: Value = resp.json().await?;
373    let max_seq = v["max_seq"].as_u64().unwrap_or(after);
374    let mut events = Vec::new();
375    if let Some(arr) = v["events"].as_array() {
376        for e in arr {
377            events.push(Event {
378                seq: e["seq"].as_u64().unwrap_or(0),
379                kind: e["kind"].as_str().unwrap_or_default().to_string(),
380                channel_id: e["channel_id"].as_str().unwrap_or_default().to_string(),
381                video_id: e["video_id"].as_str().unwrap_or_default().to_string(),
382            });
383        }
384    }
385    Ok((events, max_seq))
386}

402/// Resolve a YouTube channel URL to its `UC…` id (cached). Tries the URL itself
403/// (`/channel/UC…`) first, then scrapes the channel page (no API key needed).
404/// Resolve a YouTube channel URL to its `UC…` id (from the URL, else by scraping
405/// the channel page). Public wrapper for one-off callers (e.g. manual asset refetch)
406/// that don't keep a resolution cache.
407pub(crate) async fn resolve_channel_uc(http: &Client, url: &str) -> Option<String> {
408    let mut cache = HashMap::new();
409    resolve_uc(http, url, &mut cache).await
410}
411
412async fn resolve_uc(
413    http: &Client,
414    url: &str,
415    cache: &mut HashMap<String, String>,
416) -> Option<String> {
417    if let Some(uc) = cache.get(url) {
418        return Some(uc.clone());
419    }
420    if let Some(uc) = uc_from_url(url) {
421        cache.insert(url.to_string(), uc.clone());
422        return Some(uc);
423    }
424    if let Some(uc) = scrape_uc(http, url).await {
425        cache.insert(url.to_string(), uc.clone());
426        return Some(uc);
427    }
428    None
429}

431/// Fetch the channel page and extract its `UC…` id from the HTML.
432async fn scrape_uc(http: &Client, url: &str) -> Option<String> {
433    // Fetch the channel page itself, not its `/live` redirect target.
434    let page = url.trim().trim_end_matches('/');
435    let page = page.strip_suffix("/live").unwrap_or(page);
436    let resp = http
437        .get(page)
438        .header("Accept-Language", "en-US,en;q=0.9")
439        // Bypass the EU consent interstitial that otherwise replaces the page.
440        .header("Cookie", "CONSENT=YES+1; SOCS=CAI")
441        .send()
442        .await
443        .ok()?;
444    let body = resp.text().await.ok()?;
445    find_uc(&body)
446}

448/// True for characters that can appear in a `UC…` channel id (URL-safe base64).
449fn is_uc_char(c: char) -> bool {
450    c.is_ascii_alphanumeric() || c == '-' || c == '_'
451}
452
453/// A valid YouTube channel id is `UC` + 22 URL-safe chars (24 total).
454fn valid_uc(id: &str) -> bool {
455    id.len() == 24 && id.starts_with("UC") && id.chars().all(is_uc_char)
456}
457
458/// Extract a `UC…` id directly from a `/channel/UC…` URL.
459fn uc_from_url(url: &str) -> Option<String> {
460    let pos = url.find("/channel/")?;
461    let rest = &url[pos + "/channel/".len()..];
462    let id: String = rest.chars().take_while(|c| is_uc_char(*c)).collect();
463    valid_uc(&id).then_some(id)
464}
465
466/// Find the channel's own `UC…` id in page HTML. Prefers the channel-identifying
467/// markers (`externalId`, canonical `/channel/`) over a generic `channelId`.
468fn find_uc(html: &str) -> Option<String> {
469    for marker in ["\"externalId\":\"", "/channel/", "\"channelId\":\""] {
470        let mut from = 0;
471        while let Some(rel) = html[from..].find(marker) {
472            let start = from + rel + marker.len();
473            let id: String = html[start..]
474                .chars()
475                .take_while(|c| is_uc_char(*c))
476                .collect();
477            if valid_uc(&id) {
478                return Some(id);
479            }
480            from = start;
481        }
482    }
483    None
484}

490    #[test]
491    fn uc_from_channel_url() {
492        assert_eq!(
493            uc_from_url("https://www.youtube.com/channel/UCaaaaaaaaaaaaaaaaaaaaaa/live").as_deref(),
494            Some("UCaaaaaaaaaaaaaaaaaaaaaa")
495        );
496        assert_eq!(
497            uc_from_url("https://www.youtube.com/channel/UC_x5XG1OV2P6uZZ5FSM9Ttw").as_deref(),
498            Some("UC_x5XG1OV2P6uZZ5FSM9Ttw")
499        );
500        // Handles aren't UC ids.
501        assert_eq!(uc_from_url("https://www.youtube.com/@LofiGirl/live"), None);
502        // Too short to be a real id.
503        assert_eq!(uc_from_url("https://www.youtube.com/channel/UCshort"), None);
504    }
505
506    #[test]
507    fn finds_uc_in_html() {
508        let html = r#"...<link rel="canonical" href="https://www.youtube.com/channel/UC_x5XG1OV2P6uZZ5FSM9Ttw">...,"externalId":"UCaaaaaaaaaaaaaaaaaaaaaa",..."#;
509        // externalId is preferred (the channel's own id).
510        assert_eq!(find_uc(html).as_deref(), Some("UCaaaaaaaaaaaaaaaaaaaaaa"));
511
512        let html2 = r#"...href="/channel/UCbbbbbbbbbbbbbbbbbbbbbb">..."#;
513        assert_eq!(find_uc(html2).as_deref(), Some("UCbbbbbbbbbbbbbbbbbbbbbb"));
514
515        assert_eq!(find_uc("no channel id here"), None);
516    }
517
518    #[test]
519    fn valid_uc_rules() {
520        assert!(valid_uc("UC_x5XG1OV2P6uZZ5FSM9Ttw"));
521        assert!(!valid_uc("UC")); // too short
522        assert!(!valid_uc("XXaaaaaaaaaaaaaaaaaaaaaa")); // wrong prefix
523        assert!(!valid_uc("UCaaaaaaaaaaaaaaaaaaaaaa!")); // 25 chars / bad char
524    }
