49fn main() -> Result<()> {
50    let _tracing_guard = init_tracing();
51
52    // Diagnostics that run and exit (also handy for scripting/headless use).
53    let args: Vec<String> = std::env::args().collect();
54    if let Some(pos) = args.iter().position(|a| a == "--probe") {
55        return run_probe(args.get(pos + 1).cloned().unwrap_or_default());
56    }
57    if let Some(pos) = args.iter().position(|a| a == "--add") {
58        return run_add(&args, pos);
59    }
60    if args.iter().any(|a| a == "--list") {
61        return run_list();
62    }
63    if args.iter().any(|a| a == "--recordings") {
64        return run_recordings();
65    }
66    if args.iter().any(|a| a == "--twitch-login") {
67        return run_twitch_login();
68    }
69    if let Some(pos) = args.iter().position(|a| a == "--set-setting") {
70        let key = args.get(pos + 1).cloned().unwrap_or_default();
71        let value = args.get(pos + 2).cloned().unwrap_or_default();
72        let store = Store::open(&app_paths::db_path())?;
73        store.set_setting(&key, &value)?;
74        println!("set {key}");
75        return Ok(());
76    }
77    if let Some(pos) = args.iter().position(|a| a == "--capture-test") {
78        return run_capture_test(&args, pos);
79    }
80    if let Some(pos) = args.iter().position(|a| a == "--run-for") {
81        let secs: u64 = args.get(pos + 1).and_then(|s| s.parse().ok()).unwrap_or(30);
82        return run_headless(secs);
83    }
84    if let Some(pos) = args.iter().position(|a| a == "--manual-test") {
85        let id: i64 = args.get(pos + 1).and_then(|s| s.parse().ok()).unwrap_or(1);
86        let secs: u64 = args.get(pos + 2).and_then(|s| s.parse().ok()).unwrap_or(12);
87        return run_manual_test(id, secs);
88    }
89
90    // Single-instance guard: hold the loopback bind for the process lifetime.
91    let _instance_guard = match platform::acquire_single_instance() {
92        Some(guard) => guard,
93        None => {
94            info!("another StreamArchiver instance is already running; exiting");
95            return Ok(());
96        }
97    };
98
99    // GUI path only (the CLI diagnostic sub-commands above all returned already):
100    // turn a process-killing panic into a visible native dialog before the runtime
101    // aborts, so startup panics (store/core/tray/font setup) aren't silent crashes.
102    watchdog::install_panic_dialog();
103
104    let store = Store::open(&app_paths::db_path()).context("opening data store")?;
105
106    // Load the optional custom crash/freeze dialog icon (set in Settings → Diagnostics).
107    let dialog_icon_path = store
108        .get_setting(models::K_DIALOG_ICON)
109        .ok()
110        .flatten()
111        .filter(|s| !s.is_empty())
112        .map(std::path::PathBuf::from);
113    watchdog::set_dialog_icon(dialog_icon_path);
114
115    // Crash recovery for on-demand downloads: any left mid-flight is stale.
116    // (In-flight live recordings are handled in `core.start()` →
117    // `Supervisor::resume_inflight`, which resumes SABR-resumable captures and
118    // orphans the rest, then sweeps stale `.cache\` working files.)
119    match store.mark_orphaned_videos(models::now_unix()) {
120        Ok(n) if n > 0 => info!("marked {n} orphaned video download(s) from a previous run"),
121        Ok(_) => {}
122        Err(e) => tracing::warn!("video orphan recovery failed: {e:#}"),
123    }
124    let core = AppCore::new(Arc::new(store)).context("starting core runtime")?;
125    core.start(); // launch the background scheduler + download supervisor
126
127    // `--hidden` (used by the autostart entry) launches straight to the tray.
128    let start_hidden = std::env::args().any(|a| a == "--hidden");
129    info!(start_hidden, "core started; launching UI");
130
131    let (rgba, w, h) = platform::app_icon_rgba();
132    let native_options = eframe::NativeOptions {
133        // Persist egui's Memory to disk so resized column widths, window
134        // positions, and other UI state survive restarts.
135        persistence_path: Some(crate::app_paths::data_dir().join("egui_state.ron")),
136        viewport: egui::ViewportBuilder::default()
137            .with_title(concat!(
138                "StreamArchiver v",
139                env!("APP_VERSION"),
140                " (",
141                env!("GIT_HASH"),
142                ")"
143            ))
144            .with_inner_size([960.0, 600.0])
145            .with_min_inner_size([680.0, 420.0])
146            .with_visible(!start_hidden)
147            .with_icon(egui::IconData {
148                rgba,
149                width: w,
150                height: h,
151            }),
152        ..Default::default()
153    };
154
155    // UI-freeze watchdog: the app stamps a heartbeat each frame; this background
156    // thread pops a native dialog (off the UI thread) if the heartbeat goes stale
157    // while the UI is meant to be rendering, so a hard GUI hang surfaces as an
158    // error instead of a silent "Not Responding" freeze. Started right before the
159    // UI loop so the gap to the first frame can't trip the threshold.
160    let heartbeat = watchdog::Heartbeat::new();
161    watchdog::start_watchdog(
162        heartbeat.clone(),
163        std::time::Duration::from_secs(10),
164        false, // inform-only: downloads are detached, so don't auto-kill the UI
165    );
166
167    let core_for_app = core.clone();
168    eframe::run_native(
169        "StreamArchiver",
170        native_options,
171        Box::new(move |cc| {
172            // Add OS CJK/Unicode fallback fonts so non-Latin channel names (e.g.
173            // Japanese VTuber names, fullwidth 【】) render instead of tofu boxes.
174            fonts::install_unicode_fonts(&cc.egui_ctx);
175            let (tray, ui_rx) = build_tray(cc.egui_ctx.clone())
176                .map_err(|e| -> Box<dyn std::error::Error + Send + Sync> { e.into() })?;
177            Ok(Box::new(ui::StreamArchiverApp::new(
178                core_for_app,
179                tray,
180                ui_rx,
181                heartbeat,
182            )))
183        }),
184    )
185    .map_err(|e| anyhow::anyhow!("eframe failed: {e}"))?;
186
187    // The UI loop has exited (Quit). By default we DETACH: leave the tool process
188    // trees running so the app can restart/rebuild without stopping downloads (the
189    // next launch re-attaches). Only stop them when the user asked to (the
190    // "Quit & stop recordings" tray item or the stop_downloads_on_quit setting).
191    core.shutdown_on_exit();
192
193    Ok(())
194}

196/// Create the system tray icon + menu and a background thread that forwards
197/// menu events to the UI (waking the reactive egui loop via `request_repaint`).
198fn build_tray(ctx: egui::Context) -> Result<(TrayIcon, Receiver<UiCommand>)> {
199    let menu = Menu::new();
200    let open_item = MenuItem::new("Open StreamArchiver", true, None);
201    // Default Quit detaches (downloads keep running); the second item force-stops.
202    let quit_item = MenuItem::new("Quit (keep recording)", true, None);
203    let quit_stop_item = MenuItem::new("Quit & stop recordings", true, None);
204    menu.append(&open_item)?;
205    menu.append(&PredefinedMenuItem::separator())?;
206    menu.append(&quit_item)?;
207    menu.append(&quit_stop_item)?;
208
209    let tray = tray_icon::TrayIconBuilder::new()
210        .with_menu(Box::new(menu))
211        .with_tooltip(concat!(
212            "StreamArchiver v",
213            env!("APP_VERSION"),
214            " (",
215            env!("GIT_HASH"),
216            ")"
217        ))
218        .with_icon(platform::tray_icon_image()?)
219        .build()?;
220
221    let (tx, rx) = std::sync::mpsc::channel::<UiCommand>();
222    let open_id = open_item.id().clone();
223    let quit_id = quit_item.id().clone();
224    let quit_stop_id = quit_stop_item.id().clone();
225    let menu_rx = MenuEvent::receiver().clone();
226
227    std::thread::Builder::new()
228        .name("tray-events".into())
229        .spawn(move || {
230            // Blocks until a tray menu event arrives, so it is idle (~0 CPU)
231            // when nothing is happening.
232            while let Ok(event) = menu_rx.recv() {
233                let cmd = if event.id == open_id {
234                    Some(UiCommand::ShowWindow)
235                } else if event.id == quit_id {
236                    Some(UiCommand::Quit)
237                } else if event.id == quit_stop_id {
238                    Some(UiCommand::QuitAndStop)
239                } else {
240                    None
241                };
242                if let Some(cmd) = cmd {
243                    if tx.send(cmd).is_err() {
244                        break;
245                    }
246                    ctx.request_repaint();
247                }
248            }
249        })
250        .context("spawning tray event thread")?;
251
252    // Keep the menu item handles alive for the life of the process so their ids
253    // remain valid for event matching (the tray owns the Menu itself).
254    std::mem::forget(open_item);
255    std::mem::forget(quit_item);
256
257    Ok((tray, rx))
258}

260/// One-shot detection for diagnostics. Uses scrape/probe (no credentials),
261/// except Twitch which falls through to the generic streamlink probe.
262fn run_probe(url: String) -> Result<()> {
263    if url.is_empty() {
264        anyhow::bail!("usage: streamarchiver --probe <url>");
265    }
266    let store = Arc::new(Store::open(&app_paths::db_path()).context("opening data store")?);
267    let rt = tokio::runtime::Runtime::new()?;
268    let (events_tx, _) = tokio::sync::broadcast::channel(16);
269    let ctx = DetectContext::new(store, events_tx);
270    let platform = Platform::detect(&url);
271    let item = DetectItem {
272        monitor_id: 0,
273        url: url.clone(),
274        platform,
275    };
276    let outcome = rt.block_on(ctx.detect_scrape(&item));
277    println!(
278        "url={url}\nplatform={platform:?}\nlive={}\nerror={}\ndetail={}",
279        outcome.live, outcome.error, outcome.detail
280    );
281    Ok(())
282}

284/// `--add <name> <url> [method] [tool]` inserts a channel + one monitor.
285/// Output dir honors the `STREAMARCHIVER_OUT` env var (else the default).
286fn run_add(args: &[String], pos: usize) -> Result<()> {
287    let name = args.get(pos + 1).cloned().unwrap_or_default();
288    let url = args.get(pos + 2).cloned().unwrap_or_default();
289    if name.is_empty() || url.is_empty() {
290        anyhow::bail!("usage: streamarchiver --add <name> <url> [detection_method] [tool]");
291    }
292    let platform = Platform::detect(&url);
293    let method = args
294        .get(pos + 3)
295        .map(|s| models::DetectionMethod::parse(s))
296        .unwrap_or_else(|| platform.default_detection());
297    let tool = args
298        .get(pos + 4)
299        .map(|s| models::Tool::parse(s))
300        .unwrap_or_else(|| platform.default_tool());
301    let output_dir = std::env::var("STREAMARCHIVER_OUT").unwrap_or_else(|_| {
302        app_paths::default_output_dir()
303            .to_string_lossy()
304            .to_string()
305    });
306
307    let store = Store::open(&app_paths::db_path()).context("opening data store")?;
308    let channel_id = store.upsert_channel(&name, &url, platform)?;
309    let monitor = models::Monitor {
310        id: 0,
311        channel_id,
312        url: url.clone(),
313        enabled: true,
314        tool,
315        detection_method: method,
316        poll_interval_secs: 30,
317        quality: "best".into(),
318        output_dir,
319        filename_template: "{name}_{date}_{time}".into(),
320        container: models::Container::Mkv,
321        capture_from_start: std::env::var("STREAMARCHIVER_FROM_START").as_deref() != Ok("0"),
322        dual_capture: false,
323        ad_free: false,
324        auth_kind: models::AuthKind::Inherit,
325        auth_value: String::new(),
326        audio_tracks: "all".into(),
327        subtitle_tracks: "all".into(),
328        chat_log: true,
329        fetch_thumbnail: true,
330        thumbnail_in_toast: false,
331        fetch_chat_assets: true,
332        extra_args: String::new(),
333        max_concurrent: 1,
334        last_checked_at: None,
335        last_state: "idle".into(),
336    };
337    let monitor_id = store.insert_monitor(&monitor)?;
338    println!("added monitor {monitor_id} (channel {channel_id}, {platform:?}, {method:?})");
339    Ok(())
340}

359/// `--capture-test <tool> <url> <secs>` records for a fixed time, kills the tree
360/// (taskkill /T), and remuxes — exercising the real capture lifecycle end-to-end.
361fn run_capture_test(args: &[String], pos: usize) -> Result<()> {
362    use std::process::Stdio;
363    let tool = models::Tool::parse(&args.get(pos + 1).cloned().unwrap_or_default());
364    let url = args.get(pos + 2).cloned().unwrap_or_default();
365    let secs: u64 = args.get(pos + 3).and_then(|s| s.parse().ok()).unwrap_or(15);
366    if url.is_empty() {
367        anyhow::bail!("usage: streamarchiver --capture-test <tool> <url> <secs>");
368    }
369    let out_dir = std::env::var("STREAMARCHIVER_OUT").unwrap_or_else(|_| {
370        app_paths::default_output_dir()
371            .to_string_lossy()
372            .to_string()
373    });
374    let row = models::MonitorWithChannel {
375        channel: models::Channel {
376            id: 0,
377            name: "captest".into(),
378            url: url.clone(),
379            platform: Platform::detect(&url),
380            created_at: 0,
381            color: String::new(),
382            preferred_platform: None,
383            enabled: true,
384        },
385        monitor: models::Monitor {
386            id: 0,
387            channel_id: 0,
388            url: url.clone(),
389            enabled: true,
390            tool,
391            detection_method: models::DetectionMethod::GenericProbe,
392            poll_interval_secs: 60,
393            quality: "best".into(),
394            output_dir: out_dir,
395            filename_template: "captest_{time}".into(),
396            container: models::Container::Mkv,
397            capture_from_start: false,
398            dual_capture: false,
399            ad_free: false,
400            auth_kind: models::AuthKind::Inherit,
401            auth_value: String::new(),
402            audio_tracks: String::new(),
403            subtitle_tracks: String::new(),
404            chat_log: false,
405            fetch_thumbnail: false,
406            thumbnail_in_toast: false,
407            fetch_chat_assets: false,
408            extra_args: String::new(),
409            max_concurrent: 1,
410            last_checked_at: None,
411            last_state: "idle".into(),
412        },
413        last_recording_started: None,
414        last_recording_ended: None,
415        last_recording_status: None,
416        last_recording_went_live: None,
417        last_recording_went_live_approx: false,
418        last_recording_lost_secs: None,
419        last_recording_ad_count: 0,
420        last_recording_ad_secs: 0,
421        last_recording_meta_changes: 0,
422        last_recording_title: String::new(),
423        last_recording_category: String::new(),
424        last_recording_log: String::new(),
425        ad_free_sub: None,
426        recording_count: 0,
427        next_stream_at: None,
428        next_stream_title: String::new(),
429    };
430    let plan = downloader::build_plan(
431        &row,
432        models::now_unix(),
433        &downloader::AuthSource::None,
434        &[],
435        None,
436        "",
437        None,
438        0,
439        &downloader::YtDlpBins::default(),
440    );
441    println!("plan: {} {:?}", plan.program, plan.args);
442
443    let rt = tokio::runtime::Runtime::new()?;
444    rt.block_on(async move {
445        if let Some(parent) = plan.capture_path.parent() {
446            let _ = tokio::fs::create_dir_all(parent).await;
447        }
448        let mut cmd = tokio::process::Command::new(&plan.program);
449        cmd.args(&plan.args)
450            .stdin(Stdio::null())
451            .stdout(Stdio::null())
452            .stderr(Stdio::null());
453        let mut child = cmd.spawn().context("spawn tool")?;
454        let pid = child.id().unwrap_or(0);
455        println!(
456            "spawned {} pid={pid}; recording {secs}s -> {}",
457            plan.program,
458            plan.capture_path.display()
459        );
460        tokio::time::sleep(std::time::Duration::from_secs(secs)).await;
461
462        platform::kill_process_tree(pid);
463        let _ = tokio::time::timeout(std::time::Duration::from_secs(5), child.wait()).await;
464        tokio::time::sleep(std::time::Duration::from_secs(1)).await;
465
466        let ts_len = tokio::fs::metadata(&plan.capture_path)
467            .await
468            .map(|m| m.len())
469            .unwrap_or(0);
470        println!(
471            "captured .ts: {} KB (survives the hard kill)",
472            ts_len / 1024
473        );
474        if plan.remux_to_mkv && ts_len > 0 {
475            match downloader::remux_ts_to_mkv(&plan.capture_path, &plan.final_path, None, &Default::default()).await {
476                Ok(()) => {
477                    let mkv = tokio::fs::metadata(&plan.final_path)
478                        .await
479                        .map(|m| m.len())
480                        .unwrap_or(0);
481                    println!(
482                        "remuxed -> {} ({} KB)",
483                        plan.final_path.display(),
484                        mkv / 1024
485                    );
486                }
487                Err(e) => println!("remux failed: {e:#}"),
488            }
489        }
490        Ok::<_, anyhow::Error>(())
491    })?;
492    Ok(())
493}

495/// `--run-for <secs>` runs the full core (scheduler + supervisor) headlessly for
496/// a fixed time, then gracefully stops — for testing the real pipeline and as a
497/// no-UI daemon mode.
498fn run_headless(secs: u64) -> Result<()> {
499    let store = Store::open(&app_paths::db_path()).context("opening data store")?;
500    let _ = store.mark_orphaned_recordings(models::now_unix());
501    let _ = store.promote_intact_orphans();
502    let _ = store.mark_orphaned_videos(models::now_unix());
503    let core = AppCore::new(Arc::new(store)).context("starting core runtime")?;
504    core.start();
505    info!("headless: running core for {secs}s");
506    std::thread::sleep(std::time::Duration::from_secs(secs));
507    info!("headless: stopping");
508    core.stop_all_recordings();
509    Ok(())
510}

512/// `--manual-test <monitor_id> <secs>` exercises the on-demand Start/Stop path:
513/// it DISABLES the monitor (so the scheduler won't touch it), then manually
514/// Starts it, records for `<secs>`, and Stops it.
515fn run_manual_test(id: i64, secs: u64) -> Result<()> {
516    use events::ManualCommand;
517    let store = Store::open(&app_paths::db_path()).context("opening data store")?;
518    store.set_monitor_enabled(id, false)?; // prove the scheduler isn't doing it
519    let core = AppCore::new(Arc::new(store)).context("starting core")?;
520    core.start();
521    info!("manual-test: Start({id})");
522    core.manual(ManualCommand::Start { id, notify_offline: true });
523    std::thread::sleep(std::time::Duration::from_secs(secs));
524    info!("manual-test: Stop({id})");
525    core.manual(ManualCommand::Stop(id));
526    std::thread::sleep(std::time::Duration::from_secs(5));
527    core.stop_all_recordings();
528    Ok(())
529}

531/// `--twitch-login` runs the Twitch device-code OAuth flow interactively.
532fn run_twitch_login() -> Result<()> {
533    let store = Store::open(&app_paths::db_path()).context("opening data store")?;
534    let client_id = store.get_setting("twitch_client_id")?.unwrap_or_default();
535    if client_id.is_empty() {
536        anyhow::bail!("set a Twitch Client ID in Settings first");
537    }
538    let rt = tokio::runtime::Runtime::new()?;
539    rt.block_on(async move {
540        let http = reqwest::Client::builder()
541            .timeout(std::time::Duration::from_secs(20))
542            .build()?;
543        let dc = oauth::start_device(&http, &client_id).await?;
544        println!(
545            "\n  Open: {}\n  Enter code: {}\n",
546            dc.verification_uri, dc.user_code
547        );
548        println!("Waiting for authorization…");
549        let tokens = oauth::poll_token(&http, &client_id, &dc).await?;
550        match oauth::fetch_user(&http, &client_id, &tokens.access).await {
551            Ok((login, user_id)) => {
552                oauth::store_tokens(&store, &tokens, &login)?;
553                store.set_setting(oauth::K_USER_ID, &user_id)?;
554                println!("Connected as {login}.");
555            }
556            // Keep the valid tokens (detection only needs the token); the account
557            // lookup failed, so ad-free sub detection stays off until a reconnect.
558            Err(e) => {
559                oauth::store_tokens(&store, &tokens, "")?;
560                println!("Connected (couldn't read account: {e}). Reconnect later to enable ad-free sub detection.");
561            }
562        }
563        Ok::<_, anyhow::Error>(())
564    })?;
565    Ok(())
566}

568/// `--recordings` prints the recent recording log.
569fn run_recordings() -> Result<()> {
570    let store = Store::open(&app_paths::db_path()).context("opening data store")?;
571    for r in store.recent_recordings(50)? {
572        let went = match r.went_live_at {
573            Some(w) => format!("{}{}", if r.went_live_approx { "~" } else { "" }, w),
574            None => "-".into(),
575        };
576        let lost = match r.went_live_at {
577            Some(w) => format!("{}s", (r.started_at - w).max(0)),
578            None => "-".into(),
579        };
580        println!(
581            "rec[{}] monitor={} status={:<10} bytes={:<9} started={} went_live={} lost={} {}",
582            r.id, r.monitor_id, r.status, r.bytes, r.started_at, went, lost, r.output_path
583        );
584    }
585    Ok(())
586}

588/// Initialise tracing: stderr at the env-filter level **and** a daily-rotating
589/// log file under `data_dir()/logs/streamarchiver.YYYY-MM-DD.log` (kept for 7
590/// days via `tracing_appender::rolling::daily`; older files are not auto-pruned
591/// by the crate, so we prune on startup).
592///
593/// Returns the worker-thread guard — drop it only when the process is about to
594/// exit, otherwise buffered log lines may be lost.
595fn init_tracing() -> tracing_appender::non_blocking::WorkerGuard {
596    use tracing_subscriber::{EnvFilter, layer::SubscriberExt, util::SubscriberInitExt};
597
598    let filter = EnvFilter::try_from_default_env()
599        .unwrap_or_else(|_| EnvFilter::new("info,streamarchiver=debug"));
600
601    // ── rotating file layer ───────────────────────────────────────────────────
602    let log_dir = crate::app_paths::logs_dir();
603    prune_old_logs(&log_dir, 7);
604    let file_appender = tracing_appender::rolling::daily(&log_dir, "streamarchiver.log");
605    let (non_blocking, guard) = tracing_appender::non_blocking(file_appender);
606    let file_layer = tracing_subscriber::fmt::layer()
607        .with_ansi(false)
608        .with_target(false)
609        .with_writer(non_blocking);
610
611    // ── stderr layer ─────────────────────────────────────────────────────────
612    let stderr_layer = tracing_subscriber::fmt::layer()
613        .with_target(false)
614        .with_writer(std::io::stderr);
615
616    tracing_subscriber::registry()
617        .with(filter)
618        .with(file_layer)
619        .with(stderr_layer)
620        .init();
621
622    guard
623}
624
625/// Delete log files older than `keep_days` days from `dir`.
626fn prune_old_logs(dir: &std::path::Path, keep_days: u64) {
627    let cutoff = std::time::SystemTime::now()
628        .checked_sub(std::time::Duration::from_secs(keep_days * 86_400))
629        .unwrap_or(std::time::SystemTime::UNIX_EPOCH);
630    let Ok(entries) = std::fs::read_dir(dir) else { return };
631    for entry in entries.flatten() {
632        let path = entry.path();
633        if path.extension().map(|e| e == "log").unwrap_or(false) {
634            if let Ok(meta) = std::fs::metadata(&path) {
635                if meta.modified().map(|m| m < cutoff).unwrap_or(false) {
636                    let _ = std::fs::remove_file(&path);
637                }
638            }
639        }
640    }
641}
