1//! Google OAuth2 **Device Code** flow ("Connect YouTube").
2//!
3//! Mirrors the Twitch device flow in [`crate::oauth`], but for a Google account so
4//! we can read the user's YouTube subscriptions (`youtube.readonly`). Device code
5//! suits a desktop app: the user is shown a short code + URL, authorizes in a
6//! browser, and we poll for the token. Unlike Twitch, Google's device flow needs
7//! both a Client ID **and** a Client Secret (a "TV and Limited Input devices" OAuth
8//! client created in Google Cloud Console); refresh keeps the same refresh token.
9//!
10//! NOTE: live verification needs a Google Cloud OAuth client with the device flow
11//! enabled and the YouTube Data API turned on for the project.
12

24const DEVICE_URL: &str = "https://oauth2.googleapis.com/device/code";
25const TOKEN_URL: &str = "https://oauth2.googleapis.com/token";
26const CHANNELS_URL: &str = "https://www.googleapis.com/youtube/v3/channels";
27
28/// Read-only access to the user's YouTube account (subscriptions, channel info).
29const SCOPE: &str = "https://www.googleapis.com/auth/youtube.readonly";
30
31// Settings keys for the stored Google connection + credentials.
32pub const K_CLIENT_ID: &str = "google_client_id";
33pub const K_CLIENT_SECRET: &str = "google_client_secret";
34pub const K_USER_TOKEN: &str = "google_user_token";
35pub const K_REFRESH: &str = "google_refresh_token";
36pub const K_EXPIRY: &str = "google_token_expiry";
37/// Display identity (the connected account's YouTube channel title), best-effort.
38pub const K_IDENTITY: &str = "google_user_identity";

40/// Request a device code to begin authorization.
41pub async fn start_device(http: &Client, client_id: &str) -> Result<DeviceCode> {
42    let resp = http
43        .post(DEVICE_URL)
44        .form(&[("client_id", client_id), ("scope", SCOPE)])
45        .send()
46        .await?;
47    if !resp.status().is_success() {
48        let s = resp.status();
49        bail!(
50            "device code request failed: {s} {}",
51            resp.text().await.unwrap_or_default()
52        );
53    }
54    let v: Value = resp.json().await?;
55    Ok(DeviceCode {
56        device_code: v["device_code"]
57            .as_str()
58            .context("no device_code")?
59            .to_string(),
60        user_code: v["user_code"].as_str().context("no user_code")?.to_string(),
61        // Google returns `verification_url` (Twitch uses `verification_uri`).
62        verification_uri: v["verification_url"]
63            .as_str()
64            .or_else(|| v["verification_uri"].as_str())
65            .context("no verification_url")?
66            .to_string(),
67        interval: v["interval"].as_u64().unwrap_or(5).max(1),
68        expires_in: v["expires_in"].as_u64().unwrap_or(1800),
69    })
70}

72/// Poll the token endpoint until the user authorizes (or it expires). Google needs
73/// the Client Secret here in addition to the Client ID.
74pub async fn poll_token(
75    http: &Client,
76    client_id: &str,
77    client_secret: &str,
78    dc: &DeviceCode,
79) -> Result<Tokens> {
80    let mut wait = dc.interval;
81    let mut elapsed = 0u64;
82    loop {
83        tokio::time::sleep(Duration::from_secs(wait)).await;
84        elapsed += wait;
85        if elapsed > dc.expires_in {
86            bail!("authorization timed out");
87        }
88        let resp = http
89            .post(TOKEN_URL)
90            .form(&[
91                ("client_id", client_id),
92                ("client_secret", client_secret),
93                ("device_code", dc.device_code.as_str()),
94                ("grant_type", "urn:ietf:params:oauth:grant-type:device_code"),
95            ])
96            .send()
97            .await?;
98        let status = resp.status();
99        let v: Value = resp.json().await.unwrap_or(Value::Null);
100        if status.is_success() {
101            return Ok(Tokens {
102                access: v["access_token"]
103                    .as_str()
104                    .context("no access_token")?
105                    .to_string(),
106                refresh: v["refresh_token"].as_str().unwrap_or_default().to_string(),
107                // Google access tokens last 1 h; default to that if the field is
108                // ever missing so the token isn't treated as instantly expired.
109                expires_in: v["expires_in"].as_u64().unwrap_or(3600),
110            });
111        }
112        // Google signals flow state via the JSON `error` field, not the HTTP code.
113        let err = v["error"].as_str().unwrap_or("");
114        if err == "authorization_pending" {
115            continue;
116        }
117        if err == "slow_down" {
118            wait += 5;
119            continue;
120        }
121        let desc = v["error_description"].as_str().unwrap_or(err);
122        bail!("authorization failed: {status} {desc}");
123    }
124}

126/// Exchange a refresh token for a fresh access token. Google does not rotate the
127/// refresh token, so the caller keeps the existing one.
128pub async fn refresh(
129    http: &Client,
130    client_id: &str,
131    client_secret: &str,
132    refresh_token: &str,
133) -> Result<Tokens> {
134    let resp = http
135        .post(TOKEN_URL)
136        .form(&[
137            ("client_id", client_id),
138            ("client_secret", client_secret),
139            ("refresh_token", refresh_token),
140            ("grant_type", "refresh_token"),
141        ])
142        .send()
143        .await?;
144    if !resp.status().is_success() {
145        let status = resp.status();
146        let body = resp.text().await.unwrap_or_default();
147        bail!("token refresh failed: {status} — {body}");
148    }
149    let v: Value = resp.json().await?;
150    Ok(Tokens {
151        access: v["access_token"]
152            .as_str()
153            .context("no access_token")?
154            .to_string(),
155        // Google omits refresh_token on refresh — keep the existing one.
156        refresh: v["refresh_token"]
157            .as_str()
158            .filter(|s| !s.is_empty())
159            .unwrap_or(refresh_token)
160            .to_string(),
161        expires_in: v["expires_in"].as_u64().unwrap_or(3600),
162    })
163}

165/// Best-effort display name: the connected account's own YouTube channel title.
166/// Returns `None` (and the connection is still kept) when it can't be resolved.
167pub async fn fetch_identity(http: &Client, token: &str) -> Option<String> {
168    let resp = http
169        .get(CHANNELS_URL)
170        .query(&[("part", "snippet"), ("mine", "true")])
171        .bearer_auth(token)
172        .send()
173        .await
174        .ok()?;
175    if !resp.status().is_success() {
176        return None;
177    }
178    let v: Value = resp.json().await.ok()?;
179    v["items"][0]["snippet"]["title"]
180        .as_str()
181        .filter(|s| !s.is_empty())
182        .map(str::to_string)
183}

185/// Persist a freshly obtained set of tokens.
186pub fn store_tokens(store: &Store, tokens: &Tokens) -> Result<()> {
187    store.set_setting(K_USER_TOKEN, &tokens.access)?;
188    if !tokens.refresh.is_empty() {
189        store.set_setting(K_REFRESH, &tokens.refresh)?;
190    } else if store
191        .get_setting(K_REFRESH)
192        .ok()
193        .flatten()
194        .filter(|s| !s.is_empty())
195        .is_none()
196    {
197        // No refresh token now and none stored before — the access token will
198        // expire in ~1 h with no way to renew. Google only re-issues a refresh
199        // token on fresh consent, so a reconnect after revoking access is needed.
200        warn!(
201            "Google returned no refresh token; the YouTube connection will expire in ~1h. \
202             Revoke this app under your Google account's third-party access, then reconnect."
203        );
204    }
205    let expiry = now_unix() + tokens.expires_in as i64;
206    store.set_setting(K_EXPIRY, &expiry.to_string())?;
207    Ok(())
208}

210/// Forget the stored Google connection.
211pub fn disconnect(store: &Store) -> Result<()> {
212    for k in [K_USER_TOKEN, K_REFRESH, K_EXPIRY, K_IDENTITY] {
213        store.set_setting(k, "")?;
214    }
215    Ok(())
216}

218/// The connected account's display identity, if any.
219pub fn connected_identity(store: &Store) -> Option<String> {
220    store
221        .get_setting(K_IDENTITY)
222        .ok()
223        .flatten()
224        .filter(|s| !s.is_empty())
225}

227/// Whether a Google account is connected (a user token is stored).
228pub fn is_connected(store: &Store) -> bool {
229    store
230        .get_setting(K_USER_TOKEN)
231        .ok()
232        .flatten()
233        .is_some_and(|s| !s.is_empty())
234}

236/// Return a *usable* user access token, refreshing it if it's near/past expiry.
237/// `None` when not connected or the token is dead and can't be refreshed.
238pub async fn valid_token(http: &Client, store: &Store) -> Option<String> {
239    let token = store.get_setting(K_USER_TOKEN).ok().flatten()?;
240    if token.is_empty() {
241        return None;
242    }
243    let expiry: i64 = store
244        .get_setting(K_EXPIRY)
245        .ok()
246        .flatten()
247        .and_then(|s| s.parse().ok())
248        .unwrap_or(0);
249    if now_unix() < expiry - 300 {
250        return Some(token);
251    }
252    let refresh_token = store.get_setting(K_REFRESH).ok().flatten().unwrap_or_default();
253    let client_id = store.get_setting(K_CLIENT_ID).ok().flatten().unwrap_or_default();
254    let client_secret = store
255        .get_setting(K_CLIENT_SECRET)
256        .ok()
257        .flatten()
258        .unwrap_or_default();
259    if !refresh_token.is_empty() && !client_id.is_empty() && !client_secret.is_empty() {
260        match refresh(http, &client_id, &client_secret, &refresh_token).await {
261            Ok(t) => {
262                let _ = store_tokens(store, &t);
263                return Some(t.access);
264            }
265            Err(e) => warn!("Google token refresh failed: {e}"),
266        }
267    } else if !refresh_token.is_empty() {
268        warn!(
269            "Google token refresh skipped: add your Google Client ID + Secret in \
270             Settings → YouTube account (Google OAuth)."
271        );
272    }
273    // Return the old token while it's still within its stated lifetime so the
274    // caller can make one more attempt; once fully expired, return None.
275    if now_unix() < expiry {
276        Some(token)
277    } else {
278        None
279    }
