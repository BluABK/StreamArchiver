1//! Anonymous Twitch chat capture over the IRC-over-WebSocket gateway.
2//!
3//! Twitch chat is plain IRCv3 over `wss://irc-ws.chat.twitch.tv`. We log in
4//! anonymously (a `justinfan*` nick — read-only, no token), request the tags +
5//! commands capabilities (for timestamps / display names / colors), JOIN the
6//! channel, and append every chat message to a `.chat.jsonl` sidecar next to the
7//! recording. Uses the already-present `tokio-tungstenite` (no new dependency).
8//!
9//! YouTube chat is handled separately by yt-dlp (`--sub-langs live_chat`), not
10//! here. Kick chat is not yet supported.
11
12use std::path::{Path, PathBuf};
13use std::sync::Arc;
14use std::sync::atomic::{AtomicBool, AtomicU64, Ordering};
15use std::time::Duration;
16
17use futures_util::{SinkExt, StreamExt};
18use serde::Serialize;
19use tokio::io::AsyncWriteExt;
20use tokio::time::timeout;
21use tracing::{debug, info};

23const WS_URL: &str = "wss://irc-ws.chat.twitch.tv:443";
24
25/// Process-wide counter so concurrent anonymous logins get distinct nicks even
26/// when two recordings start in the same second.
27static NICK_SEQ: AtomicU64 = AtomicU64::new(0);
28
29/// Bound the connect + login handshake so a slow/unreachable gateway can't block
30/// the recording's finalize (which joins this task when the capture ends).
31const CONNECT_TIMEOUT: Duration = Duration::from_secs(8);
32
33/// Resolve once `done` or `shutdown` is set, to race against a blocking connect.
34async fn wait_stopped(done: &AtomicBool, shutdown: &AtomicBool) {
35    while !done.load(Ordering::SeqCst) && !shutdown.load(Ordering::SeqCst) {
36        tokio::time::sleep(Duration::from_millis(200)).await;
37    }
38}

40/// One logged chat message (serialized as a JSON line in the sidecar).
41#[derive(Serialize)]
42struct ChatLine<'a> {
43    /// Milliseconds since the epoch (Twitch `tmi-sent-ts` when present).
44    ts: i64,
45    /// Sender's login (lowercase).
46    login: &'a str,
47    /// Display name (falls back to `login` when unset).
48    name: &'a str,
49    /// Message body (the IRC trailing parameter, unescaped).
50    text: &'a str,
51    /// Chat color `#RRGGBB`, omitted when unset.
52    #[serde(skip_serializing_if = "str::is_empty")]
53    color: &'a str,
54    /// Raw `badges` tag (e.g. `subscriber/12,moderator/1`), omitted when empty.
55    #[serde(skip_serializing_if = "str::is_empty")]
56    badges: &'a str,
57    /// Raw IRCv3 `emotes` tag (e.g. `25:0-4,12-16/1902:6-10`) — first-party emote
58    /// ID + inclusive code-point ranges into `text`. Stored verbatim (the value is
59    /// only digits/`:`/`-`/`,`/`/`, so no IRCv3 unescaping applies). Omitted when
60    /// empty; old logs without it simply render emote words as plain text.
61    #[serde(skip_serializing_if = "str::is_empty")]
62    emotes: &'a str,
63}

65/// Capture `url`'s Twitch chat to `path` until `done` (recording ended) or
66/// `shutdown` is set. Best-effort: connection failures are logged and retried
67/// with a short interruptible backoff; this never panics. No-ops for a URL that
68/// isn't a Twitch channel.
69pub async fn log_twitch_chat(
70    url: String,
71    path: PathBuf,
72    done: Arc<AtomicBool>,
73    shutdown: Arc<AtomicBool>,
74) {
75    let Some(login) = crate::detectors::twitch_login(&url) else {
76        return;
77    };
78    info!("chat: logging {login} -> {}", path.display());
79    while !done.load(Ordering::SeqCst) && !shutdown.load(Ordering::SeqCst) {
80        if let Err(e) = session(&login, &path, &done, &shutdown).await {
81            debug!("chat ({login}): {e:#}; reconnecting");
82        }
83        // Interruptible backoff before reconnecting (checks flags every 250ms).
84        for _ in 0..8 {
85            if done.load(Ordering::SeqCst) || shutdown.load(Ordering::SeqCst) {
86                return;
87            }
88            tokio::time::sleep(Duration::from_millis(250)).await;
89        }
90    }
91}

93/// One connection's lifetime: connect, anonymous login, JOIN, then append every
94/// PRIVMSG until a stop flag is set (Ok) or the connection drops (Err).
95async fn session(
96    login: &str,
97    path: &Path,
98    done: &AtomicBool,
99    shutdown: &AtomicBool,
100) -> anyhow::Result<()> {
101    use tokio_tungstenite::tungstenite::Message;
102
103    let seq = NICK_SEQ.fetch_add(1, Ordering::Relaxed);
104    let nick = format!(
105        "justinfan{}",
106        100_000 + (crate::models::now_unix() as u64).wrapping_add(seq) % 9_000_000
107    );
108    // Connect + anonymous, read-only login (request tags+commands for metadata +
109    // PINGs). Bounded by a timeout and raced against the stop flags so a stalled
110    // handshake can't hang the finalize that joins this task.
111    let connect = async {
112        let (mut ws, _) = tokio_tungstenite::connect_async(WS_URL).await?;
113        ws.send(Message::Text(
114            "CAP REQ :twitch.tv/tags twitch.tv/commands".into(),
115        ))
116        .await?;
117        ws.send(Message::Text(format!("NICK {nick}").into())).await?;
118        ws.send(Message::Text(format!("JOIN #{login}").into())).await?;
119        Ok::<_, anyhow::Error>(ws)
120    };
121    let mut ws = tokio::select! {
122        biased;
123        _ = wait_stopped(done, shutdown) => return Ok(()),
124        r = tokio::time::timeout(CONNECT_TIMEOUT, connect) => match r {
125            Ok(Ok(ws)) => ws,
126            Ok(Err(e)) => return Err(e),
127            Err(_) => return Err(anyhow::anyhow!("chat connect/login timed out")),
128        },
129    };
130
131    // Opened lazily on the first message so a stream with no chat (or a recording
132    // that fails immediately) doesn't leave an empty sidecar. Append mode means
133    // reconnects continue the same file rather than truncating it.
134    let mut file: Option<tokio::fs::File> = None;
135
136    loop {
137        if done.load(Ordering::SeqCst) || shutdown.load(Ordering::SeqCst) {
138            return Ok(());
139        }
140        // 1s read timeout so the stop flags are re-checked even on a quiet chat.
141        let msg = match timeout(Duration::from_secs(1), ws.next()).await {
142            Ok(Some(Ok(m))) => m,
143            Ok(Some(Err(e))) => return Err(e.into()),
144            Ok(None) => return Err(anyhow::anyhow!("chat websocket closed")),
145            Err(_) => continue, // read timeout -> re-check flags
146        };
147        match msg {
148            Message::Text(text) => {
149                // A frame can carry several CRLF-separated IRC lines.
150                for line in text.lines() {
151                    let line = line.trim_end_matches('\r');
152                    if line.is_empty() {
153                        continue;
154                    }
155                    // Twitch IRC keepalive: reply so the server doesn't drop us.
156                    if let Some(token) = line.strip_prefix("PING ") {
157                        ws.send(Message::Text(format!("PONG {token}").into())).await?;
158                        continue;
159                    }
160                    if let Some(json) = parse_privmsg(line) {
161                        if file.is_none() {
162                            file = Some(
163                                tokio::fs::OpenOptions::new()
164                                    .create(true)
165                                    .append(true)
166                                    .open(path)
167                                    .await?,
168                            );
169                        }
170                        let f = file.as_mut().unwrap();
171                        f.write_all(json.as_bytes()).await?;
172                        f.write_all(b"\n").await?;
173                    }
174                }
175            }
176            Message::Ping(payload) => {
177                let _ = ws.send(Message::Pong(payload)).await;
178            }
179            Message::Close(_) => return Err(anyhow::anyhow!("chat websocket close frame")),
180            _ => {}
181        }
182    }
183}

185/// Parse a (possibly tag-prefixed) IRC line into a JSON log line, or `None` if it
186/// isn't a chat message (`PRIVMSG`). Tag values keep Twitch's IRCv3 escaping in
187/// the rare cases it applies; the message body is the unescaped trailing param.
188fn parse_privmsg(line: &str) -> Option<String> {
189    // Optional IRCv3 tags: "@k=v;k=v <rest>".
190    let (tags, rest) = if let Some(s) = line.strip_prefix('@') {
191        let sp = s.find(' ')?;
192        (&s[..sp], &s[sp + 1..])
193    } else {
194        ("", line)
195    };
196    // rest = ":login!user@host PRIVMSG #chan :message"
197    let rest = rest.strip_prefix(':')?;
198    let sp = rest.find(' ')?;
199    let prefix = &rest[..sp];
200    let after = &rest[sp + 1..];
201    if !after.starts_with("PRIVMSG ") {
202        return None;
203    }
204    // The message is the trailing parameter, after the first " :".
205    let text = after.find(" :").map(|i| &after[i + 2..]).unwrap_or("");
206    let login = prefix.split('!').next().unwrap_or(prefix);
207
208    let (mut display, mut color, mut badges, mut emotes, mut ts_ms) = ("", "", "", "", 0i64);
209    for kv in tags.split(';') {
210        let mut it = kv.splitn(2, '=');
211        let (k, v) = (it.next().unwrap_or(""), it.next().unwrap_or(""));
212        match k {
213            "display-name" => display = v,
214            "color" => color = v,
215            "badges" => badges = v,
216            "emotes" => emotes = v,
217            "tmi-sent-ts" => ts_ms = v.parse().unwrap_or(0),
218            _ => {}
219        }
220    }
221    if ts_ms == 0 {
222        ts_ms = crate::models::now_unix() * 1000;
223    }
224    let name = if display.is_empty() { login } else { display };
225    serde_json::to_string(&ChatLine {
226        ts: ts_ms,
227        login,
228        name,
229        text,
230        color,
231        badges,
232        emotes,
233    })
234    .ok()
235}

237#[cfg(test)]
238mod tests {
239    use super::*;
240
241    #[test]
242    fn parses_tagged_privmsg() {
243        let line = "@badges=subscriber/12;color=#FF0000;display-name=CoolViewer;\
244                    emotes=25:0-4,12-16/1902:6-10;\
245                    tmi-sent-ts=1700000000123 :coolviewer!coolviewer@coolviewer.tmi.twitch.tv \
246                    PRIVMSG #streamer :hello there : world";
247        let json = parse_privmsg(line).expect("should parse");
248        let v: serde_json::Value = serde_json::from_str(&json).unwrap();
249        assert_eq!(v["login"], "coolviewer");
250        assert_eq!(v["name"], "CoolViewer");
251        // The first " :" is the param separator; the rest (incl. ": world") is text.
252        assert_eq!(v["text"], "hello there : world");
253        assert_eq!(v["color"], "#FF0000");
254        assert_eq!(v["ts"], 1700000000123i64);
255        assert_eq!(v["badges"], "subscriber/12");
256        // The raw emotes tag is captured verbatim for first-party emote replay.
257        assert_eq!(v["emotes"], "25:0-4,12-16/1902:6-10");
258    }
259
260    #[test]
261    fn omits_empty_emotes_tag() {
262        // A plain message has `emotes=` (empty); the field is skipped, like badges.
263        let line = "@badges=;color=;display-name=Bob;emotes=;tmi-sent-ts=1700000000000 \
264                    :bob!bob@bob.tmi.twitch.tv PRIVMSG #streamer :hi";
265        let json = parse_privmsg(line).expect("should parse");
266        let v: serde_json::Value = serde_json::from_str(&json).unwrap();
267        assert_eq!(v["text"], "hi");
268        assert!(v.get("emotes").is_none());
269    }
270
271    #[test]
272    fn untagged_privmsg_falls_back_to_login_and_clock() {
273        let line = ":bob!bob@bob.tmi.twitch.tv PRIVMSG #streamer :hi";
274        let json = parse_privmsg(line).expect("should parse");
275        let v: serde_json::Value = serde_json::from_str(&json).unwrap();
276        assert_eq!(v["login"], "bob");
277        assert_eq!(v["name"], "bob"); // no display-name tag -> login
278        assert_eq!(v["text"], "hi");
279        // color/badges omitted when empty.
280        assert!(v.get("color").is_none());
281        assert!(v.get("badges").is_none());
282        assert!(v["ts"].as_i64().unwrap() > 0);
283    }
284
285    #[test]
286    fn ignores_non_privmsg() {
287        assert!(parse_privmsg(":tmi.twitch.tv 001 justinfan1 :Welcome").is_none());
288        assert!(parse_privmsg("PING :tmi.twitch.tv").is_none());
289        assert!(parse_privmsg(":streamer!streamer@streamer.tmi.twitch.tv JOIN #streamer").is_none());
290    }
