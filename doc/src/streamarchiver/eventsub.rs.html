1//! Twitch EventSub real-time detection over WebSocket.
2//!
3//! Two transport modes are supported, chosen automatically:
4//!
5//! * **Conduit + app token** (Client ID + Secret): app `client_credentials`
6//!   token, conduit bound to the WebSocket shard. Scales to 10,000 channels.
7//! * **Direct WebSocket + user token** (Client ID + "Connect with Twitch"
8//!   OAuth): the stored user access token is used directly. No conduit or
9//!   Client Secret required. Twitch caps this at 10 `stream.online`
10//!   subscriptions — sufficient for most users.
11//!
12//! When both are present the conduit path is preferred (higher channel cap).
13//!
14//! Flow (conduit): app token -> resolve logins to user IDs -> ensure a conduit
15//! -> open the websocket -> bind shard 0 -> subscribe `stream.online` for each
16//! channel -> reconcile current live state via Get Streams -> stream events.
17//!
18//! Flow (user token): resolve user IDs -> open websocket -> read welcome ->
19//! subscribe `stream.online` (websocket transport) -> reconcile -> stream events.
20//!
21//! EventSub has no replay, so we reconcile on every (re)connect.
22//!
23//! NOTE: This path needs live Twitch credentials + a channel going live to
24//! verify end-to-end; use `--eventsub-test` to exercise it.
25

45const WS_URL: &str = "wss://eventsub.wss.twitch.tv/ws";
46const HELIX: &str = "https://api.twitch.tv/helix";
47/// How often to re-attempt when idle (no eventsub monitors / no creds).
48const IDLE_RETRY: Duration = Duration::from_secs(30);

50/// Run the EventSub manager until shutdown. Idles cheaply when there are no
51/// Twitch monitors using the EventSub method.
52pub async fn run(
53    store: Arc<Store>,
54    live_tx: UnboundedSender<LiveSignal>,
55    shutdown: Arc<AtomicBool>,
56) {
57    // Log the "have monitors but no creds" idle reason once, not every retry.
58    let mut warned_no_creds = false;
59    while !shutdown.load(Ordering::SeqCst) {
60        match run_session(&store, &live_tx, &shutdown, &mut warned_no_creds).await {
61            Ok(true) => warned_no_creds = false, // session ran; reset the one-shot warning
62            Ok(false) => sleep_cancellable(IDLE_RETRY, &shutdown).await, // nothing to do
63            Err(e) => {
64                warn!("eventsub: {e:#}");
65                sleep_cancellable(Duration::from_secs(10), &shutdown).await;
66            }
67        }
68    }
69}

71/// Returns Ok(true) if a websocket session was established, Ok(false) if there
72/// was nothing to do (no eventsub monitors / missing creds).
73async fn run_session(
74    store: &Arc<Store>,
75    live_tx: &UnboundedSender<LiveSignal>,
76    shutdown: &Arc<AtomicBool>,
77    warned_no_creds: &mut bool,
78) -> Result<bool> {
79    // login(lowercased) -> [monitor_id]
80    let login_to_monitors = load_eventsub_monitors(store)?;
81    if login_to_monitors.is_empty() {
82        return Ok(false);
83    }
84
85    let http = Client::builder().timeout(Duration::from_secs(20)).build()?;
86
87    // Prefer conduit + app token (scales to 10k channels). Fall back to direct
88    // WebSocket with the stored user OAuth token (≤10 channels, no Client Secret).
89    enum Transport {
90        Conduit { token: String },
91        UserToken { token: String },
92    }
93    let (client_id, transport) = if let Some((id, secret)) = app_credentials(store) {
94        let token = app_token(&http, &id, &secret).await?;
95        (id, Transport::Conduit { token })
96    } else if let Some((id, token)) = user_credentials(&http, store).await {
97        if login_to_monitors.len() > 10 {
98            warn!(
99                "eventsub: user-token mode supports up to 10 channels; {} configured \
100                 (add a Client Secret in Settings for unlimited)",
101                login_to_monitors.len()
102            );
103        }
104        (id, Transport::UserToken { token })
105    } else {
106        if !*warned_no_creds {
107            debug!(
108                "eventsub: have EventSub monitor(s) but no credentials; idling \
109                 (connect with Twitch in Settings, or set Client ID + Client Secret)"
110            );
111            *warned_no_creds = true;
112        }
113        return Ok(false);
114    };
115
116    let token = match &transport {
117        Transport::Conduit { token, .. } | Transport::UserToken { token } => token.as_str(),
118    };
119
120    let logins: Vec<String> = login_to_monitors.keys().cloned().collect();
121    let login_to_uid = resolve_user_ids(&http, &client_id, token, &logins).await?;
122    // user_id -> [monitor_id]
123    let mut uid_to_monitors: HashMap<String, Vec<i64>> = HashMap::new();
124    for (login, mons) in &login_to_monitors {
125        if let Some(uid) = login_to_uid.get(login) {
126            uid_to_monitors.entry(uid.clone()).or_default().extend(mons);
127        } else {
128            warn!("eventsub: could not resolve Twitch user id for '{login}'");
129        }
130    }
131    if uid_to_monitors.is_empty() {
132        return Ok(false);
133    }
134
135    let (mut ws, _) = tokio_tungstenite::connect_async(WS_URL)
136        .await
137        .context("connecting eventsub websocket")?;
138    let (session_id, keepalive) = read_welcome(&mut ws).await?;
139
140    match &transport {
141        Transport::Conduit { token } => {
142            let conduit_id = ensure_conduit(&http, &client_id, token).await?;
143            bind_shard(&http, &client_id, token, &conduit_id, &session_id).await?;
144            let n = subscribe_online(&http, &client_id, token, &conduit_id, &uid_to_monitors).await;
145            info!(
146                "eventsub: connected (conduit {conduit_id}); {n}/{} channel(s) subscribed",
147                uid_to_monitors.len()
148            );
149        }
150        Transport::UserToken { token } => {
151            let n = subscribe_online_ws(&http, &client_id, token, &session_id, &uid_to_monitors).await;
152            info!(
153                "eventsub: connected (user-token); {n}/{} channel(s) subscribed",
154                uid_to_monitors.len()
155            );
156        }
157    }
158    // Reconcile: anything already live should start recording now.
159    reconcile(&http, &client_id, token, &login_to_monitors, live_tx).await;
160
161    // Event loop. Twitch sends `session_keepalive` every `keepalive` seconds; if
162    // none (plus slack) arrives, treat the connection as dead.
163    let read_timeout = Duration::from_secs(keepalive + 15);
164    loop {
165        if shutdown.load(Ordering::SeqCst) {
166            return Ok(true);
167        }
168        let next = timeout(read_timeout, ws.next()).await;
169        let msg = match next {
170            Err(_) => bail!("keepalive timeout"),
171            Ok(None) => bail!("websocket closed"),
172            Ok(Some(Err(e))) => bail!("websocket error: {e}"),
173            Ok(Some(Ok(m))) => m,
174        };
175        use tokio_tungstenite::tungstenite::Message;
176        match msg {
177            Message::Text(text) => {
178                if handle_message(&text, &uid_to_monitors, live_tx) {
179                    // session_reconnect — drop out and reconnect fresh.
180                    return Ok(true);
181                }
182            }
183            Message::Ping(payload) => {
184                let _ = ws.send(Message::Pong(payload)).await;
185            }
186            Message::Close(_) => bail!("websocket close frame"),
187            _ => {}
188        }
189    }
190}

192/// Handle one EventSub text frame. Returns true if the caller should reconnect.
193fn handle_message(
194    text: &str,
195    uid_to_monitors: &HashMap<String, Vec<i64>>,
196    live_tx: &UnboundedSender<LiveSignal>,
197) -> bool {
198    let Ok(v) = serde_json::from_str::<Value>(text) else {
199        return false;
200    };
201    match v["metadata"]["message_type"].as_str().unwrap_or("") {
202        "session_keepalive" => {}
203        "session_reconnect" => return true,
204        "notification" => {
205            if v["payload"]["subscription"]["type"].as_str() == Some("stream.online") {
206                let event = &v["payload"]["event"];
207                let went_live = event["started_at"].as_str().and_then(parse_ts);
208                let stream_id = event["id"].as_str().map(str::to_string);
209                if let Some(uid) = event["broadcaster_user_id"].as_str() {
210                    if let Some(mons) = uid_to_monitors.get(uid) {
211                        for &mid in mons {
212                            info!("eventsub: stream.online -> monitor {mid}");
213                            let _ = live_tx.send(
214                                LiveSignal::new(mid, went_live, false)
215                                    .with_stream_id(stream_id.clone()),
216                            );
217                        }
218                    }
219                }
220            }
221        }
222        "revocation" => warn!("eventsub: a subscription was revoked"),
223        _ => {}
224    }
225    false
226}

228fn app_credentials(store: &Store) -> Option<(String, String)> {
229    let id = store.get_setting("twitch_client_id").ok().flatten()?;
230    let secret = store.get_setting("twitch_client_secret").ok().flatten()?;
231    if id.is_empty() || secret.is_empty() {
232        None
233    } else {
234        Some((id, secret))
235    }
236}

238async fn user_credentials(http: &Client, store: &Store) -> Option<(String, String)> {
239    let client_id = store
240        .get_setting("twitch_client_id")
241        .ok()
242        .flatten()
243        .filter(|s| !s.is_empty())?;
244    let token = oauth::valid_user_token(http, store).await?;
245    Some((client_id, token))
246}

248fn load_eventsub_monitors(store: &Store) -> Result<HashMap<String, Vec<i64>>> {
249    let mut map: HashMap<String, Vec<i64>> = HashMap::new();
250    for row in store.list_monitors_with_channels()? {
251        if row.monitor.platform() == Platform::Twitch
252            && matches!(
253                row.monitor.detection_method,
254                DetectionMethod::EventSub | DetectionMethod::EventSubHelix
255            )
256        {
257            if let Some(login) = twitch_login(&row.monitor.url) {
258                map.entry(login).or_default().push(row.monitor.id);
259            }
260        }
261    }
262    Ok(map)
263}

265async fn app_token(http: &Client, client_id: &str, secret: &str) -> Result<String> {
266    let resp = http
267        .post("https://id.twitch.tv/oauth2/token")
268        .form(&[
269            ("client_id", client_id),
270            ("client_secret", secret),
271            ("grant_type", "client_credentials"),
272        ])
273        .send()
274        .await?;
275    if !resp.status().is_success() {
276        bail!("token request failed: {}", resp.status());
277    }
278    let v: Value = resp.json().await?;
279    v["access_token"]
280        .as_str()
281        .map(str::to_string)
282        .context("no access_token in token response")
283}

285async fn resolve_user_ids(
286    http: &Client,
287    client_id: &str,
288    token: &str,
289    logins: &[String],
290) -> Result<HashMap<String, String>> {
291    let mut out = HashMap::new();
292    for chunk in logins.chunks(100) {
293        let query: Vec<(&str, &str)> = chunk.iter().map(|l| ("login", l.as_str())).collect();
294        let resp = http
295            .get(format!("{HELIX}/users"))
296            .header("Client-Id", client_id)
297            .bearer_auth(token)
298            .query(&query)
299            .send()
300            .await?;
301        if !resp.status().is_success() {
302            bail!("get users failed: {}", resp.status());
303        }
304        let v: Value = resp.json().await?;
305        if let Some(arr) = v["data"].as_array() {
306            for u in arr {
307                if let (Some(login), Some(id)) = (u["login"].as_str(), u["id"].as_str()) {
308                    out.insert(login.to_lowercase(), id.to_string());
309                }
310            }
311        }
312    }
313    Ok(out)
314}

316/// Reuse an existing conduit if present, else create one with a single shard.
317async fn ensure_conduit(http: &Client, client_id: &str, token: &str) -> Result<String> {
318    let resp = http
319        .get(format!("{HELIX}/eventsub/conduits"))
320        .header("Client-Id", client_id)
321        .bearer_auth(token)
322        .send()
323        .await?;
324    if resp.status().is_success() {
325        let v: Value = resp.json().await?;
326        if let Some(id) = v["data"]
327            .as_array()
328            .and_then(|a| a.first())
329            .and_then(|c| c["id"].as_str())
330        {
331            return Ok(id.to_string());
332        }
333    }
334    let resp = http
335        .post(format!("{HELIX}/eventsub/conduits"))
336        .header("Client-Id", client_id)
337        .bearer_auth(token)
338        .json(&json!({ "shard_count": 1 }))
339        .send()
340        .await?;
341    if !resp.status().is_success() {
342        bail!("create conduit failed: {}", resp.status());
343    }
344    let v: Value = resp.json().await?;
345    v["data"][0]["id"]
346        .as_str()
347        .map(str::to_string)
348        .context("no conduit id in create response")
349}

351async fn read_welcome(
352    ws: &mut tokio_tungstenite::WebSocketStream<
353        tokio_tungstenite::MaybeTlsStream<tokio::net::TcpStream>,
354    >,
355) -> Result<(String, u64)> {
356    use tokio_tungstenite::tungstenite::Message;
357    // The welcome must arrive within ~10s. Skip non-text frames (e.g. Ping)
358    // that may precede it; reply to Pings so the server doesn't drop us.
359    let text = loop {
360        let msg = timeout(Duration::from_secs(15), ws.next())
361            .await
362            .context("timed out waiting for session_welcome")?
363            .context("websocket closed before welcome")??;
364        match msg {
365            Message::Text(t) => break t,
366            Message::Ping(p) => { let _ = ws.send(Message::Pong(p)).await; }
367            _ => {}
368        }
369    };
370    let v: Value = serde_json::from_str(&text)?;
371    if v["metadata"]["message_type"].as_str() != Some("session_welcome") {
372        bail!(
373            "expected session_welcome, got {}",
374            v["metadata"]["message_type"]
375        );
376    }
377    let session_id = v["payload"]["session"]["id"]
378        .as_str()
379        .context("no session id")?
380        .to_string();
381    let keepalive = v["payload"]["session"]["keepalive_timeout_seconds"]
382        .as_u64()
383        .unwrap_or(10);
384    Ok((session_id, keepalive))
385}

387async fn bind_shard(
388    http: &Client,
389    client_id: &str,
390    token: &str,
391    conduit_id: &str,
392    session_id: &str,
393) -> Result<()> {
394    let resp = http
395        .patch(format!("{HELIX}/eventsub/conduits/shards"))
396        .header("Client-Id", client_id)
397        .bearer_auth(token)
398        .json(&json!({
399            "conduit_id": conduit_id,
400            "shards": [ { "id": "0", "transport": { "method": "websocket", "session_id": session_id } } ]
401        }))
402        .send()
403        .await?;
404    if !resp.status().is_success() {
405        bail!("bind shard failed: {}", resp.status());
406    }
407    Ok(())
408}

410/// Subscribe stream.online for each user id (skipping ones already enabled).
411/// Returns the count of channels covered.
412async fn subscribe_online(
413    http: &Client,
414    client_id: &str,
415    token: &str,
416    conduit_id: &str,
417    uid_to_monitors: &HashMap<String, Vec<i64>>,
418) -> usize {
419    let existing = existing_online_subs(http, client_id, token).await;
420    let mut covered = 0;
421    for uid in uid_to_monitors.keys() {
422        if existing.contains(uid) {
423            covered += 1;
424            continue;
425        }
426        let resp = http
427            .post(format!("{HELIX}/eventsub/subscriptions"))
428            .header("Client-Id", client_id)
429            .bearer_auth(token)
430            .json(&json!({
431                "type": "stream.online",
432                "version": "1",
433                "condition": { "broadcaster_user_id": uid },
434                "transport": { "method": "conduit", "conduit_id": conduit_id }
435            }))
436            .send()
437            .await;
438        match resp {
439            Ok(r) if r.status().is_success() => covered += 1,
440            Ok(r) => warn!("eventsub: subscribe {uid} failed: {}", r.status()),
441            Err(e) => warn!("eventsub: subscribe {uid} error: {e}"),
442        }
443    }
444    covered
445}

485async fn existing_online_subs(
486    http: &Client,
487    client_id: &str,
488    token: &str,
489) -> std::collections::HashSet<String> {
490    let mut set = std::collections::HashSet::new();
491    let resp = http
492        .get(format!("{HELIX}/eventsub/subscriptions"))
493        .header("Client-Id", client_id)
494        .bearer_auth(token)
495        .query(&[("status", "enabled")])
496        .send()
497        .await;
498    if let Ok(r) = resp {
499        if let Ok(v) = r.json::<Value>().await {
500            if let Some(arr) = v["data"].as_array() {
501                for s in arr {
502                    if s["type"].as_str() == Some("stream.online") {
503                        if let Some(uid) = s["condition"]["broadcaster_user_id"].as_str() {
504                            set.insert(uid.to_string());
505                        }
506                    }
507                }
508            }
509        }
510    }
511    set
512}

514/// Send a live signal for any subscribed channel that is currently live (covers
515/// the gap since EventSub has no event replay).
516async fn reconcile(
517    http: &Client,
518    client_id: &str,
519    token: &str,
520    login_to_monitors: &HashMap<String, Vec<i64>>,
521    live_tx: &UnboundedSender<LiveSignal>,
522) {
523    let logins: Vec<String> = login_to_monitors.keys().cloned().collect();
524    for chunk in logins.chunks(100) {
525        let query: Vec<(&str, &str)> = chunk.iter().map(|l| ("user_login", l.as_str())).collect();
526        let resp = http
527            .get(format!("{HELIX}/streams"))
528            .header("Client-Id", client_id)
529            .bearer_auth(token)
530            .query(&query)
531            .send()
532            .await;
533        if let Ok(r) = resp {
534            if let Ok(v) = r.json::<Value>().await {
535                if let Some(arr) = v["data"].as_array() {
536                    for s in arr {
537                        if s["type"].as_str() == Some("live") {
538                            let went_live = s["started_at"].as_str().and_then(parse_ts);
539                            let stream_id = s["id"].as_str().map(str::to_string);
540                            if let Some(login) = s["user_login"].as_str() {
541                                if let Some(mons) = login_to_monitors.get(&login.to_lowercase()) {
542                                    for &mid in mons {
543                                        let _ = live_tx.send(
544                                            LiveSignal::new(mid, went_live, false)
545                                                .with_stream_id(stream_id.clone()),
546                                        );
547                                    }
548                                }
549                            }
550                        }
551                    }
552                }
553            }
554        }
555    }
556}

558/// Parse an RFC3339 timestamp (Twitch `started_at`) to unix seconds.
559fn parse_ts(s: &str) -> Option<i64> {
560    chrono::DateTime::parse_from_rfc3339(s)
561        .ok()
562        .map(|dt| dt.timestamp())
563}

565/// Extract the Twitch login from a channel URL (`twitch.tv/<login>`).
566fn twitch_login(url: &str) -> Option<String> {
567    let trimmed = url.trim().trim_end_matches('/');
568    let lower = trimmed.to_lowercase();
569    let pos = lower.find("twitch.tv/")?;
570    let rest = &trimmed[pos + "twitch.tv/".len()..];
571    let login = rest.split(['/', '?', '#']).next()?.trim();
572    (!login.is_empty()).then(|| login.to_lowercase())
573}
