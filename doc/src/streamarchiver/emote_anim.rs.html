1//! Decoded, downscaled, animation-aware emote/emoji image cache.
2//!
3//! egui's built-in image loaders decode animated GIF/WebP at *source* resolution
4//! (a 7TV 4× emote ≈ 384 px, ×N frames) and never evict — a wall of animated
5//! emotes can balloon RAM into the hundreds of MB. This module decodes frames off
6//! the UI thread and **downscales each to render size** before GPU upload (the
7//! single biggest win — ~16–45× fewer bytes/frame), drives animation off a global
8//! wall-clock, and lets the UI enforce an LRU memory budget. Static images are just
9//! a one-frame animation, so this one path serves PNG/JPEG, animated GIF, and
10//! animated WebP — and the animate/static toggle is "cycle frames" vs "frame 0".
11
12use eframe::egui;
13use image::AnimationDecoder;
14
15/// Frames are downscaled so their longest side is ≤ this (≈2× the 28 px render
16/// height, kept for crispness on hi-DPI). The dominant memory saving.
17const TARGET_PX: u32 = 56;
18/// Per-frame delay floor (20 fps) — clamps absurdly fast GIFs and the repaint rate.
19const MIN_DELAY: f32 = 0.05;
20
21/// One emote's state in the cache.
22pub enum EmoteLoad {
23    /// A background decode has been spawned.
24    Loading,
25    /// Decoded + downscaled off-thread; awaiting GPU upload on the UI thread.
26    Decoded(Vec<egui::ColorImage>, Vec<f32>),
27    /// Uploaded and ready to draw.
28    Ready(EmoteAnim),
29    /// Unreadable / undecodable — the renderer falls back to text.
30    Failed,
31}
32
33/// A decoded emote: 1 frame (static) or N frames (animated), uploaded to textures.
34pub struct EmoteAnim {
35    frames: Vec<egui::TextureHandle>,
36    /// Cumulative frame end-times in seconds; `cum.last()` is the loop duration.
37    cum: Vec<f32>,
38    /// Approximate GPU bytes (for the LRU budget).
39    pub bytes: usize,
40    /// Wall-clock time this emote was last drawn (for LRU eviction).
41    pub last_drawn: f64,
42}
43
44impl EmoteAnim {
45    pub fn is_animated(&self) -> bool {
46        self.frames.len() > 1 && self.cum.last().is_some_and(|&t| t > 0.0)
47    }
48
49    pub fn size(&self) -> egui::Vec2 {
50        self.frames[0].size_vec2()
51    }
52
53    /// `(texture, seconds-until-next-frame)` for global clock time `t`. For a static
54    /// emote (or `t` infinite-loop edge) returns frame 0 with no scheduled repaint.
55    pub fn frame_at(&self, t: f64) -> (&egui::TextureHandle, f32) {
56        let total = self.cum.last().copied().unwrap_or(0.0);
57        if !self.is_animated() {
58            return (&self.frames[0], f32::INFINITY);
59        }
60        let pos = (t.rem_euclid(total as f64)) as f32;
61        let idx = self.cum.iter().position(|&c| c > pos).unwrap_or(0);
62        let remaining = (self.cum[idx] - pos).max(MIN_DELAY);
63        (&self.frames[idx], remaining)
64    }
65}

67/// Decode (and downscale) image bytes into frames + per-frame delays, off the UI
68/// thread. `None` on an unreadable/undecodable input.
69pub fn decode(bytes: &[u8]) -> Option<(Vec<egui::ColorImage>, Vec<f32>)> {
70    let (rgba, delays) = match image::guess_format(bytes).ok()? {
71        image::ImageFormat::Gif => {
72            let dec = image::codecs::gif::GifDecoder::new(std::io::Cursor::new(bytes)).ok()?;
73            collect_frames(dec)?
74        }
75        image::ImageFormat::WebP => {
76            let dec = image::codecs::webp::WebPDecoder::new(std::io::Cursor::new(bytes)).ok()?;
77            if dec.has_animation() {
78                collect_frames(dec)?
79            } else {
80                (vec![image::load_from_memory(bytes).ok()?.to_rgba8()], vec![0.0])
81            }
82        }
83        _ => (vec![image::load_from_memory(bytes).ok()?.to_rgba8()], vec![0.0]),
84    };
85    let images = rgba
86        .into_iter()
87        .map(|img| {
88            let img = downscale(img);
89            let size = [img.width() as usize, img.height() as usize];
90            egui::ColorImage::from_rgba_unmultiplied(size, img.as_raw())
91        })
92        .collect();
93    Some((images, delays))
94}
95
96fn collect_frames<'a>(
97    dec: impl AnimationDecoder<'a>,
98) -> Option<(Vec<image::RgbaImage>, Vec<f32>)> {
99    let frames = dec.into_frames().collect_frames().ok()?;
100    if frames.is_empty() {
101        return None;
102    }
103    let mut imgs = Vec::with_capacity(frames.len());
104    let mut delays = Vec::with_capacity(frames.len());
105    for f in frames {
106        let (num, den) = f.delay().numer_denom_ms();
107        let secs = if den == 0 {
108            0.1
109        } else {
110            (num as f32 / den as f32) / 1000.0
111        };
112        delays.push(secs.max(MIN_DELAY));
113        imgs.push(f.into_buffer());
114    }
115    Some((imgs, delays))
116}
117
118fn downscale(img: image::RgbaImage) -> image::RgbaImage {
119    let (w, h) = img.dimensions();
120    let longest = w.max(h);
121    if longest <= TARGET_PX {
122        return img;
123    }
124    let scale = TARGET_PX as f32 / longest as f32;
125    let nw = ((w as f32 * scale).round() as u32).max(1);
126    let nh = ((h as f32 * scale).round() as u32).max(1);
127    image::imageops::resize(&img, nw, nh, image::imageops::FilterType::Triangle)
128}

130/// Upload decoded frames to GPU textures on the UI thread, producing a drawable
131/// [`EmoteAnim`]. `key` makes the texture names unique + stable per source path.
132pub fn upload(
133    images: Vec<egui::ColorImage>,
134    delays: Vec<f32>,
135    ctx: &egui::Context,
136    key: &str,
137) -> EmoteAnim {
138    let mut frames = Vec::with_capacity(images.len());
139    let mut cum = Vec::with_capacity(images.len());
140    let mut bytes = 0usize;
141    let mut acc = 0.0f32;
142    for (i, ci) in images.into_iter().enumerate() {
143        bytes += ci.width() * ci.height() * 4;
144        acc += delays.get(i).copied().unwrap_or(0.0);
145        cum.push(acc);
146        let tex = ctx.load_texture(format!("emote_{key}_{i}"), ci, egui::TextureOptions::LINEAR);
147        frames.push(tex);
148    }
149    EmoteAnim {
150        frames,
151        cum,
152        bytes,
153        last_drawn: 0.0,
154    }
155}

157#[cfg(test)]
158mod tests {
159    use super::*;
160
161    fn anim(cum: Vec<f32>) -> EmoteAnim {
162        EmoteAnim {
163            frames: Vec::new(), // not dereferenced by frame_at when computing index logic below
164            cum,
165            bytes: 0,
166            last_drawn: 0.0,
167        }
168    }
169
170    #[test]
171    fn frame_index_math_wraps_on_loop() {
172        // 3 frames, 0.1s each → cumulative [0.1, 0.2, 0.3], loop 0.3s.
173        let a = anim(vec![0.1, 0.2, 0.3]);
174        // helper replicating frame_at's index pick (without touching textures).
175        let pick = |t: f64| {
176            let total = *a.cum.last().unwrap();
177            let pos = (t.rem_euclid(total as f64)) as f32;
178            a.cum.iter().position(|&c| c > pos).unwrap_or(0)
179        };
180        assert_eq!(pick(0.0), 0);
181        assert_eq!(pick(0.15), 1);
182        assert_eq!(pick(0.25), 2);
183        assert_eq!(pick(0.30), 0); // wraps
184        assert_eq!(pick(0.45), 1); // 0.45 % 0.3 = 0.15 → frame 1
185    }
