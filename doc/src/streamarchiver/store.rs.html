15/// Current date in US Pacific Time (UTC-8), which is when Google's API quotas
16/// reset. Using PST (UTC-8) exactly matches the reset in winter; in summer PDT
17/// (UTC-7) the local Pacific day starts 1h earlier than our boundary, so we
18/// carry at most 1 hour of extra headroom — the safe direction vs. resetting
19/// 9+ hours early when using the user's local timezone.
20fn quota_date_today() -> String {
21    let utc_secs = chrono::Utc::now().timestamp();
22    let pst = chrono::FixedOffset::west_opt(8 * 3600).unwrap();
23    pst.timestamp_opt(utc_secs, 0)
24        .unwrap()
25        .format("%Y-%m-%d")
26        .to_string()
27}

81/// RAII guard returned by [`Store::db`]. Logs a warning when the lock is held
82/// longer than 200 ms, showing the call-site that acquired it — useful for
83/// identifying which store method is the bottleneck.
84struct DbGuard<'a> {
85    inner: parking_lot::FairMutexGuard<'a, Connection>,
86    acquired_at: std::time::Instant,
87    caller: &'static std::panic::Location<'static>,
88}
89
90impl std::ops::Deref for DbGuard<'_> {
91    type Target = Connection;
92    fn deref(&self) -> &Connection { &*self.inner }
93}
94
95impl std::ops::DerefMut for DbGuard<'_> {
96    fn deref_mut(&mut self) -> &mut Connection { &mut *self.inner }
97}
98
99impl Drop for DbGuard<'_> {
100    fn drop(&mut self) {
101        let ms = self.acquired_at.elapsed().as_millis();
102        if ms >= 200 {
103            tracing::warn!(
104                hold_ms = ms,
105                file = self.caller.file(),
106                line = self.caller.line(),
107                "store: long DB lock hold"
108            );
109        } else if ms >= 50 {
110            tracing::debug!(
111                hold_ms = ms,
112                file = self.caller.file(),
113                line = self.caller.line(),
114                "store: DB lock hold"
115            );
116        }
117    }
118}

120impl Store {
121    /// Acquire the DB connection. Logs a warning when contention caused a wait
122    /// longer than 50 ms (waiter side) or when the lock is held longer than
123    /// 200 ms (holder side). `#[track_caller]` embeds the caller's source
124    /// location in both log lines so slow call-sites are immediately visible.
125    #[track_caller]
126    fn db(&self) -> DbGuard<'_> {
127        let caller = std::panic::Location::caller();
128        let t = std::time::Instant::now();
129        let g = self.conn.lock();
130        let wait_ms = t.elapsed().as_millis();
131        if wait_ms >= 50 {
132            tracing::warn!(
133                wait_ms,
134                file = caller.file(),
135                line = caller.line(),
136                "store: slow DB lock – another thread held the connection"
137            );
138        } else if wait_ms >= 5 {
139            tracing::debug!(
140                wait_ms,
141                file = caller.file(),
142                line = caller.line(),
143                "store: DB lock wait"
144            );
145        }
146        DbGuard { inner: g, acquired_at: std::time::Instant::now(), caller }
147    }

149    /// Open (or create) the database at `path`, set pragmas, and migrate.
150    pub fn open(path: &Path) -> Result<Store> {
151        let conn = Connection::open(path)
152            .with_context(|| format!("opening database at {}", path.display()))?;
153        Self::configure(&conn)?;
154        let store = Store {
155            conn: FairMutex::new(conn),
156        };
157        store.migrate()?;
158        Ok(store)
159    }

174        conn.pragma_update(None, "journal_mode", "WAL")?;
175        conn.pragma_update(None, "busy_timeout", 5000)?;

601            // `source`, so queries filtering by source scanned every row for
602            // that monitor. On an accumulated historical archive (past segments
603            // are kept as history) this caused multi-second lock holds per call.
604            conn.execute_batch(
605                "CREATE INDEX IF NOT EXISTS idx_schedule_source
606                 ON schedule_segment(monitor_id, source, start_time);",
607            )?;
608            conn.pragma_update(None, "user_version", 31)?;
609        }
610        if version < 32 {
611            // Index for all_upcoming_schedule: the query filters
612            // `canceled = 0 AND start_time >= ?` across all monitors, but the
613            // existing idx_schedule_monitor leads with monitor_id, so SQLite
614            // had to full-scan the entire table. With months of historical rows
615            // accumulated this caused 4+ second lock holds on Schedule tab clicks.
616            conn.execute_batch(
617                "CREATE INDEX IF NOT EXISTS idx_schedule_canceled_start
618                 ON schedule_segment(canceled, start_time);",
619            )?;
620            conn.pragma_update(None, "user_version", 32)?;
621        }
622        if version < 33 {
623            // Per-provider, per-day API quota tracking. Currently used for the
624            // YouTube Data API (10,000 free units/day). `provider` is a short key
625            // like "youtube"; `date` is an ISO date string "YYYY-MM-DD".
626            conn.execute_batch(
627                "CREATE TABLE IF NOT EXISTS api_quota (
628                    provider TEXT NOT NULL,
629                    date     TEXT NOT NULL,
630                    units    INTEGER NOT NULL DEFAULT 0,
631                    PRIMARY KEY (provider, date)
632                );",
633            )?;
634            conn.pragma_update(None, "user_version", 33)?;
635        }
636        if version < 34 {
637            // Schedule segment merge support: `merged_into` links a secondary
638            // segment to its primary (manual merge) so the secondary is hidden
639            // from the calendar in favour of the primary. `auto_merge_excluded`
640            // opts a segment out of automatic time-overlap merge grouping with
641            // same-channel events.
642            conn.execute_batch(
643                "ALTER TABLE schedule_segment ADD COLUMN merged_into       INTEGER;
644                 ALTER TABLE schedule_segment ADD COLUMN auto_merge_excluded INTEGER NOT NULL DEFAULT 0;",
645            )?;
646            conn.pragma_update(None, "user_version", 34)?;
647        }
648        if version < 35 {
649            // User-defined filename-template presets (name + template string).
650            conn.execute_batch(
651                "CREATE TABLE IF NOT EXISTS filename_preset (
652                     id       INTEGER PRIMARY KEY,
653                     name     TEXT NOT NULL,
654                     template TEXT NOT NULL
655                 );",
656            )?;
657            conn.pragma_update(None, "user_version", 35)?;
658        }
659        if version < 36 {
660            // Deduplicate schedule_segment rows that accumulated due to a bug where
661            // OCR-cadence cache hits re-inserted past rows on every 60-second tick
662            // (replace_schedule_source deletes only future rows, so past rows doubled
663            // each tick). Keep the earliest id per (monitor, source, start_time,
664            // canceled) tuple; window function avoids an expensive NOT-IN subquery.
665            conn.execute_batch(
666                "DELETE FROM schedule_segment WHERE id IN (
667                     SELECT id FROM (
668                         SELECT id,
669                                ROW_NUMBER() OVER (
670                                    PARTITION BY monitor_id, source, start_time, canceled
671                                    ORDER BY id
672                                ) AS rn
673                         FROM schedule_segment
674                     ) WHERE rn > 1
675                 );",
676            )?;
677            conn.pragma_update(None, "user_version", 36)?;
678        }
679        debug_assert_eq!(SCHEMA_VERSION, 36);
680        Ok(())
681    }

1461    /// Replace one monitor's schedule for a single `source` (`"platform"` or
1462    /// `"discord"`), leaving its other sources intact (delete-then-insert in one
1463    /// transaction). Lets the platform schedule and Discord events coexist without
1464    /// clobbering each other.
1465    pub fn replace_schedule_source(
1466        &self,
1467        monitor_id: i64,
1468        source: &str,
1469        segs: &[ScheduleSegment],
1470    ) -> Result<()> {
1471        let now = now_unix();
1472        // Phase 1 — short lock: read suppressed start instants so the lock is
1473        // released before the heavier write transaction below.  Other threads
1474        // (save, reload-rows) can acquire the DB between the two phases.
1475        //
1476        // Start instants an automatic source must NOT re-create for this monitor:
1477        //  • those claimed by a protected manual row (a user's correction — don't
1478        //    duplicate it at the same instant), and
1479        //  • those a user explicitly removed or moved away from (tombstones,
1480        //    canceled = 1 — re-inserting would resurrect a deleted occurrence, or
1481        //    re-add the pre-correction copy of a rescheduled one).
1482        // Storing the manual source itself doesn't self-suppress on manual rows,
1483        // but still honours tombstones.
1484        let suppressed_starts: std::collections::HashSet<i64> = {
1485            let conn = self.db();
1486            let mut stmt = conn.prepare(
1487                "SELECT start_time FROM schedule_segment
1488                 WHERE monitor_id = ?1
1489                   AND ((source = 'manual' AND ?2 <> 'manual') OR canceled = 1)",
1490            )?;
1491            stmt.query_map(params![monitor_id, source], |r| r.get::<_, i64>(0))?
1492                .collect::<rusqlite::Result<std::collections::HashSet<_>>>()?
1493        };
1494        // DB lock released — other threads get a turn between phases.
1495
1496        // Pre-filter in memory: only the segments we'll actually write.
1497        let segs_to_write: Vec<&ScheduleSegment> = segs
1498            .iter()
1499            .filter(|s| !suppressed_starts.contains(&s.start_time))
1500            .collect();
1501
1502        // Phase 2 — write transaction: all SQL is writes-only, keeping this
1503        // lock hold as short as possible.
1504        let mut conn = self.db();
1505        let tx = conn.transaction()?;
1506        // Drop stale tombstones (canceled rows well past their instant): a
1507        // tombstone's job is to suppress re-insertion of a deleted event on the
1508        // next refresh. Because we only ever re-insert UPCOMING events (the delete
1509        // below is future-only), a tombstone becomes inert once its start_time is
1510        // comfortably in the past. Keep a 7-day grace so a deletion stays visible
1511        // while the user can still see that week in the calendar.
1512        tx.execute(
1513            "DELETE FROM schedule_segment
1514             WHERE monitor_id = ?1 AND canceled = 1 AND start_time < ?2",
1515            params![monitor_id, now - 7 * 86_400],
1516        )?;
1517        // Replace only this source's FUTURE non-canceled rows. Past events are
1518        // intentionally preserved as a schedule archive — the next banner fetch only
1519        // re-emits upcoming streams, so old rows accumulate as history. Tombstones
1520        // (canceled = 1) are also preserved so a user's "Delete" keeps suppressing
1521        // a recurring event on future re-fetches.
1522        tx.execute(
1523            "DELETE FROM schedule_segment
1524             WHERE monitor_id = ?1 AND source = ?2 AND canceled = 0
1525               AND start_time >= ?3",
1526            params![monitor_id, source, now],
1527        )?;
1528        // For each segment the winning source is about to write, evict any other
1529        // automatic source's live row at that SAME instant — both past and future.
1530        // This prevents cross-source duplicates from accumulating in the archive
1531        // when two sources (e.g. Twitch schedule + OCR banner) previously both
1532        // stored the same event (possibly with different title casing). The winning
1533        // source's version is the only one that survives per (monitor, instant).
1534        {
1535            let mut evict = tx.prepare(
1536                "DELETE FROM schedule_segment
1537                 WHERE monitor_id = ?1 AND source <> ?2 AND source <> 'manual'
1538                   AND start_time = ?3 AND canceled = 0",
1539            )?;
1540            for s in &segs_to_write {
1541                evict.execute(params![monitor_id, source, s.start_time])?;
1542            }
1543        }
1544        {
1545            let mut stmt = tx.prepare(
1546                "INSERT INTO schedule_segment(monitor_id, start_time, end_time, title, category, canceled, source, video_id)
1547                 VALUES(?1, ?2, ?3, ?4, ?5, ?6, ?7, ?8)",
1548            )?;
1549            for s in &segs_to_write {
1550                stmt.execute(params![
1551                    monitor_id,
1552                    s.start_time,
1553                    s.end_time,
1554                    s.title,
1555                    s.category,
1556                    s.canceled as i64,
1557                    source,
1558                    s.video_id.as_deref(),
1559                ])?;
1560            }
1561        }
1562        tx.commit()?;
1563        Ok(())
1564    }

1584    /// Convert one schedule segment into a protected manual entry, overwriting its
1585    /// time/title/category. The row's `source` becomes `"manual"` and `canceled`
1586    /// is cleared, so subsequent automatic refreshes of other sources leave it
1587    /// intact (see [`Self::clear_other_schedule_sources`]). Backs the calendar's
1588    /// "Edit…" dialog. Returns the number of rows updated — `0` means the segment
1589    /// no longer exists (e.g. a background refresh cleared it mid-edit), so the
1590    /// caller can report the edit didn't apply instead of falsely claiming success.
1591    ///
1592    /// When the edit MOVES an automatic event to a new instant, a tombstone
1593    /// (`canceled = 1`) is left at the original instant so the next refresh of
1594    /// that source doesn't re-create the stream at its old, uncorrected time
1595    /// alongside the correction (see [`Self::replace_schedule_source`]).
1596    pub fn update_schedule_segment_manual(
1597        &self,
1598        id: i64,
1599        start_time: i64,
1600        end_time: Option<i64>,
1601        title: &str,
1602        category: &str,
1603    ) -> Result<usize> {
1604        let mut conn = self.db();
1605        let tx = conn.transaction()?;
1606        // Capture the pre-edit state so we know whether a time-move tombstone is
1607        // needed and what source to attribute it to.
1608        let orig: Option<(i64, i64, String)> = tx
1609            .query_row(
1610                "SELECT monitor_id, start_time, source FROM schedule_segment WHERE id = ?1",
1611                params![id],
1612                |r| Ok((r.get::<_, i64>(0)?, r.get::<_, i64>(1)?, r.get::<_, String>(2)?)),
1613            )
1614            .optional()?;
1615        let Some((monitor_id, orig_start, orig_source)) = orig else {
1616            return Ok(0);
1617        };
1618        let n = tx.execute(
1619            "UPDATE schedule_segment
1620                SET start_time = ?2, end_time = ?3, title = ?4, category = ?5,
1621                    source = 'manual', canceled = 0
1622              WHERE id = ?1",
1623            params![id, start_time, end_time, title, category],
1624        )?;
1625        if n > 0 && orig_source != "manual" && orig_start != start_time {
1626            tx.execute(
1627                "INSERT INTO schedule_segment
1628                     (monitor_id, start_time, end_time, title, category, canceled, source, video_id)
1629                 VALUES (?1, ?2, NULL, '', '', 1, ?3, NULL)",
1630                params![monitor_id, orig_start, orig_source],
1631            )?;
1632        }
1633        tx.commit()?;
1634        Ok(n)
1635    }

3339    fn community_post_archive_dedupes_and_preserves_ocr() {
3340        let store = Store::open_in_memory().unwrap();
3341        let cid = store.create_container("Streamer").unwrap();
3342        let mut m = sample_monitor(cid);
3343        m.channel_id = cid;
3344        let mid = store.insert_monitor(&m).unwrap();
3345
3346        // First download of an image.
3347        store
3348            .community_post_upsert(mid, "youtube_community_ocr", "https://x/a.jpg", "1111", "C:/c/1111", 100)
3349            .unwrap();
3350        assert_eq!(store.community_post_count(mid).unwrap(), 1);
3351
3352        // Cache it as OCR'd with one decoded event.
3353        store
3354            .community_post_set_decoded(mid, "1111", 1, r#"[{"t":"x"}]"#)
3355            .unwrap();
3356        let got = store.community_post_get(mid, "1111").unwrap().unwrap();
3357        assert!(got.ocr_attempted);
3358        assert_eq!(got.decoded_events, 1);
3359        assert_eq!(got.decoded_json, r#"[{"t":"x"}]"#);
3360
3361        // Re-upserting the SAME (monitor, hash) with a new URL/path must not add a
3362        // row and must NOT wipe the cached OCR result (idempotent on the unique key).
3363        store
3364            .community_post_upsert(mid, "youtube_community_ocr", "https://x/b.jpg", "1111", "C:/c/1111b", 200)
3365            .unwrap();
3366        assert_eq!(store.community_post_count(mid).unwrap(), 1);
3367        let got = store.community_post_get(mid, "1111").unwrap().unwrap();
3368        assert!(got.ocr_attempted, "re-download must not clear prior OCR");
3369        assert_eq!(got.decoded_events, 1);
3370        assert_eq!(got.local_path, "C:/c/1111b", "path updated to the latest download");
3371
3372        // A different content hash is a distinct archived image.
3373        store
3374            .community_post_upsert(mid, "youtube_community_ocr", "https://x/c.jpg", "2222", "C:/c/2222", 300)
3375            .unwrap();
3376        assert_eq!(store.community_post_count(mid).unwrap(), 2);
3377        // The new image hasn't been OCR'd yet.
3378        assert!(!store.community_post_get(mid, "2222").unwrap().unwrap().ocr_attempted);
3379
3380        // The same content hash under a DIFFERENT monitor is independent (the unique
3381        // index is on the pair, and each monitor archives its own copy).
3382        let mid2 = store.insert_monitor(&m).unwrap();
3383        store
3384            .community_post_upsert(mid2, "youtube_community_ocr", "https://x/a.jpg", "1111", "C:/c/1111", 100)
3385            .unwrap();
3386        assert_eq!(store.community_post_count(mid2).unwrap(), 1);
3387        // mid2's copy is its own un-OCR'd row; mid's stays OCR'd.
3388        assert!(!store.community_post_get(mid2, "1111").unwrap().unwrap().ocr_attempted);
3389        assert!(store.community_post_get(mid, "1111").unwrap().unwrap().ocr_attempted);
3390
3391        // Deleting a monitor cascades to its archived posts (FK ON DELETE CASCADE).
3392        store.delete_monitor(mid).unwrap();
3393        assert_eq!(store.community_post_count(mid).unwrap(), 0);
3394        assert!(store.community_post_get(mid, "1111").unwrap().is_none());
3395        // The other monitor's archive is untouched.
3396        assert_eq!(store.community_post_count(mid2).unwrap(), 1);
3397    }
