1//! Build identity, baked in at compile time by `build.rs`.
2//!
3//! [`build_id`] is stamped into every detached download's registry row
4//! (`detached_process.spawn_build`) so a later launch knows exactly which build
5//! started an in-flight download and can apply per-build compat fixups when it
6//! re-attaches (see [`crate::compat`]). The local patch counter in `APP_VERSION`
7//! increments on every rebuild, so this distinguishes even successive dev builds
8//! of the same commit.
9
10/// A stable identifier for the running build, e.g. `0.1.42/a1b2c3d4e-dirty`.
11/// `APP_VERSION` is `<major>.<minor>.<local-build-counter>`; `GIT_HASH` is the
12/// short commit hash with a `-dirty` suffix when tracked files were modified.
13pub fn build_id() -> &'static str {
14    concat!(env!("APP_VERSION"), "/", env!("GIT_HASH"))
15}
16
17#[cfg(test)]
18mod tests {
19    #[test]
20    fn build_id_is_version_slash_git() {
21        let id = super::build_id();
22        let (version, git) = id
23            .split_once('/')
24            .unwrap_or_else(|| panic!("build_id should be <version>/<git>, got {id:?}"));
25        // Version is dotted (major.minor.patch); both parts are non-empty.
26        assert!(version.contains('.'), "version part: {version:?}");
27        assert!(!git.is_empty(), "git part empty in {id:?}");
28    }
29}
