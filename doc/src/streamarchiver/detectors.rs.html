200/// FNV-1a 64-bit hash — simple, stable, and fast; used instead of `DefaultHasher`
201/// (which is not guaranteed stable across Rust versions) for the persisted OCR
202/// image cache.
203fn fnv64(data: &[u8]) -> u64 {
204    const PRIME: u64 = 1_099_511_628_211;
205    const BASIS: u64 = 14_695_981_039_346_656_037;
206    let mut h = BASIS;
207    for &b in data {
208        h ^= b as u64;
209        h = h.wrapping_mul(PRIME);
210    }
211    h
212}

215    pub fn new(store: Arc<Store>, events: EventTx) -> DetectContext {
216        let browser = store
217            .get_setting("cookies_browser")
218            .ok()
219            .flatten()
220            .unwrap_or_default();
221        let browser_name = browser.split(':').next().unwrap_or("chrome");
222        let fingerprint = build_browser_fingerprint(if browser_name.is_empty() {
223            "chrome"
224        } else {
225            browser_name
226        });
227        let http = Client::builder()
228            .user_agent(fingerprint.ua.clone())
229            .timeout(Duration::from_secs(20))
230            .build()
231            .expect("building reqwest client");
232
233        // Pre-populate the OCR cache from the persisted hash map so unchanged
234        // images are not re-OCR'd after an app restart. The segments come from
235        // the DB (they were already stored by the previous OCR run).
236        let ocr_cache = {
237            let hashes: HashMap<String, u64> = store
238                .get_setting(crate::models::K_OCR_IMAGE_HASHES)
239                .ok()
240                .flatten()
241                .and_then(|s| serde_json::from_str(&s).ok())
242                .unwrap_or_default();
243            let mut map: HashMap<(i64, String), (u64, Vec<ScheduleSegment>)> = HashMap::new();
244            for (key_str, hash) in hashes {
245                if let Some((mid_str, source_id)) = key_str.split_once(':') {
246                    if let Ok(monitor_id) = mid_str.parse::<i64>() {
247                        let segs = store
248                            .schedule_segments_for_source(monitor_id, source_id)
249                            .unwrap_or_default();
250                        map.insert((monitor_id, source_id.to_string()), (hash, segs));
251                    }
252                }
253            }
254            Mutex::new(map)
255        };
256
257        DetectContext {
258            http,
259            store,
260            events,
261            twitch_token: Mutex::new(None),
262            kick_token: Mutex::new(None),
263            twitch_refresh: Mutex::new(()),
264            fingerprint,
265            ocr_cache,
266        }
267    }

336    async fn twitch_app_token(&self) -> Result<String> {
337        if let Some(tok) = self.twitch_token.lock().await.as_ref() {
338            if tok.expires_at > Instant::now() {
339                return Ok(tok.access_token.clone());
340            }
341        }
342        let client_id = self
343            .store
344            .get_setting("twitch_client_id")?
345            .unwrap_or_default();
346        let client_secret = self
347            .store
348            .get_setting("twitch_client_secret")?
349            .unwrap_or_default();
350        if client_id.is_empty() || client_secret.is_empty() {
351            bail!("Twitch credentials not set (Settings)");
352        }
353
354        let resp = self
355            .http
356            .post("https://id.twitch.tv/oauth2/token")
357            .form(&[
358                ("client_id", client_id.as_str()),
359                ("client_secret", client_secret.as_str()),
360                ("grant_type", "client_credentials"),
361            ])
362            .send()
363            .await?;
364        if !resp.status().is_success() {
365            let status = resp.status();
366            let body = resp.text().await.unwrap_or_default();
367            bail!("Twitch token request failed: {status} {body}");
368        }
369
370        #[derive(Deserialize)]
371        struct TokenResp {
372            access_token: String,
373            expires_in: u64,
374        }
375        let tr: TokenResp = resp.json().await?;
376        let token = tr.access_token.clone();
377        // Refresh a minute early.
378        let ttl = Duration::from_secs(tr.expires_in.saturating_sub(60).max(60));
379        *self.twitch_token.lock().await = Some(TwitchToken {
380            access_token: tr.access_token,
381            expires_at: Instant::now() + ttl,
382        });
383        Ok(token)
384    }

386    /// Check whether the connected Twitch account is subscribed to
387    /// `broadcaster_login`. `Some(true)`/`Some(false)` when conclusively known;
388    /// `None` when undeterminable (not connected, missing the
389    /// `user:read:subscriptions` scope / 401, or a lookup error).
390    pub async fn check_twitch_sub(&self, broadcaster_login: &str) -> Option<bool> {
391        // Cheap local gates first, so an account that can't yield a determinate
392        // answer (no stored user id, no Client ID) bails before any network work.
393        let user_id = crate::oauth::connected_user_id(self.store.as_ref())?;
394        let client_id = self
395            .store
396            .get_setting("twitch_client_id")
397            .ok()
398            .flatten()
399            .unwrap_or_default();
400        if client_id.is_empty() {
401            return None;
402        }
403        // Serialize token refresh (device-code refresh tokens are one-time-use), as
404        // detect_twitch does, so a concurrent detection pass can't double-spend it.
405        let token = {
406            let _guard = self.twitch_refresh.lock().await;
407            crate::oauth::valid_user_token(&self.http, self.store.as_ref()).await
408        }?;
409        let broadcaster_id = self
410            .twitch_user_id(&client_id, &token, broadcaster_login)
411            .await?;
412        let resp = self
413            .http
414            .get("https://api.twitch.tv/helix/subscriptions/user")
415            .header("Client-Id", &client_id)
416            .bearer_auth(&token)
417            .query(&[
418                ("broadcaster_id", broadcaster_id.as_str()),
419                ("user_id", user_id.as_str()),
420            ])
421            .send()
422            .await
423            .ok()?;
424        match resp.status() {
425            s if s.is_success() => Some(true),
426            reqwest::StatusCode::NOT_FOUND => Some(false), // 404 = not subscribed
427            // 400 is returned when broadcaster_id == user_id (you can't subscribe to
428            // your own channel). That's conclusive ("no sub benefit"), so cache it
429            // instead of re-querying this monitor every refresh pass forever.
430            reqwest::StatusCode::BAD_REQUEST => Some(false),
431            _ => None, // 401 (scope missing/expired), 5xx -> unknown (retry later)
432        }
433    }

453    pub async fn detect_twitch(&self, items: &[DetectItem]) -> Vec<DetectOutcome> {
454        let client_id = self
455            .store
456            .get_setting("twitch_client_id")
457            .ok()
458            .flatten()
459            .unwrap_or_default();
460
461        let mut outcomes = Vec::new();
462        let mut login_to_mons: HashMap<String, Vec<i64>> = HashMap::new();
463        for it in items {
464            match twitch_login(&it.url) {
465                Some(login) => login_to_mons.entry(login).or_default().push(it.monitor_id),
466                None => outcomes.push(DetectOutcome::err(
467                    it.monitor_id,
468                    "cannot parse twitch login",
469                )),
470            }
471        }
472
473        // Resolve an auth token: a connected user token (refreshed if needed,
474        // serialized because device-code refresh tokens are one-time-use), else an
475        // app token (client-credentials, which needs the Client Secret).
476        let user_token = {
477            let _guard = self.twitch_refresh.lock().await;
478            crate::oauth::valid_user_token(&self.http, self.store.as_ref()).await
479        };
480        let mut using_user_token = user_token.is_some();
481        let mut token = match user_token {
482            Some(t) => t,
483            None => match self.twitch_app_token().await {
484                Ok(t) => t,
485                Err(e) => {
486                    let msg = twitch_auth_error(&e.to_string());
487                    for mons in login_to_mons.values() {
488                        for mid in mons {
489                            outcomes.push(DetectOutcome::err(*mid, msg.clone()));
490                        }
491                    }
492                    return outcomes;
493                }
494            },
495        };
496
497        #[derive(Deserialize)]
498        struct Stream {
499            id: String,
500            user_login: String,
501            user_id: String,
502            thumbnail_url: String,
503            #[serde(rename = "type")]
504            kind: String,
505            started_at: Option<String>,
506            title: Option<String>,
507        }
508        #[derive(Deserialize)]
509        struct StreamsResp {
510            data: Vec<Stream>,
511        }
512
513        let logins: Vec<String> = login_to_mons.keys().cloned().collect();
514        for chunk in logins.chunks(100) {
515            let query: Vec<(&str, &str)> =
516                chunk.iter().map(|l| ("user_login", l.as_str())).collect();
517            // Up to two attempts: the chosen token, then an app-token fallback if a
518            // user token is rejected (stale/revoked/Client-Id mismatch -> 401).
519            loop {
520                let resp = self
521                    .http
522                    .get("https://api.twitch.tv/helix/streams")
523                    .header("Client-Id", &client_id)
524                    .bearer_auth(&token)
525                    .query(&query)
526                    .send()
527                    .await;
528
529                match resp {
530                    Ok(r) if r.status().is_success() => {
531                        // login -> (went_live, stream_id, user_id, thumbnail_url, title)
532                        let live: HashMap<String, (Option<i64>, Option<String>, String, String, Option<String>)> =
533                            match r.json::<StreamsResp>().await
534                        {
535                            Ok(sr) => sr
536                                .data
537                                .into_iter()
538                                .filter(|s| s.kind == "live")
539                                .map(|s| {
540                                    let when = s.started_at.as_deref().and_then(parse_rfc3339);
541                                    (
542                                        s.user_login.to_lowercase(),
543                                        (when, Some(s.id), s.user_id, s.thumbnail_url, s.title),
544                                    )
545                                })
546                                .collect(),
547                            Err(e) => {
548                                for l in chunk {
549                                    for mid in &login_to_mons[l] {
550                                        outcomes.push(DetectOutcome::err(
551                                            *mid,
552                                            format!("helix parse: {e}"),
553                                        ));
554                                    }
555                                }
556                                break;
557                            }
558                        };
559                        for l in chunk {
560                            let key = l.to_lowercase();
561                            for mid in &login_to_mons[l] {
562                                outcomes.push(match live.get(&key) {
563                                    Some((went, id, uid, thumb, title)) => {
564                                        DetectOutcome::live_at(*mid, "live", *went)
565                                            .with_stream_id(id.clone())
566                                            .with_broadcaster_id(Some(uid.clone()))
567                                            .with_thumbnail_url(Some(thumb.clone()))
568                                            .with_stream_title(title.clone())
569                                    }
570                                    None => DetectOutcome::offline(*mid),
571                                });
572                            }
573                        }
574                        break;
575                    }
576                    // A user token was rejected — fall back to the app token once.
577                    Ok(r)
578                        if r.status() == reqwest::StatusCode::UNAUTHORIZED && using_user_token =>
579                    {
580                        match self.twitch_app_token().await {
581                            Ok(app) => {
582                                token = app;
583                                using_user_token = false;
584                                continue;
585                            }
586                            Err(_) => {
587                                for l in chunk {
588                                    for mid in &login_to_mons[l] {
589                                        outcomes.push(DetectOutcome::err(
590                                            *mid,
591                                            "Twitch token expired — reconnect in Settings, or set a Client Secret",
592                                        ));
593                                    }
594                                }
595                                break;
596                            }
597                        }
598                    }
599                    Ok(r) => {
600                        let status = r.status();
601                        let msg = if status == reqwest::StatusCode::UNAUTHORIZED {
602                            "Twitch auth rejected (401) — reconnect in Settings or check Client ID/Secret".to_string()
603                        } else {
604                            format!("helix {status}")
605                        };
606                        for l in chunk {
607                            for mid in &login_to_mons[l] {
608                                outcomes.push(DetectOutcome::err(*mid, msg.clone()));
609                            }
610                        }
611                        break;
612                    }
613                    Err(e) => {
614                        for l in chunk {
615                            for mid in &login_to_mons[l] {
616                                outcomes.push(DetectOutcome::err(*mid, e.to_string()));
617                            }
618                        }
619                        break;
620                    }
621                }
622            }
623        }
624        outcomes
625    }

700    /// A Twitch channel's upcoming scheduled streams via Helix `Get Channel Stream
701    /// Schedule`. Mirrors [`twitch_stream_meta`](Self::twitch_stream_meta)'s token
702    /// handling. `Some(vec)` on success (empty when no schedule is set up — Helix
703    /// returns 404); `None` on error / missing credentials, so the refresher won't
704    /// wipe a previously-fetched schedule on a transient failure.
705    pub async fn twitch_schedule(&self, url: &str) -> Option<Vec<ScheduleSegment>> {
706        let login = twitch_login(url)?;
707        let client_id = self
708            .store
709            .get_setting("twitch_client_id")
710            .ok()
711            .flatten()
712            .unwrap_or_default();
713        if client_id.is_empty() {
714            return None;
715        }
716        let user_token = {
717            let _guard = self.twitch_refresh.lock().await;
718            crate::oauth::valid_user_token(&self.http, self.store.as_ref()).await
719        };
720        let mut using_user_token = user_token.is_some();
721        let mut token = match user_token {
722            Some(t) => t,
723            None => self.twitch_app_token().await.ok()?,
724        };
725        // Resolve the broadcaster id; if a user token is rejected here, fall back
726        // to the app token (the 401 loop below only covers the schedule call).
727        let broadcaster_id = match self.twitch_user_id(&client_id, &token, &login).await {
728            Some(id) => id,
729            None if using_user_token => {
730                token = self.twitch_app_token().await.ok()?;
731                using_user_token = false;
732                self.twitch_user_id(&client_id, &token, &login).await?
733            }
734            None => return None,
735        };
736
737        #[derive(Deserialize)]
738        struct Cat {
739            #[serde(default)]
740            name: String,
741        }
742        #[derive(Deserialize)]
743        struct Seg {
744            start_time: Option<String>,
745            end_time: Option<String>,
746            #[serde(default)]
747            title: String,
748            canceled_until: Option<String>,
749            category: Option<Cat>,
750        }
751        #[derive(Deserialize)]
752        struct Data {
753            #[serde(default)]
754            segments: Vec<Seg>,
755        }
756        #[derive(Deserialize)]
757        struct Resp {
758            data: Data,
759        }
760
761        loop {
762            let resp = self
763                .http
764                .get("https://api.twitch.tv/helix/schedule")
765                .header("Client-Id", &client_id)
766                .bearer_auth(&token)
767                .query(&[("broadcaster_id", broadcaster_id.as_str()), ("first", "25")])
768                .send()
769                .await
770                .ok()?;
771            match resp.status() {
772                s if s.is_success() => {
773                    let r: Resp = resp.json().await.ok()?;
774                    let segs = r
775                        .data
776                        .segments
777                        .into_iter()
778                        .filter_map(|seg| {
779                            let start = seg.start_time.as_deref().and_then(parse_rfc3339)?;
780                            Some(ScheduleSegment {
781                                id: 0,
782                                monitor_id: 0,
783                                start_time: start,
784                                end_time: seg.end_time.as_deref().and_then(parse_rfc3339),
785                                title: seg.title,
786                                category: seg.category.map(|c| c.name).unwrap_or_default(),
787                                canceled: seg.canceled_until.is_some(),
788                                video_id: None,
789                            })
790                        })
791                        .collect();
792                    return Some(segs);
793                }
794                // The broadcaster hasn't set up a schedule.
795                reqwest::StatusCode::NOT_FOUND => return Some(Vec::new()),
796                reqwest::StatusCode::UNAUTHORIZED if using_user_token => {
797                    token = self.twitch_app_token().await.ok()?;
798                    using_user_token = false;
799                    continue;
800                }
801                _ => return None,
802            }
803        }
804    }

806    /// A YouTube channel's upcoming/scheduled livestreams, scraped from its
807    /// `/streams` page (no credentials, no quota). `Some(vec)` on a successful
808    /// fetch (possibly empty); `None` on a network/HTTP error.
809    pub async fn youtube_schedule(&self, url: &str) -> Option<Vec<ScheduleSegment>> {
810        let jitter_ms = (std::time::SystemTime::now()
811            .duration_since(std::time::UNIX_EPOCH)
812            .unwrap_or_default()
813            .subsec_nanos()
814            % 2000) as u64;
815        tokio::time::sleep(Duration::from_millis(jitter_ms)).await;
816        let streams_url = youtube_streams_url(url);
817        let rb = self
818            .http
819            .get(&streams_url)
820            // Force English + US locale. YouTube geo-localizes the server-rendered
821            // schedule strings to the viewer's IP, so a non-US IP yields e.g. the
822            // Norwegian "Planlagt for 18.06.2026, 20:00" instead of the
823            // "Scheduled for 6/18/26, 8:00 PM" that `parse_yt_scheduled_text`
824            // expects — every segment was silently dropped. The `hl`/`gl` query
825            // params override that more reliably than `Accept-Language` alone.
826            .query(&[("hl", "en"), ("gl", "US")])
827            .header("Accept-Language", "en-US,en;q=0.9")
828            .header("Cookie", "CONSENT=YES+1; SOCS=CAI");
829        let resp = self.fingerprint.apply_yt_nav_headers(rb).send().await.ok()?;
830        if !resp.status().is_success() {
831            return None;
832        }
833        let body = resp.text().await.ok()?;
834        let mut segs = parse_youtube_schedule(&body);
835        // Shape-independent fallback: if the structured `ytInitialData` walk found
836        // nothing (Polymer markup drift), scan the raw page for scheduled-start
837        // markers so a layout change doesn't silently zero out the schedule.
838        if segs.is_empty() {
839            collect_upcoming_fallback(&body, &mut segs);
840        }
841        Some(segs)
842    }

844    /// A YouTube channel's upcoming streams via the Data API (source `YouTubeApi`):
845    /// resolve the channel id, `search.list?eventType=upcoming` for the upcoming
846    /// video ids (~100 quota units), then `videos.list` for each one's exact
847    /// `scheduledStartTime` + title (~1 unit). `Some(vec)` on success (possibly
848    /// empty); `None` on error / missing key, so a transient failure won't wipe a
849    /// stored schedule. Gated by the caller on `youtube_api_enabled(K_YT_API_SCHEDULE)`.
850    pub async fn youtube_schedule_api(&self, url: &str) -> Option<Vec<ScheduleSegment>> {
851        let key = self
852            .store
853            .get_setting("youtube_api_key")
854            .ok()
855            .flatten()
856            .unwrap_or_default();
857        if key.is_empty() {
858            return None;
859        }
860        if self.youtube_quota_exceeded() {
861            debug!("YouTube schedule API skipped — daily quota limit reached");
862            return None;
863        }
864        let channel_id = self.youtube_channel_id(url, &key).await.ok()?;
865        // search.list for upcoming live broadcasts on this channel.
866        let resp = self
867            .http
868            .get("https://www.googleapis.com/youtube/v3/search")
869            .query(&[
870                ("part", "id"),
871                ("channelId", channel_id.as_str()),
872                ("eventType", "upcoming"),
873                ("type", "video"),
874                ("maxResults", "50"),
875                ("key", key.as_str()),
876            ])
877            .send()
878            .await
879            .ok()?;
880        if !resp.status().is_success() {
881            return None;
882        }
883        let _ = self.store.record_quota_usage("youtube", 100);
884        let _ = self.store.record_quota_usage("youtube_search", 1);
885        let v: Value = resp.json().await.ok()?;
886        let ids: Vec<String> = v["items"]
887            .as_array()
888            .into_iter()
889            .flatten()
890            .filter_map(|it| it["id"]["videoId"].as_str().map(str::to_string))
891            .collect();
892        if ids.is_empty() {
893            // The API definitively reports no upcoming streams.
894            return Some(Vec::new());
895        }
896        // videos.list for exact scheduled start + title (one batched call).
897        let id_refs: Vec<&str> = ids.iter().map(String::as_str).collect();
898        let starts = self.youtube_videos_list(&id_refs).await?;
899        let titles = self.youtube_video_titles(&id_refs, &key).await.unwrap_or_default();
900        let mut out: Vec<ScheduleSegment> = ids
901            .iter()
902            .filter_map(|id| {
903                let start = *starts.get(id)?;
904                Some(ScheduleSegment {
905                    id: 0,
906                    monitor_id: 0,
907                    start_time: start,
908                    end_time: None,
909                    title: titles.get(id).cloned().unwrap_or_default(),
910                    category: String::new(),
911                    canceled: false,
912                    video_id: Some(id.clone()),
913                })
914            })
915            .collect();
916        out.sort_by(|a, b| a.start_time.cmp(&b.start_time).then_with(|| a.title.cmp(&b.title)));
917        out.dedup_by(|a, b| a.start_time == b.start_time && a.title == b.title);
918        Some(out)
919    }

921    /// Batch `videos.list?part=snippet` to fetch each video's title. Returns a map
922    /// of `video_id → title` (~1 quota unit per 50 ids). `None` on error.

930        }

936                .get("https://www.googleapis.com/youtube/v3/videos")

946            for item in v["items"].as_array().into_iter().flatten() {

948                    (item["id"].as_str(), item["snippet"]["title"].as_str())

983    /// OCR an on-disk image into schedule segments, skipping the (multi-second,
984    /// token-spending) CLI call when the image bytes are unchanged since the last
985    /// pass for this `(monitor, source)`. Returns the cached result on a hash hit.
986    async fn ocr_image_cached(
987        &self,
988        monitor_id: i64,
989        source_id: &str,
990        channel_name: &str,
991        path: &Path,
992        cfg: &ChannelSourceConfig,
993    ) -> Option<Vec<ScheduleSegment>> {
994        let bytes = tokio::fs::read(path).await.ok()?;
995        let hash = fnv64(&bytes);
996        let key = (monitor_id, source_id.to_string());
997        if let Some((cached, segs)) = self.ocr_cache.lock().await.get(&key) {
998            if *cached == hash {
999                debug!("OCR cache hit (monitor {monitor_id}, source {source_id})");
1000                record_ocr_cache_hit(self.store.as_ref());
1001                return Some(segs.clone());
1002            }
1003        }
1004        let opts = ocr_opts_from_settings(self.store.as_ref(), cfg);
1005        info!(
1006            "OCR: scheduling claude call (monitor {monitor_id}, source {source_id}, image {})",
1007            path.display()
1008        );
1009
1010        let source_label = ScheduleSourceKind::from_id(source_id)
1011            .map(|k| k.label())
1012            .unwrap_or(source_id);
1013        let detail = format!("{source_label} · {}", opts.model);
1014        let result = self
1015            .ocr_one_with_events(channel_name, detail, path, &opts)
1016            .await;
1017
1018        let segs = result.segments?;
1019        self.ocr_cache.lock().await.insert(key.clone(), (hash, segs.clone()));
1020        // Persist the hash so this cache hit survives an app restart.
1021        self.persist_ocr_hash(key.0, &key.1, hash);
1022        Some(segs)
1023    }

1025    /// Run one OCR call on `path`, emitting the background-task start/finish events
1026    /// and accumulating CLI stats. `detail` is the task's detail line. Returns the
1027    /// raw run result so the caller decides how to cache/persist it — shared by the
1028    /// single-image cache path and the multi-post community walk.
1029    async fn ocr_one_with_events(
1030        &self,
1031        channel_name: &str,
1032        detail: String,
1033        path: &Path,
1034        opts: &crate::schedule_ocr::OcrOpts,
1035    ) -> crate::schedule_ocr::OcrRunResult {
1036        let task_id = crate::events::next_task_id();
1037        let _ = self.events.send(crate::events::AppEvent::BackgroundTaskStarted(
1038            crate::events::BackgroundTask {
1039                id: task_id,
1040                kind: crate::events::BackgroundTaskKind::OcrCall,
1041                label: channel_name.to_string(),
1042                detail,
1043                started_at: now_unix(),
1044                progress: None,
1045                progress_info: None,
1046            },
1047        ));
1048
1049        let result = ocr_schedule_image(path, opts).await;
1050        accumulate_ocr_stats(self.store.as_ref(), &result);
1051
1052        let outcome = match &result.segments {
1053            Some(segs) => {
1054                let n = segs.len();
1055                let note = if n == 1 {
1056                    "1 event decoded".to_string()
1057                } else if n > 1 {
1058                    format!("{n} events decoded")
1059                } else {
1060                    // Parsed OK but the model found nothing — likely a non-schedule image
1061                    // or all cards had vague/null times.
1062                    "0 events (nothing found)".to_string()
1063                };
1064                crate::events::TaskOutcome::CompletedWithNote(note)
1065            }
1066            None => {
1067                if result.cli_failures > 0 && result.cli_calls.is_empty() {
1068                    crate::events::TaskOutcome::Failed(format!(
1069                        "CLI failed — is '{}' on PATH?",
1070                        opts.command
1071                    ))
1072                } else {
1073                    crate::events::TaskOutcome::Failed("Parse failed".into())
1074                }
1075            }
1076        };
1077        let _ = self
1078            .events
1079            .send(crate::events::AppEvent::BackgroundTaskFinished { id: task_id, outcome });
1080        result
1081    }

1138    /// OCR images on the channel's recent YouTube community posts. Scans up to the
1139    /// configured backlog depth ([`community_max_posts`]) in order, returning the
1140    /// first that decodes a schedule.
1141    ///
1142    /// Two layers of caching keep this cheap. An in-memory combined-URL hash of the
1143    /// whole post set short-circuits the entire pass when nothing changed (no
1144    /// downloads, no OCR). When the set *has* changed (typically one new post
1145    /// pushing the others down the feed), every pulled image is archived to a
1146    /// content-addressed file + `community_post_archive` row; an image whose bytes
1147    /// match an already-decoded archive entry reuses that result instead of
1148    /// re-OCR'ing — so only genuinely new images spend tokens.
1149    async fn ocr_youtube_community(
1150        &self,
1151        row: &MonitorWithChannel,
1152        cfg: &ChannelSourceConfig,
1153    ) -> Option<Vec<ScheduleSegment>> {
1154        let max_posts = community_max_posts(self.store.as_ref(), cfg);
1155
1156        let community_url = youtube_community_url(&row.monitor.url);
1157        let jitter_ms = (std::time::SystemTime::now()
1158            .duration_since(std::time::UNIX_EPOCH)
1159            .unwrap_or_default()
1160            .subsec_nanos()
1161            % 2000) as u64;
1162        tokio::time::sleep(Duration::from_millis(jitter_ms)).await;
1163        let rb = self
1164            .http
1165            .get(&community_url)
1166            .query(&[("hl", "en"), ("gl", "US")])
1167            .header("Accept-Language", "en-US,en;q=0.9")
1168            .header("Cookie", "CONSENT=YES+1; SOCS=CAI");
1169        let resp = self.fingerprint.apply_yt_nav_headers(rb).send().await.ok()?;
1170        if !resp.status().is_success() {
1171            return None;
1172        }
1173        let body = resp.text().await.ok()?;
1174        let data = extract_json_after(&body, "ytInitialData")?;
1175
1176        let mut imgs: Vec<String> = Vec::new();
1177        community_images(&data, &mut imgs);
1178        imgs.truncate(max_posts);
1179        if imgs.is_empty() {
1180            return None;
1181        }
1182
1183        // Combined URL hash: stable identifier for the current set of posts.
1184        // If no post URL has changed the hash matches and we skip the whole pass.
1185        let url_bytes: Vec<u8> = imgs
1186            .iter()
1187            .flat_map(|u| u.as_bytes().iter().copied())
1188            .collect();
1189        let combined_hash = fnv64(&url_bytes);
1190        let source_id = ScheduleSourceKind::YouTubeCommunityOcr.id();
1191        let cache_key = (row.monitor.id, source_id.to_string());
1192        {
1193            let guard = self.ocr_cache.lock().await;
1194            if let Some((cached, segs)) = guard.get(&cache_key) {
1195                if *cached == combined_hash {
1196                    debug!(
1197                        "OCR community cache hit (monitor {}, {} posts unchanged)",
1198                        row.monitor.id,
1199                        imgs.len()
1200                    );
1201                    record_ocr_cache_hit(self.store.as_ref());
1202                    return Some(segs.clone());
1203                }
1204            }
1205        }
1206
1207        // Set changed: pull + archive every post image (durable record), then OCR
1208        // in feed order until one decodes a schedule. Unchanged images hit the
1209        // per-image archive cache below and skip OCR.
1210        let now = crate::models::now_unix();
1211        let mut archived: Vec<(String, PathBuf)> = Vec::new();
1212        for img_url in &imgs {
1213            if let Some(entry) = self.archive_community_image(row, img_url, now).await {
1214                archived.push(entry);
1215            }
1216        }
1217        if archived.is_empty() {
1218            return None;
1219        }
1220
1221        let opts = ocr_opts_from_settings(self.store.as_ref(), cfg);
1222        let n = archived.len();
1223        let mut winner: Option<Vec<ScheduleSegment>> = None;
1224        for (i, (content_hash, path)) in archived.iter().enumerate() {
1225            // Per-image archive cache: this exact image already OCR'd?
1226            if let Ok(Some(ap)) = self.store.community_post_get(row.monitor.id, content_hash) {
1227                if ap.ocr_attempted {
1228                    record_ocr_cache_hit(self.store.as_ref());
1229                    let segs: Vec<ScheduleSegment> =
1230                        serde_json::from_str(&ap.decoded_json).unwrap_or_default();
1231                    if !segs.is_empty() {
1232                        winner = Some(segs);
1233                        break;
1234                    }
1235                    continue;
1236                }
1237            }
1238
1239            let detail = if n > 1 {
1240                format!(
1241                    "{} · {} (post {}/{})",
1242                    ScheduleSourceKind::YouTubeCommunityOcr.label(),
1243                    opts.model,
1244                    i + 1,
1245                    n
1246                )
1247            } else {
1248                format!(
1249                    "{} · {}",
1250                    ScheduleSourceKind::YouTubeCommunityOcr.label(),
1251                    opts.model
1252                )
1253            };
1254            let result = self
1255                .ocr_one_with_events(&row.channel.name, detail, path, &opts)
1256                .await;
1257
1258            // Persist the decode (empty included) so an unchanged image is never
1259            // re-OCR'd. `None` (CLI/parse failure) is left un-attempted to retry.
1260            if let Some(segs) = result.segments {
1261                let json = serde_json::to_string(&segs).unwrap_or_else(|_| "[]".to_string());
1262                self.store
1263                    .community_post_set_decoded(
1264                        row.monitor.id,
1265                        content_hash,
1266                        segs.len() as i64,
1267                        &json,
1268                    )
1269                    .ok();
1270                if !segs.is_empty() {
1271                    winner = Some(segs);
1272                    break;
1273                }
1274            }
1275        }
1276
1277        // Cache the pass result (winner, or empty = "nothing found this set") under
1278        // the combined hash so an unchanged feed skips everything next pass.
1279        let segs = winner.unwrap_or_default();
1280        self.ocr_cache
1281            .lock()
1282            .await
1283            .insert(cache_key.clone(), (combined_hash, segs.clone()));
1284        self.persist_ocr_hash(cache_key.0, source_id, combined_hash);
1285        Some(segs)
1286    }

1288    /// Download a community-post image, content-hash its bytes, persist it to a
1289    /// content-addressed path under `schedule_src/`, and upsert its
1290    /// `community_post_archive` row. Returns `(content_hash, local_path)` for the
1291    /// OCR step, or `None` on a download/read failure. Idempotent: re-seeing the
1292    /// same image reuses the on-disk file and just refreshes the archive row.
1293    async fn archive_community_image(
1294        &self,
1295        row: &MonitorWithChannel,
1296        img_url: &str,
1297        fetched_at: i64,
1298    ) -> Option<(String, PathBuf)> {
1299        let ext = url_image_ext(img_url);
1300        // Download to a temp path first so we can hash the bytes before naming.
1301        let tmp = self.schedule_src_path(row, "community_tmp", ext);
1302        crate::assets::download_image(&self.http, img_url, &tmp)
1303            .await
1304            .ok()?;
1305        let bytes = tokio::fs::read(&tmp).await.ok()?;
1306        let content_hash = fnv64(&bytes).to_string();
1307
1308        // Content-addressed final path: every distinct image kept (durable archive).
1309        let dest = self.schedule_src_path(row, &format!("community_{content_hash}"), ext);
1310        if tokio::fs::try_exists(&dest).await.unwrap_or(false) {
1311            // Already archived (identical bytes) — drop the temp, keep the original.
1312            let _ = tokio::fs::remove_file(&tmp).await;
1313        } else if tokio::fs::rename(&tmp, &dest).await.is_err() {
1314            // Rename failed (e.g. cross-device) — fall back to a copy.
1315            let _ = tokio::fs::write(&dest, &bytes).await;
1316            let _ = tokio::fs::remove_file(&tmp).await;
1317        }
1318
1319        self.store
1320            .community_post_upsert(
1321                row.monitor.id,
1322                ScheduleSourceKind::YouTubeCommunityOcr.id(),
1323                img_url,
1324                &content_hash,
1325                &dest.to_string_lossy(),
1326                fetched_at,
1327            )
1328            .ok();
1329        Some((content_hash, dest))
1330    }

1411    /// GET a Discord API endpoint with the user token, parsing JSON. Handles one
1412    /// 429 (rate-limit) retry honoring `retry-after`. `None` on auth/HTTP error.
1413    /// The token is sent raw in the Authorization header (Discord user-token form).
1414    async fn discord_get(&self, token: &str, url: &str) -> Option<Value> {
1415        for attempt in 0..2 {
1416            let resp = self
1417                .http
1418                .get(url)
1419                .header("Authorization", token)
1420                .send()
1421                .await
1422                .ok()?;
1423            match resp.status() {
1424                s if s.is_success() => return resp.json::<Value>().await.ok(),
1425                reqwest::StatusCode::TOO_MANY_REQUESTS if attempt == 0 => {
1426                    let wait = resp
1427                        .headers()
1428                        .get("retry-after")
1429                        .and_then(|v| v.to_str().ok())
1430                        .and_then(|s| s.parse::<f64>().ok())
1431                        .unwrap_or(2.0)
1432                        .clamp(0.0, 10.0);
1433                    tokio::time::sleep(Duration::from_millis((wait * 1000.0) as u64 + 250)).await;
1434                }
1435                _ => return None,
1436            }
1437        }
1438        None
1439    }

1454    /// One guild's upcoming scheduled events as [`DiscordEvt`] (with matchable text).
1455    async fn discord_guild_events(&self, token: &str, guild_id: &str) -> Option<Vec<DiscordEvt>> {

1472                start_time: start,
1473                end_time: e["scheduled_end_time"].as_str().and_then(parse_rfc3339),
1474                title: name.clone(),
1475                canceled: status == 4,
1476                text: format!("{name}\n{desc}\n{loc}").to_lowercase(),
1477            });
1478        }
1479        Some(out)

1482    /// Sweep the user's Discord guilds for scheduled events and match them to
1483    /// monitors by the stream URL appearing in the event (location/description/
1484    /// name). Returns `(monitor_id -> events, complete)` where `complete` is true
1485    /// only if every guild was fetched successfully — the caller reconciles
1486    /// (clears unmatched) only on a complete sweep, so a partial outage never wipes
1487    /// a streamer whose guild happened to fail. `None` if the guild list itself
1488    /// couldn't be fetched (auth/network). Paced + shutdown-aware.
1489    async fn discord_sweep(
1490        &self,
1491        rows: &[MonitorWithChannel],
1492        shutdown: &Arc<AtomicBool>,
1493    ) -> Option<(HashMap<i64, Vec<ScheduleSegment>>, bool)> {
1494        let token = self
1495            .store
1496            .get_setting(K_DISCORD_TOKEN)
1497            .ok()
1498            .flatten()
1499            .unwrap_or_default();
1500        if token.is_empty() {
1501            return None;
1502        }
1503        let guilds = self.discord_guild_ids(&token).await?;
1504        // URL needles per enabled monitor (skip monitors we can't match by URL).
1505        let needles: Vec<(i64, Vec<String>)> = rows
1506            .iter()
1507            .filter(|r| r.channel.enabled && r.monitor.enabled)
1508            .map(|r| (r.monitor.id, monitor_needles(&r.monitor.url)))
1509            .filter(|(_, n)| !n.is_empty())
1510            .collect();
1511        if needles.is_empty() {
1512            return Some((HashMap::new(), true));
1513        }
1514        let mut matched: HashMap<i64, Vec<ScheduleSegment>> = HashMap::new();
1515        let guild_count = guilds.len();
1516        let mut ok_guilds = 0usize;
1517        for gid in guilds {
1518            if shutdown.load(Ordering::SeqCst) {
1519                return None;
1520            }
1521            // Gentle pacing so the sweep isn't a bursty rate-limit magnet.
1522            tokio::time::sleep(Duration::from_millis(250)).await;
1523            let Some(events) = self.discord_guild_events(&token, &gid).await else {
1524                continue;
1525            };
1526            ok_guilds += 1;
1527            for ev in events {
1528                for (mid, ns) in &needles {
1529                    if ns.iter().any(|n| text_contains_token(&ev.text, n)) {
1530                        matched.entry(*mid).or_default().push(ScheduleSegment {
1531                            id: 0,
1532                            monitor_id: 0,
1533                            start_time: ev.start_time,
1534                            end_time: ev.end_time,
1535                            title: ev.title.clone(),
1536                            category: String::new(),
1537                            canceled: ev.canceled,
1538                            video_id: None,
1539                        });
1540                    }
1541                }
1542            }
1543        }
1544        // If the guild list was non-empty but every per-guild fetch failed, treat
1545        // the sweep as failed (return None) rather than silently wiping stored
1546        // Discord events on a transient outage.
1547        if guild_count > 0 && ok_guilds == 0 {
1548            return None;
1549        }
1550        Some((matched, ok_guilds == guild_count))
1551    }

1831    async fn scrape_youtube(&self, item: &DetectItem) -> DetectOutcome {
1832        let jitter_ms = (std::time::SystemTime::now()
1833            .duration_since(std::time::UNIX_EPOCH)
1834            .unwrap_or_default()
1835            .subsec_nanos()
1836            % 2000) as u64;
1837        tokio::time::sleep(Duration::from_millis(jitter_ms)).await;
1838        let url = youtube_live_url(&item.url);
1839        let rb = self
1840            .http
1841            .get(&url)
1842            .header("Accept-Language", "en-US,en;q=0.9")
1843            // Bypass the EU consent interstitial that otherwise replaces the page.
1844            .header("Cookie", "CONSENT=YES+1; SOCS=CAI");
1845        let resp = self.fingerprint.apply_yt_nav_headers(rb).send().await;
1846        match resp {
1847            Ok(r) => {
1848                let body = r.text().await.unwrap_or_default();
1849                // Prefer the structured player response: `videoDetails.isLive` is
1850                // authoritative and stays `false` for ended/upcoming streams even
1851                // when other live-related strings (e.g. `hqdefault_live`, or
1852                // `"isLive":true` in badge/DVR metadata nodes) still appear on the
1853                // page for a while after a stream ends.  This stops the scrape from
1854                // returning a false positive that immediately re-triggers a recording
1855                // attempt on a just-concluded stream.
1856                let pr_opt = extract_json_after(&body, "ytInitialPlayerResponse");
1857                let live = if let Some(pr) = &pr_opt {
1858                    pr["videoDetails"]["isLive"].as_bool().unwrap_or(false)
1859                } else {
1860                    // Fallback: structured data absent (degraded/bot-challenged
1861                    // page, network truncation). The string probes are less
1862                    // precise but better than silently returning offline.
1863                    body.contains("hqdefault_live")
1864                        || body.contains("\"isLive\":true")
1865                        || body.contains("\"isLiveNow\":true")
1866                };
1867                if live {
1868                    let (broadcaster_id, thumbnail_url, video_id) = if let Some(pr) = &pr_opt {
1869                        let ch_id =
1870                            pr["videoDetails"]["channelId"].as_str().map(str::to_string);
1871                        let thumb = pr["videoDetails"]["thumbnail"]["thumbnails"]
1872                            .as_array()
1873                            .and_then(|arr| arr.last())
1874                            .and_then(|t| t["url"].as_str())
1875                            .map(str::to_string);
1876                        let vid = pr["videoDetails"]["videoId"].as_str().map(str::to_string);
1877                        (ch_id, thumb, vid)
1878                    } else {
1879                        (None, None, None)
1880                    };
1881                    DetectOutcome::live(item.monitor_id, "live")
1882                        .with_broadcaster_id(broadcaster_id)
1883                        .with_thumbnail_url(thumbnail_url)
1884                        .with_stream_id(video_id)
1885                } else {
1886                    DetectOutcome::offline(item.monitor_id)
1887                }
1888            }
1889            Err(e) => DetectOutcome::err(item.monitor_id, e.to_string()),
1890        }
1891    }

2050/// One refresh pass: check each Twitch monitor whose cached status is stale,
2051/// de-duplicating per broadcaster login. Only conclusive results are persisted;
2052/// an undeterminable result (e.g. scope not yet granted) is left to retry. Emits
2053/// a bus event when a status actually changes so the UI reloads the column.
2054async fn refresh_ad_free_once(
2055    ctx: &Arc<DetectContext>,
2056    events: &EventTx,
2057    shutdown: &Arc<AtomicBool>,
2058    stale_after: i64,
2059) {
2060    let rows = match ctx.store.twitch_monitors_for_ad_free() {
2061        Ok(r) => r,
2062        Err(_) => return,
2063    };
2064    let now = now_unix();
2065    let mut by_login: HashMap<String, Option<bool>> = HashMap::new();
2066    for row in &rows {
2067        // Quitting: stop doing network + DB work mid-pass.
2068        if shutdown.load(Ordering::SeqCst) {
2069            return;
2070        }
2071        // The manual flag already wins in the UI, so the auto result would never be
2072        // shown — don't spend a Helix call on it.
2073        if row.ad_free {
2074            continue;
2075        }
2076        if let Some(at) = row.ad_free_sub_at {
2077            if now - at < stale_after {
2078                continue; // checked recently
2079            }
2080        }
2081        let Some(login) = twitch_login(&row.url) else {
2082            continue;
2083        };
2084        let result = match by_login.get(&login) {
2085            Some(r) => *r,
2086            None => {
2087                let r = ctx.check_twitch_sub(&login).await;
2088                by_login.insert(login.clone(), r);
2089                r
2090            }
2091        };
2092        // Persist only a conclusive result, atomically gated on still being
2093        // connected: a Disconnect during the await above clears cached results, and
2094        // this guarded write won't resurrect a stale "Yes (sub)" in that race.
2095        if let Some(sub) = result {
2096            let wrote = ctx
2097                .store
2098                .set_monitor_ad_free_sub_if_connected(
2099                    row.id,
2100                    Some(sub),
2101                    now,
2102                    crate::oauth::K_USER_ID,
2103                )
2104                .unwrap_or(false);
2105            if wrote {
2106                info!(monitor_id = row.id, subscribed = sub, "ad-free sub status refreshed");
2107                // Reload the UI only when the displayed status actually changed.
2108                if result != row.ad_free_sub {
2109                    let _ = events.send(AppEvent::MonitorState {
2110                        monitor_id: row.id,
2111                        state: row.last_state.clone(),
2112                    });
2113                }
2114            }
2115        }
2116    }
2117}

2162/// Twitch/YouTube monitors. A short poll tick picks up newly-added monitors
2163/// quickly; each monitor is re-fetched at most every few hours (tracked
2164/// in-memory), and fetches are de-duplicated per URL within a pass.

2167    events: EventTx,

2172    jobs: crate::events::JobRegistry,
2173) {

2175    const TICK_SECS: u64 = 60;
2176    const REFRESH_SECS: u64 = 6 * 3600;

2178    // Persisted across restarts so the 6-hour staleness window is respected even
2179    // after a rebuild/restart — prevents every launch from burning expensive
2180    // `search.list` quota on all YouTube API schedule channels at once.
2181    let mut last_fetched: HashMap<i64, i64> = load_schedule_fetched(ctx.store.as_ref());

2229/// to a base event: ±2h, so a base event borrows the title of the nearest donor

2233/// Whether any non-canceled segment is missing a title (empty/whitespace) — the
2234/// trigger for the title-fill donor walk.

2239/// Fill blank titles on `base` from the nearest-in-time `donor` event whose title
2240/// is non-blank, within `window` seconds (±). Also copies the donor's category
2241/// when `base`'s is empty. Used when a schedule source publishes times but no
2242/// titles (e.g. a bare Twitch schedule) and a lower-priority source (banner /
2243/// community-post OCR) carries the titles. Mutates `base` in place.
2244fn fill_titles(base: &mut [ScheduleSegment], donor: &[ScheduleSegment], window: i64) {
2245    for b in base.iter_mut() {
2246        if b.canceled || !b.title.trim().is_empty() {
2247            continue;
2248        }
2249        let best = donor
2250            .iter()
2251            .filter(|d| !d.title.trim().is_empty())
2252            .map(|d| (d, (d.start_time - b.start_time).abs()))
2253            .filter(|(_, dist)| *dist <= window)
2254            .min_by_key(|(_, dist)| *dist);
2255        if let Some((d, _)) = best {
2256            b.title = d.title.clone();
2257            if b.category.trim().is_empty() && !d.category.trim().is_empty() {
2258                b.category = d.category.clone();
2259            }
2260        }
2261    }

2264/// One schedule-refresh pass over the user's ordered schedule sources.
2265///
2266/// For each enabled monitor due for a refresh, the enabled *per-monitor* sources
2267/// are tried in priority order and the first to return a **non-empty** schedule
2268/// wins: its rows are stored under the source's id and every other source's rows
2269/// for that monitor are cleared, so exactly one source owns each monitor's
2270/// schedule. A source returning `None` is a transient failure (stored rows left
2271/// alone, retried soon); `Some(empty)` is an authoritative "nothing scheduled".
2272/// The effective per-monitor source order and title-fill toggle come from the
2273/// global config plus any per-channel / per-monitor scope override (monitor over
2274/// channel over global).
2275///
2276/// YouTube `/streams` scrape winners are deferred so all their video IDs can be
2277/// refined in one batched `videos.list` call (exact `scheduledStartTime`) before
2278/// being stored under `"youtube"`. Discord runs last as a debounced batch sweep,
2279/// filling only monitors that no higher-priority source already resolved.
2280async fn refresh_schedules_once(
2281    ctx: &Arc<DetectContext>,
2282    events: &EventTx,
2283    shutdown: &Arc<AtomicBool>,
2284    last_fetched: &mut HashMap<i64, i64>,
2285    last_ocr: &mut HashMap<(i64, String), i64>,
2286    discord_last: &mut Option<Instant>,
2287    yt_video_id_refetch: &Arc<AtomicBool>,
2288    refresh_channel: &std::sync::Mutex<Option<i64>>,
2289    refresh_secs: u64,
2290) {
2291    let rows = match ctx.store.list_monitors_with_channels() {
2292        Ok(r) => r,
2293        Err(_) => return,
2294    };
2295    // Drop staleness entries for monitors that no longer exist.
2296    let live: std::collections::HashSet<i64> = rows.iter().map(|r| r.monitor.id).collect();
2297    last_fetched.retain(|id, _| live.contains(id));
2298    last_ocr.retain(|k, _| live.contains(&k.0));
2299    // `now_secs` is the authoritative clock for both `last_fetched` (persisted,
2300    // must survive restarts) and the OCR cadence.  `now` (Instant) is kept only
2301    // for `discord_last`, which is in-memory-only and doesn't need wall-clock.
2302    let now = Instant::now();
2303    let now_secs = now_unix();
2304
2305    // If the UI requested a targeted re-scrape for YouTube monitors whose stored
2306    // schedule segments are missing video IDs, clear those monitors from
2307    // `last_fetched` so they are refreshed this pass.
2308    if yt_video_id_refetch.swap(false, Ordering::SeqCst) {
2309        let missing_ids: std::collections::HashSet<i64> = ctx
2310            .store
2311            .youtube_monitors_missing_video_ids()
2312            .unwrap_or_default()
2313            .into_iter()
2314            .map(|(id, _)| id)
2315            .collect();
2316        last_fetched.retain(|id, _| !missing_ids.contains(id));
2317    }
2318
2319    // If the UI requested a per-channel refresh, restrict this pass to monitors
2320    // belonging to that channel. `last_fetched` is cleared for those monitors so
2321    // they bypass the staleness gate; all other monitors are skipped entirely
2322    // (their schedules stay untouched). `None` = normal full pass.
2323    let only_channel: Option<i64> = refresh_channel.lock().unwrap().take();
2324    if let Some(cid) = only_channel {
2325        for row in rows.iter().filter(|r| r.channel.id == cid) {
2326            last_fetched.remove(&row.monitor.id);
2327        }
2328    }
2329
2330    // The user's global ordered sources, plus per-channel / per-monitor scope
2331    // overrides (loaded once; resolved per monitor in the walk and, below, for the
2332    // Discord batch sweep — so Discord honors the same per-channel/instance control).
2333    let order = load_source_order(ctx.store.as_ref());
2334    let channel_scope = load_channel_scope_map(ctx.store.as_ref());
2335    let monitor_scope = load_monitor_scope_map(ctx.store.as_ref());
2336    let global_fill = global_title_fill(ctx.store.as_ref());
2337    // Per-channel source configs loaded once up front — avoids re-reading and
2338    // re-parsing the same JSON setting key on every iteration of the monitor loop.
2339    let channel_cfg_map = load_channel_cfg_map(ctx.store.as_ref());
2340
2341    // Whether to refine scraped YouTube timestamps with a batched videos.list call.
2342    let yt_api_enabled = ctx.youtube_api_enabled(K_YT_API_SCHEDULE);
2343    // Per-(source, URL) fetch cache: a URL shared across monitors is fetched once
2344    // per source. `None` = transient failure (don't overwrite stored rows).
2345    let mut fetched: HashMap<(&'static str, String), Option<Vec<ScheduleSegment>>> = HashMap::new();
2346    // YouTube scrape winners are deferred so all video IDs batch into one API call.
2347    let mut yt_pending: Vec<(i64, Vec<ScheduleSegment>)> = Vec::new();
2348    let mut changed = false;
2349    // On a pure transient failure, retry sooner than the full interval (but not
2350    // every 60s tick — that would hammer the source).
2351    const TRANSIENT_RETRY_SECS: u64 = 300;
2352    // Expensive OCR sources re-resolve at most this often per (monitor, source),
2353    // independent of `refresh_secs` — so a forced UI refresh re-runs the cheap
2354    // sources at once but never re-OCRs an image processed this recently.
2355    const OCR_MIN_INTERVAL_SECS: u64 = 6 * 3600;
2356
2357    for row in &rows {
2358        if shutdown.load(Ordering::SeqCst) {
2359            return;
2360        }
2361        // When targeting a single channel, skip all other channels entirely.
2362        if let Some(cid) = only_channel {
2363            if row.channel.id != cid {
2364                continue;
2365            }
2366        }
2367        if !row.channel.enabled || !row.monitor.enabled {
2368            continue;
2369        }
2370        // Re-fetch each monitor at most every `refresh_secs` (wall-clock, survives restarts).
2371        if let Some(t) = last_fetched.get(&row.monitor.id) {
2372            if now_secs.saturating_sub(*t) < refresh_secs as i64 {
2373                continue;
2374            }
2375        }
2376        let platform = row.monitor.platform();
2377        let cfg = channel_cfg_map
2378            .get(&row.channel.id.to_string())
2379            .cloned()
2380            .unwrap_or_default();
2381
2382        // Resolve this monitor's effective source order + title-fill toggle from
2383        // the global config and any channel/monitor scope override.
2384        let ch_scope = channel_scope.get(&row.channel.id.to_string());
2385        let mon_scope = monitor_scope.get(&row.monitor.id.to_string());
2386        let eff_order = effective_order_from(&order, ch_scope, mon_scope);
2387        let eff_fill = effective_title_fill_from(global_fill, ch_scope, mon_scope);
2388        let per_monitor: Vec<ScheduleSourceKind> = eff_order
2389            .iter()
2390            .filter(|e| e.enabled)
2391            .filter_map(SourceEntry::kind)
2392            .filter(|k| k.is_per_monitor())
2393            .collect();
2394
2395        // Walk the enabled per-monitor sources in priority order; first non-empty
2396        // result wins. When title-fill is on and the winner has blank titles, keep
2397        // walking lower-priority sources to borrow their titles (nearest in time)
2398        // before stopping. Track whether any source was authoritative (`Some`).
2399        let mut any_authoritative = false;
2400        let mut won: Option<(ScheduleSourceKind, Vec<ScheduleSegment>)> = None;
2401        // True when the winning source read its segments from the DB (OCR cadence
2402        // cache hit) rather than running fresh. In that case the DB already has the
2403        // correct rows and we must NOT call replace_schedule_source — doing so
2404        // would re-insert past segments that the DELETE (start_time >= now) skips,
2405        // causing exponential row accumulation on every tick.
2406        let mut won_is_db_cache = false;
2407        for &kind in &per_monitor {
2408            if !kind.applies_to(platform, &cfg) {
2409                continue;
2410            }
2411
2412            // Expensive OCR sources resolve at most once per OCR_MIN_INTERVAL_SECS
2413            // per (monitor, source), tracked separately from `last_fetched`. Within
2414            // that window we reuse the rows the source last stored (no download, no
2415            // claude call) so it still wins / donates titles exactly as before;
2416            // outside it we re-OCR. This stops a forced UI refresh (refresh_secs ==
2417            // 0, fired on every config save) from re-OCR'ing already-processed
2418            // images. `None` = not an OCR source; `Some(true/false)` = OCR due/not.
2419            let ocr_due = kind.is_ocr().then(|| {
2420                let okey = (row.monitor.id, kind.id().to_string());
2421                let due = last_ocr
2422                    .get(&okey)
2423                    .map(|t| now_secs.saturating_sub(*t) >= OCR_MIN_INTERVAL_SECS as i64)
2424                    .unwrap_or(true);
2425                if due {
2426                    // Stamp at the decision point (write-through to settings) so a
2427                    // crash or restart can't reset the cadence and re-OCR early.
2428                    ctx.persist_ocr_attempt(row.monitor.id, kind.id(), now_secs);
2429                    last_ocr.insert(okey, now_secs);
2430                }
2431                due
2432            });
2433
2434            let segs_opt = if ocr_due == Some(false) {
2435                // Within the cadence window: reuse the rows this source stored last
2436                // run. Empty → skip entirely (an empty cache must not pose as an
2437                // authoritative "nothing scheduled" and clear a real winner).
2438                let cached = ctx
2439                    .store
2440                    .schedule_segments_for_source(row.monitor.id, kind.id())
2441                    .unwrap_or_default();
2442                if cached.is_empty() {
2443                    continue;
2444                }
2445                Some(cached)
2446            } else {
2447                let key = (kind.id(), row.monitor.url.clone());
2448                match fetched.get(&key) {
2449                    Some(s) => s.clone(),
2450                    None => {
2451                        let s = ctx.resolve_source(kind, row, &cfg).await;
2452                        fetched.insert(key, s.clone());
2453                        s
2454                    }
2455                }
2456            };
2457            let Some(segs) = segs_opt else { continue };
2458            any_authoritative = true;
2459            if segs.is_empty() {
2460                continue;
2461            }
2462            match won.as_mut() {
2463                // Winner already chosen; this lower-priority source is a title donor.
2464                Some((_, win_segs)) => {
2465                    fill_titles(win_segs, &segs, TITLE_FILL_WINDOW_SECS);
2466                    if !has_blank_title(win_segs) {
2467                        break;
2468                    }
2469                }
2470                // First non-empty source wins. Stop here unless it has blank titles
2471                // and title-fill is on (then keep walking for donors).
2472                None => {
2473                    let need_titles = eff_fill && has_blank_title(&segs);
2474                    won_is_db_cache = ocr_due == Some(false);
2475                    won = Some((kind, segs));
2476                    if !need_titles {
2477                        break;
2478                    }
2479                }
2480            }
2481        }
2482
2483        // Stamp staleness: full interval on success/authoritative-empty; sooner on a
2484        // pure transient failure (every applicable source returned `None`).
2485        let stamp: i64 = if any_authoritative {
2486            now_secs
2487        } else {
2488            // Back-date so the next check fires after TRANSIENT_RETRY_SECS, not REFRESH_SECS.
2489            now_secs.saturating_sub(refresh_secs.saturating_sub(TRANSIENT_RETRY_SECS) as i64)
2490        };
2491        last_fetched.insert(row.monitor.id, stamp);
2492
2493        match won {
2494            // Defer scrape winners for the batched videos.list refinement below.
2495            Some((ScheduleSourceKind::YouTubeScrape, segs)) => {
2496                yt_pending.push((row.monitor.id, segs));
2497            }
2498            Some((kind, segs)) => {
2499                if won_is_db_cache {
2500                    // Rows already in DB from the last real OCR run — writing them
2501                    // back would duplicate past segments on every tick.
2502                    let _ = ctx.store.clear_other_schedule_sources(row.monitor.id, kind.id());
2503                } else if ctx
2504                    .store
2505                    .replace_schedule_source(row.monitor.id, kind.id(), &segs)
2506                    .is_ok()
2507                {
2508                    let _ = ctx.store.clear_other_schedule_sources(row.monitor.id, kind.id());
2509                    changed = true;
2510                }
2511            }
2512            None => {
2513                // No source won. If at least one was authoritative (returned
2514                // `Some(empty)` — definitively nothing), drop stale per-monitor rows
2515                // but keep Discord (managed by its own sweep). If every source was
2516                // transient (`None`), leave the stored schedule untouched.
2517                if any_authoritative
2518                    && ctx
2519                        .store
2520                        .clear_other_schedule_sources(row.monitor.id, "discord")
2521                        .is_ok()
2522                {
2523                    changed = true;
2524                }
2525            }
2526        }
2527    }
2528
2529    // Batch videos.list for all pending YouTube scrape winners (one call, ~1 quota
2530    // unit). API timestamps supersede the approximate local-time parse from scrape.
2531    if yt_api_enabled && !yt_pending.is_empty() {
2532        let all_ids: Vec<&str> = yt_pending
2533            .iter()
2534            .flat_map(|(_, segs)| segs.iter())
2535            .filter_map(|s| s.video_id.as_deref())
2536            .collect::<std::collections::HashSet<_>>()
2537            .into_iter()
2538            .collect();
2539        if !all_ids.is_empty() {
2540            if let Some(api_times) = ctx.youtube_videos_list(&all_ids).await {
2541                for (_, segs) in &mut yt_pending {
2542                    for seg in segs.iter_mut() {
2543                        if let Some(vid) = seg.video_id.as_deref() {
2544                            if let Some(&t) = api_times.get(vid) {
2545                                seg.start_time = t;
2546                            }
2547                        }
2548                    }
2549                }
2550            }
2551        }
2552    }
2553
2554    // Store YouTube scrape results (after optional API refinement) under "youtube".
2555    for (monitor_id, segs) in &yt_pending {
2556        if ctx.store.replace_schedule_source(*monitor_id, "youtube", segs).is_ok() {
2557            let _ = ctx.store.clear_other_schedule_sources(*monitor_id, "youtube");
2558            changed = true;
2559        }
2560    }
2561
2562    // Discord scheduled-events import — only when enabled in the source list AND a
2563    // token is configured. Runs on the platform cadence, but never more often than
2564    // DISCORD_MIN_SECS even on a forced reload (refresh_secs == 0) — a sweep hits the
2565    // user-token endpoints for every guild, so we debounce it to avoid bursty,
2566    // ban-flaggable traffic.
2567    const DISCORD_MIN_SECS: u64 = 60;
2568    let discord_interval = refresh_secs.max(DISCORD_MIN_SECS);
2569    // Discord runs as one batch sweep (not per-monitor), but still honors per-channel
2570    // / per-instance scope: resolve each monitor's effective order and sweep only the
2571    // monitors where Discord is enabled. Built only when a sweep could actually run
2572    // (token ready + debounce elapsed), so the per-monitor resolve isn't paid every tick.
2573    let discord_ready = ctx.discord_enabled()
2574        && discord_last.is_none_or(|t| now.duration_since(t).as_secs() >= discord_interval);
2575    let discord_monitors: std::collections::HashSet<i64> = if discord_ready {
2576        rows.iter()
2577            .filter(|row| {
2578                let ch_scope = channel_scope.get(&row.channel.id.to_string());
2579                let mon_scope = monitor_scope.get(&row.monitor.id.to_string());
2580                effective_order_from(&order, ch_scope, mon_scope)
2581                    .iter()
2582                    .any(|e| e.enabled && e.kind() == Some(ScheduleSourceKind::Discord))
2583            })
2584            .map(|row| row.monitor.id)
2585            .collect()
2586    } else {
2587        std::collections::HashSet::new()
2588    };
2589    let discord_due = discord_ready && !discord_monitors.is_empty();
2590    if discord_due {
2591        // Stamp the attempt up front so a failing token / outage retries on the
2592        // interval, not every 60s tick (which would hammer Discord's auth endpoint).
2593        *discord_last = Some(now);
2594        if let Some((matched, complete)) = ctx.discord_sweep(&rows, shutdown).await {
2595            // Don't attach a Discord event to a monitor that already resolved a
2596            // schedule from a higher-priority source, so the two never duplicate.
2597            // Use the start-of-today floor the calendar uses, so an in-progress block
2598            // (started earlier today) still counts as a resolved schedule.
2599            let resolved = ctx
2600                .store
2601                .monitors_with_upcoming_non_discord(local_day_start())
2602                .unwrap_or_default();
2603            if complete {
2604                // Full sweep: authoritative — reconcile every monitor (clear ones
2605                // with no matched event).
2606                for row in &rows {
2607                    let mid = row.monitor.id;
2608                    // Clear when Discord is disabled for this monitor's scope, when a
2609                    // higher-priority source already resolved it, or when no event
2610                    // matched; otherwise store the swept events.
2611                    let segs = if !discord_monitors.contains(&mid) || resolved.contains(&mid) {
2612                        Vec::new()
2613                    } else {
2614                        matched.get(&mid).cloned().unwrap_or_default()
2615                    };
2616                    let _ = ctx.store.replace_schedule_source(mid, "discord", &segs);
2617                }
2618            } else {
2619                // Partial sweep: only update monitors we actually got events for, so
2620                // a streamer whose guild failed this pass keeps their stored events.
2621                for (mid, found) in &matched {
2622                    // A monitor with Discord disabled in its scope gets cleared even on
2623                    // a partial sweep, so a per-channel / per-instance opt-out takes
2624                    // effect without waiting for a full reconciliation pass.
2625                    let segs: &[ScheduleSegment] =
2626                        if !discord_monitors.contains(mid) || resolved.contains(mid) {
2627                            &[]
2628                        } else {
2629                            found
2630                        };
2631                    let _ = ctx.store.replace_schedule_source(*mid, "discord", segs);
2632                }
2633            }
2634            changed = true;
2635        }
2636    }
2637
2638    if changed {
2639        // Wake the UI to reload the Next stream column (monitor_id 0 = "any").
2640        let _ = events.send(AppEvent::MonitorState {
2641            monitor_id: 0,
2642            state: String::new(),
2643        });
2644    }
2645
2646    // Persist the updated fetch timestamps so the staleness window survives
2647    // the next restart (avoids a search.list burst on every rebuild/relaunch).
2648    persist_schedule_fetched(ctx.store.as_ref(), last_fetched);
2649}

2700/// Lowercased substrings that, if present in a Discord event's text, identify it as
2701/// belonging to this monitor's channel — the platform-specific `host/path`
2702/// (e.g. `twitch.tv/login`, `youtube.com/@handle`, `kick.com/slug`) plus the bare
2703/// normalized URL. Empty when nothing distinctive can be derived.
2704fn monitor_needles(url: &str) -> Vec<String> {
2705    let mut out: Vec<String> = Vec::new();
2706    match Platform::detect(url) {
2707        Platform::Twitch => {
2708            if let Some(l) = twitch_login(url) {
2709                out.push(format!("twitch.tv/{l}"));
2710            }
2711        }
2712        Platform::YouTube => {
2713            if let Some(h) = youtube_handle(url) {
2714                // Host-qualified only — a bare "@handle" would substring-match
2715                // unrelated @-mentions in event text.
2716                out.push(format!("youtube.com/{}", h.to_lowercase()));
2717            }
2718        }
2719        Platform::Kick => {
2720            if let Some(s) = kick_slug(url) {
2721                out.push(format!("kick.com/{}", s.to_lowercase()));
2722            }
2723        }
2724        Platform::Generic => {}
2725    }
2726    // The bare host/path (scheme + www stripped), as a catch-all / for generic URLs.
2727    let norm = url
2728        .trim()
2729        .trim_start_matches("https://")
2730        .trim_start_matches("http://")
2731        .trim_start_matches("www.")
2732        .trim_end_matches('/')
2733        .to_lowercase();
2734    if norm.len() > 3 && !out.contains(&norm) {
2735        out.push(norm);
2736    }
2737    out
2738}

2740/// Whether `needle` occurs in `haystack` as a whole token — i.e. not immediately
2741/// preceded or followed by an identifier char (alphanumeric / `_` / `-`). This
2742/// stops a needle like `twitch.tv/ana` from matching `twitch.tv/anastasia` (and a
2743/// `youtube.com/@ab` from matching `…@abby`), while still allowing `www.` prefixes
2744/// and trailing punctuation. Both inputs are expected lowercased.
2745fn text_contains_token(haystack: &str, needle: &str) -> bool {
2746    if needle.is_empty() {
2747        return false;
2748    }
2749    let is_ident = |c: char| c.is_ascii_alphanumeric() || c == '_' || c == '-';
2750    let mut from = 0;
2751    while let Some(rel) = haystack[from..].find(needle) {
2752        let i = from + rel;
2753        let before_ok = haystack[..i].chars().next_back().map(is_ident) != Some(true);
2754        let after = i + needle.len();
2755        let after_ok = haystack[after..].chars().next().map(is_ident) != Some(true);
2756        if before_ok && after_ok {
2757            return true;
2758        }
2759        from = i + 1;
2760    }
2761    false
2762}

3269/// must be a standalone token (the char before it is not a JS identifier char, so
3270/// `fooMARKER` is rejected) followed (whitespace-insensitively) by `=` then `{`.
3271/// This rejects unrelated mentions — an identifier *ending* in the marker, a
3272/// quoted key, a minified deref like `a.ytInitialPlayerResponse,…`, an HTML
3273/// attribute's stray `=`, or `marker = null` — instead of latching onto the next
3274/// brace anywhere on the page. Walks every occurrence, returning the first that

3337/// skip, rather than logging a blank title).
3338fn parse_kick_meta(v: &Value) -> Option<StreamMeta> {
3339    let ls = &v["livestream"];
3340    if !ls.is_object() || ls["is_live"].as_bool() == Some(false) {
3341        return None;
3342    }
3343    let title = ls["session_title"]
3344        .as_str()
3345        .filter(|s| !s.is_empty())?
3346        .to_string();
3347    Some(StreamMeta {
3348        title,
3349        // v2 exposes the category under `categories[0].name`.
3350        game: ls["categories"][0]["name"]
3351            .as_str()
3352            .unwrap_or_default()
3353            .to_string(),
3354    })
3355}
