1//! Install OS fonts that cover non-Latin glyphs (CJK, Hangul, fullwidth `【】`,
2//! emoji, etc.) as *fallbacks* behind egui's bundled default.
3//!
4//! egui's default font is Latin-only, so channel names like Japanese VTuber names
5//! (or `Nimi Nightmare【Phase Connect】`) — and the emoji chat viewers spam — otherwise
6//! render as tofu boxes. We read a few fonts already present on the system and append
7//! them after the defaults, so Latin text + the UI icon glyphs keep the default look
8//! and only the missing glyphs fall through to these. Nothing is bundled into the
9//! binary (keeps it lean); if none of the candidates exist we leave the defaults
10//! untouched.
11//!
12//! Emoji caveat: egui's renderer rasterizes glyph *outlines* only — it ignores the
13//! colour tables (COLR/CPAL, sbix, CBDT) in colour-emoji fonts. So emoji render
14//! **monochrome** (the base outline) where the chosen font provides one, and stay
15//! tofu where it only has a colour bitmap. Segoe UI Emoji (Windows) and Noto Emoji
16//! (Linux) carry outlines and render mono; Apple Color Emoji does not, so macOS
17//! falls back to the symbol fonts (partial coverage).
18
19use std::sync::Arc;
20
21use eframe::egui::{self, FontData, FontFamily};

23/// Fallback font groups, in priority order. For each group we load the **first**
24/// file that exists (the entries within a group are equivalent alternatives), so
25/// we don't load several overlapping Japanese fonts. CJK collection files (`.ttc`)
26/// load face 0 (the regular weight), which is what `FontData::from_owned` selects.
27#[cfg(windows)]
28const FONT_GROUPS: &[&[&str]] = &[
29    // Japanese (kana + kanji + CJK punctuation) — primary for VTuber names.
30    &[
31        r"C:\Windows\Fonts\YuGothM.ttc",
32        r"C:\Windows\Fonts\meiryo.ttc",
33        r"C:\Windows\Fonts\msgothic.ttc",
34    ],
35    // Korean (Hangul).
36    &[r"C:\Windows\Fonts\malgun.ttf"],
37    // Simplified Chinese (Han).
38    &[r"C:\Windows\Fonts\msyh.ttc"],
39    // Emoji — Segoe UI Emoji's base glyphs are monochrome outlines the renderer can
40    // rasterize (the COLR colour layers are ignored), so modern emoji show as B&W
41    // silhouettes instead of tofu.
42    &[r"C:\Windows\Fonts\seguiemj.ttf"],
43    // Older emoji + dingbats/symbols Segoe UI Emoji may not cover.
44    &[r"C:\Windows\Fonts\seguisym.ttf"],
45];

83/// Append available system CJK/Unicode fonts as fallbacks. No-op (keeps the egui
84/// defaults) when none of the candidates are present.
85pub fn install_unicode_fonts(ctx: &egui::Context) {
86    let mut fonts = egui::FontDefinitions::default();
87    let mut added: Vec<String> = Vec::new();
88
89    for group in FONT_GROUPS {
90        for path in *group {
91            match std::fs::read(path) {
92                Ok(bytes) => {
93                    let key = format!("sys:{path}");
94                    fonts
95                        .font_data
96                        .insert(key.clone(), Arc::new(FontData::from_owned(bytes)));
97                    added.push(key);
98                    break; // first match in the group wins
99                }
100                Err(_) => continue,
101            }
102        }
103    }
104
105    if added.is_empty() {
106        return;
107    }
108
109    // Fallbacks: keep the default font primary, try these only for missing glyphs.
110    for family in [FontFamily::Proportional, FontFamily::Monospace] {
111        let list = fonts.families.entry(family).or_default();
112        for key in &added {
113            list.push(key.clone());
114        }
115    }
116
117    ctx.set_fonts(fonts);
118    tracing::info!("installed {} fallback font(s) for non-Latin glyphs", added.len());
119}
