1//! Image → schedule OCR.
2//!
3//! Several schedule sources publish the week as an *image* (a Twitch offline
4//! banner, a YouTube community post, a pinned tweet). We read those by shelling
5//! out to an LLM CLI — by default the `claude` CLI — with a strict OCR→JSON
6//! prompt, then map the returned events to [`ScheduleSegment`]s. This ports the
7//! user's working `scripts/decode-schedule-img.ps1` into the app.
8//!
9//! Everything here fails *soft*: a missing CLI, an auth prompt, a non-zero exit,
10//! or unparseable output all yield `None` so a transient OCR failure never wipes a
11//! previously-stored schedule.
12

51/// Result of one `ocr_schedule_image` attempt (primary + optional fallback call).
52pub struct OcrRunResult {
53    /// Decoded segments — `None` if every CLI call failed or parse failed entirely.
54    pub segments: Option<Vec<ScheduleSegment>>,
55    /// Stats from each successful CLI call (0 = CLI failed, 1 = primary ok, 2 = fallback also ran).
56    pub cli_calls: Vec<OcrCallStats>,
57    /// CLI invocations that returned a non-zero exit / timed out / failed to spawn.
58    pub cli_failures: u32,
59    /// CLI calls that returned output but whose JSON couldn't be parsed as schedule events.
60    pub parse_failures: u32,
61}

86/// Resolved options for one OCR run.
87#[derive(Clone, Debug)]
88pub struct OcrOpts {
89    /// The LLM CLI executable (default `claude`).
90    pub command: String,
91    /// Primary model to try first.
92    pub model: String,
93    /// Stronger model to retry with if the primary returns invalid JSON.
94    pub fallback_model: String,
95    /// IANA timezone the banner times are assumed to be in (empty = let the model
96    /// use whatever the banner shows).
97    pub timezone: String,
98    /// UTC offset string matching the timezone/season, e.g. `"+02:00"` (empty =
99    /// derive from the machine's current local offset).
100    pub offset: String,
101    /// Year to assume for dates (banners rarely show it).
102    pub year: i32,
103    /// `--max-budget-usd` value (empty = omit flag, no per-call USD cap).
104    pub max_budget_usd: String,
105    /// Process timeout in seconds (0 = use `DEFAULT_TIMEOUT_SECS`).
106    pub timeout_secs: u64,
107    /// `--effort` level (empty = omit flag; valid: low/medium/high/xhigh/max).
108    pub effort: String,
109}

111/// Build [`OcrOpts`] from global settings, with per-channel overrides taking
112/// precedence (model + timezone + offset).
113pub fn ocr_opts_from_settings(store: &Store, cfg: &ChannelSourceConfig) -> OcrOpts {
114    let get = |key: &str| store.get_setting(key).ok().flatten().unwrap_or_default();
115    let non_empty = |s: String, default: &str| {
116        if s.trim().is_empty() {
117            default.to_string()
118        } else {
119            s
120        }
121    };
122    let pick = |chan: &str, global: String| {
123        if !chan.trim().is_empty() {
124            chan.to_string()
125        } else {
126            global
127        }
128    };
129
130    let timezone = pick(&cfg.ocr_timezone, get(K_OCR_TIMEZONE));
131    let offset = pick(&cfg.ocr_offset, get(K_OCR_OFFSET));
132    // Only fall back to the MACHINE's local offset when no timezone info was given
133    // at all. If a primary timezone is named but the offset is blank, pairing that
134    // zone with the machine offset would be a contradiction (the per-channel
135    // override exists precisely for zones *other* than the machine's) and would
136    // shift every event — so leave the offset empty and let the prompt have the
137    // model derive the correct (DST-aware) offset from the named zone instead.
138    let offset = if offset.trim().is_empty() && timezone.trim().is_empty() {
139        local_offset_string()
140    } else {
141        offset
142    };
143
144    let timeout_secs = get(K_OCR_TIMEOUT_SECS)
145        .trim()
146        .parse::<u64>()
147        .unwrap_or(0);
148
149    OcrOpts {
150        command: non_empty(get(K_OCR_COMMAND), DEFAULT_COMMAND),
151        model: non_empty(pick(&cfg.ocr_model, get(K_OCR_MODEL)), DEFAULT_MODEL),
152        fallback_model: non_empty(get(K_OCR_FALLBACK_MODEL), DEFAULT_FALLBACK_MODEL),
153        timezone,
154        offset,
155        year: chrono::Local::now().year(),
156        max_budget_usd: get(K_OCR_MAX_BUDGET),
157        timeout_secs,
158        effort: get(K_OCR_EFFORT),
159    }
160}

172/// Persist cumulative OCR stats back to the settings store.
173pub fn save_ocr_stats(store: &Store, stats: &OcrStats) {

175        let _ = store.set_setting(K_OCR_STATS, &json);
176    }

179/// Fold one `OcrRunResult` into the persistent cumulative stats.
180pub fn accumulate_ocr_stats(store: &Store, result: &OcrRunResult) {
181    if result.cli_calls.is_empty() && result.cli_failures == 0 && result.parse_failures == 0 {
182        return;
183    }
184    let mut stats = load_ocr_stats(store);
185    stats.cli_failures += result.cli_failures as u64;
186    stats.parse_failures += result.parse_failures as u64;
187    for call in &result.cli_calls {
188        stats.calls += 1;
189        stats.input_tokens += call.input_tokens;
190        stats.output_tokens += call.output_tokens;
191        stats.cache_read_tokens += call.cache_read_tokens;
192        stats.cache_creation_tokens += call.cache_creation_tokens;
193        stats.cost_usd += call.cost_usd;
194        stats.last_call_at = Some(now_unix());
195        let m: &mut OcrModelStats = stats.by_model.entry(call.model.clone()).or_default();
196        m.calls += 1;
197        m.input_tokens += call.input_tokens;
198        m.output_tokens += call.output_tokens;
199        m.cost_usd += call.cost_usd;
200    }
201    save_ocr_stats(store, &stats);
202}

204/// Increment the cache-hit counter in the persistent stats.
205pub fn record_ocr_cache_hit(store: &Store) {
206    let mut stats = load_ocr_stats(store);
207    stats.cache_hits += 1;
208    save_ocr_stats(store, &stats);
209}

211/// `"+02:00"`-style string for the machine's current local UTC offset.
212fn local_offset_string() -> String {
213    use chrono::Offset;
214    let secs = chrono::Local::now().offset().fix().local_minus_utc();
215    let sign = if secs < 0 { '-' } else { '+' };
216    let secs = secs.abs();
217    format!("{sign}{:02}:{:02}", secs / 3600, (secs % 3600) / 60)
218}

236/// OCR a schedule image. Always returns an `OcrRunResult` so callers can accumulate
237/// stats even on failure; `result.segments` is `None` when nothing parseable came back.
238pub async fn ocr_schedule_image(image_path: &Path, opts: &OcrOpts) -> OcrRunResult {
239    if !image_path.exists() {
240        debug!("OCR: image not found: {}", image_path.display());
241        return OcrRunResult { segments: None, cli_calls: vec![], cli_failures: 0, parse_failures: 0 };
242    }
243    let dir = image_path.parent().unwrap_or_else(|| Path::new("."));
244    let img = image_path.to_string_lossy().replace('\\', "/");
245    let dir_s = dir.to_string_lossy().replace('\\', "/");
246    let prompt = build_prompt(&img, opts);
247
248    let mut cli_calls: Vec<OcrCallStats> = Vec::new();
249    let mut cli_failures: u32 = 0;
250    let mut parse_failures: u32 = 0;
251
252    // Try the cheap model first, then the stronger fallback on a parse miss.
253    let events = match run_cli(&opts.command, &opts.model, &dir_s, &prompt, opts).await {
254        Some((raw, call_stats)) => {
255            cli_calls.push(call_stats);
256            match parse_events(&raw) {
257                Some(ev) => Some(ev),
258                None => {
259                    parse_failures += 1;
260                    None
261                }
262            }
263        }
264        None => {
265            cli_failures += 1;
266            None
267        }
268    };
269
270    let events = if events.is_none() && opts.fallback_model != opts.model {
271        debug!("OCR: retrying with fallback model {}", opts.fallback_model);
272        match run_cli(&opts.command, &opts.fallback_model, &dir_s, &prompt, opts).await {
273            Some((raw, call_stats)) => {
274                cli_calls.push(call_stats);
275                match parse_events(&raw) {
276                    Some(ev) => Some(ev),
277                    None => {
278                        parse_failures += 1;
279                        None
280                    }
281                }
282            }
283            None => {
284                cli_failures += 1;
285                None
286            }
287        }
288    } else {
289        events
290    };
291
292    let segments = events.map(|ev| map_events(ev, opts));
293    OcrRunResult { segments, cli_calls, cli_failures, parse_failures }
294}

296/// Substitute the per-image placeholders into the strict OCR→JSON prompt
297/// (ported from `scripts/decode-schedule-img.ps1`).
298///
299/// Timezones are the subtle part: image schedules print times in whatever zone(s)
300/// the streamer chose — and some print the SAME stream in several zones at once
301/// (e.g. PDT / JST / GMT), sometimes with a `*` "next day" mark when a conversion
302/// rolls past midnight. We make the model resolve each stream block to ONE
303/// absolute instant (RFC 3339 with its real offset) so multi-zone rows can't
304/// inflate into duplicate events, anchoring date headers to the channel's primary
305/// timezone when one is configured.
306fn build_prompt(image_path: &str, opts: &OcrOpts) -> String {
307    let primary = opts.timezone.trim();
308    let offset = opts.offset.trim();
309    let tz_rules = if !primary.is_empty() {
310        // Only state a numeric offset if the user actually supplied one. Otherwise
311        // tell the model to work out the correct (DST-aware) offset for the named
312        // zone itself — never pair the zone with an unrelated machine offset.
313        let zone_clause = if offset.is_empty() {
314            format!("{primary} — work out its correct UTC offset for each date yourself, accounting for daylight saving")
315        } else {
316            format!("{primary} (UTC offset {offset})")
317        };
318        let no_label_clause = if offset.is_empty() {
319            format!("assume {primary} (using its correct offset for that date)")
320        } else {
321            format!("assume {primary} with offset {offset}")
322        };
323        format!(
324            "- Timezone: this schedule's day/date headers are written in the primary timezone {zone_clause}.\n\
325- A single stream may print its start time in SEVERAL timezones at once (e.g. 'PDT / JST / GMT'). Those are the SAME moment shown more than once — NOT separate streams. Emit exactly ONE event per stream block.\n\
326- To fix the instant: prefer the time labelled {primary}, combined with that row's date header (read in {primary}). If {primary} is not among the printed labels, use any one printed timezone with its real UTC offset; never multiply events.\n\
327- A '*' or 'next day' mark beside a converted time only means that conversion lands on the following calendar day — it does not change the instant. Resolve everything to one absolute UTC instant and let the offset carry the date.\n\
328- If NO timezone is printed anywhere, {no_label_clause}. Never convert to my machine's local timezone.",
329        )
330    } else {
331        // No primary timezone configured. Provide an explicit abbreviation→offset
332        // table so the model doesn't fall back to the machine's local offset when
333        // a recognisable label (e.g. PDT, EST) IS printed on the banner.
334        format!(
335            "- Timezone: identify the timezone abbreviation(s) printed on the banner and map each to its real UTC offset using this table:\n\
336  PDT=-07:00  PST=-08:00  MDT=-06:00  MST=-07:00  CDT=-05:00  CST=-06:00\n\
337  EDT=-04:00  EST=-05:00  BRT=-03:00  GMT=+00:00  UTC=+00:00  BST=+01:00\n\
338  CET=+01:00  CEST=+02:00  EET=+02:00  EEST=+03:00  MSK=+03:00\n\
339  IST=+05:30  ICT=+07:00  HKT=+08:00  SGT=+08:00  AWST=+08:00\n\
340  JST=+09:00  KST=+09:00  AEST=+10:00  AEDT=+11:00  NZST=+12:00  NZDT=+13:00\n\
341  CRITICAL: when a timezone abbreviation IS visible on the banner, always use that abbreviation's real UTC offset from the table — NEVER use the assumed offset below.\n\
342  Only if absolutely NO timezone abbreviation appears anywhere on the banner should you assume UTC offset {offset}.\n\
343- A single stream may print its start time in SEVERAL timezones at once (e.g. 'PDT / JST / GMT'). Those are the SAME moment shown more than once — NOT separate streams. Emit exactly ONE event per stream block, anchored to the first/topmost printed timezone with its real UTC offset from the table.\n\
344- A '*' or 'next day' mark beside a converted time only means that conversion lands on the following calendar day — it does not change the instant.\n\
345- Never convert to my machine's local timezone.",
346        )
347    };
348    format!(
349        "You are an OCR-to-JSON extractor. Read the streamer schedule in {image_path} and output an array of event objects.\n\
350\n\
351RULES:\n\
352- Output ONLY raw JSON. No markdown, no code fences, no backticks, no commentary, no leading or trailing text. The first character of your reply must be '[' and the last must be ']'.\n\
353{tz_rules}\n\
354- The year is {year}.\n\
355- Transcribe titles literally. 'w' or 'W' before a name means 'with' (a collaborator), e.g. 'FEARS TO FATHOM w CRELLY' -> title 'Fears to Fathom', collab 'Crelly'. Do not guess or 'correct' names.\n\
356- Skip any card marked OFFLINE, marked as a non-stream note (e.g. 'podcasting', 'break', 'TBD'), or with an unknown date ('????').\n\
357- If a time is vague (e.g. 'Evening'), set time and datetime to null but keep the raw text in time_label.\n\
358\n\
359Each object has these fields:\n\
360- title (string)\n\
361- collab (string or null)\n\
362- date (YYYY-MM-DD, in the timezone you used for datetime)\n\
363- day (weekday name)\n\
364- time (HH:MM 24-hour in the timezone you used, or null)\n\
365- time_label (raw time text from banner, e.g. '12.00 P.M.' or 'Evening')\n\
366- timezone (IANA name of the timezone you used)\n\
367- datetime (full ISO 8601 instant WITH its UTC offset, e.g. 2026-06-21T13:00:00-07:00, or null if no exact time)\n\
368- source_image (set this to the filename: {image_path})",
369        year = opts.year,
370    )
371}

373/// Run the LLM CLI for one image+model using `--output-format json`.
374/// Returns `(result_text, stats)` on success, `None` on any failure.
375/// If the output isn't a recognisable JSON envelope (older CLI), falls back to
376/// treating stdout as raw text with zeroed stats so the parse step still runs.
377async fn run_cli(command: &str, model: &str, dir: &str, prompt: &str, opts: &OcrOpts) -> Option<(String, OcrCallStats)> {
378    let timeout = Duration::from_secs(if opts.timeout_secs > 0 { opts.timeout_secs } else { DEFAULT_TIMEOUT_SECS });
379    let mut cmd = tokio::process::Command::new(command);
380    cmd.args(["--model", model, "--add-dir", dir, "-p", prompt, "--output-format", "json"])
381        .stdin(Stdio::null())
382        .stdout(Stdio::piped())
383        .stderr(Stdio::null());
384    let budget = opts.max_budget_usd.trim();
385    if !budget.is_empty() {
386        cmd.args(["--max-budget-usd", budget]);
387    }
388    let effort = opts.effort.trim();
389    if !effort.is_empty() {
390        cmd.args(["--effort", effort]);
391    }
392    #[cfg(windows)]
393    cmd.creation_flags(CREATE_NO_WINDOW);
394
395    let mut log_flags = format!("--model {model}");
396    if !budget.is_empty() { log_flags.push_str(&format!(" --max-budget-usd {budget}")); }
397    if !effort.is_empty() { log_flags.push_str(&format!(" --effort {effort}")); }
398    info!("OCR: invoking '{command}' {log_flags} (timeout {timeout:?}, dir {dir})");
399    let child = match cmd.spawn() {
400        Ok(c) => c,
401        Err(e) => {
402            warn!("OCR: failed to spawn '{command}': {e} (is the CLI installed/on PATH?)");
403            return None;
404        }
405    };
406    let out = match tokio::time::timeout(timeout, child.wait_with_output()).await {
407        Ok(Ok(o)) => o,
408        Ok(Err(e)) => {
409            warn!("OCR: CLI wait failed: {e}");
410            return None;
411        }
412        Err(_) => {
413            warn!("OCR: CLI timed out after {}s", timeout.as_secs());
414            return None;
415        }
416    };
417    if !out.status.success() {
418        warn!("OCR: CLI exited with {:?}", out.status.code());
419        return None;
420    }
421    let stdout = String::from_utf8_lossy(&out.stdout).trim().to_string();
422    if stdout.is_empty() {
423        return None;
424    }
425    // Try to parse the JSON envelope (`--output-format json`).
426    // On parse failure (older CLI without that flag), treat stdout as raw OCR text
427    // and report zero tokens/cost.
428    match serde_json::from_str::<CliEnvelope>(&stdout) {
429        Ok(env) if !env.result.is_empty() => {
430            let stats = OcrCallStats {
431                model: model.to_string(),
432                input_tokens: env.usage.input_tokens,
433                output_tokens: env.usage.output_tokens,
434                cache_read_tokens: env.usage.cache_read_input_tokens,
435                cache_creation_tokens: env.usage.cache_creation_input_tokens,
436                cost_usd: env.total_cost_usd,
437            };
438            info!(
439                "OCR: {model} → {} in / {} out / ${:.4} (cache r={} c={})",
440                env.usage.input_tokens,
441                env.usage.output_tokens,
442                env.total_cost_usd,
443                env.usage.cache_read_input_tokens,
444                env.usage.cache_creation_input_tokens,
445            );
446            Some((env.result, stats))
447        }
448        _ => {
449            // Envelope parse failed — either the CLI doesn't support --output-format json
450            // or the response was an error object. Fall back to raw text with no stats.
451            debug!("OCR: output wasn't a CLI JSON envelope; treating as raw text");
452            Some((stdout, OcrCallStats { model: model.to_string(), ..Default::default() }))
453        }
454    }
455}

457/// Clean and parse the CLI's reply into events: strip any stray markdown fences,
458/// trim to the outermost JSON array, then deserialize. `None` on invalid JSON.
459fn parse_events(raw: &str) -> Option<Vec<OcrEvent>> {
460    let clean = raw.replace("```json", "").replace("```", "");
461    // Trim to the outermost JSON array, in case of leading/trailing prose.
462    let start = clean.find('[')?;
463    let end = clean.rfind(']')?;
464    if end <= start {
465        return None;
466    }
467    let slice = &clean[start..=end];
468    match serde_json::from_str::<Vec<OcrEvent>>(slice) {
469        Ok(v) => Some(v),
470        Err(e) => {
471            debug!("OCR: invalid JSON ({e})");
472            None
473        }
474    }
475}

477/// Map decoded events to schedule segments, dropping any without a resolvable
478/// exact start time (vague "Evening" cards carry no datetime).
479fn map_events(events: Vec<OcrEvent>, opts: &OcrOpts) -> Vec<ScheduleSegment> {
480    let mut out: Vec<ScheduleSegment> = events
481        .into_iter()
482        .filter_map(|e| {
483            let start = event_start(&e, opts)?;
484            let title = match e.collab.as_deref().map(str::trim).filter(|s| !s.is_empty()) {
485                Some(c) if !e.title.trim().is_empty() => format!("{} w/ {}", e.title.trim(), c),
486                _ => e.title.trim().to_string(),
487            };
488            if title.is_empty() {
489                return None;
490            }
491            Some(ScheduleSegment {
492                id: 0,
493                monitor_id: 0,
494                start_time: start,
495                end_time: None,
496                title,
497                category: String::new(),
498                canceled: false,
499                video_id: None,
500            })
501        })
502        .collect();
503    out.sort_by(|a, b| a.start_time.cmp(&b.start_time).then_with(|| a.title.cmp(&b.title)));
504    out.dedup_by(|a, b| a.start_time == b.start_time && a.title == b.title);
505    // A channel streams one thing at a time, so two events at the SAME instant are
506    // the same stream printed twice — typically multi-timezone rows the model
507    // failed to merge into one, sometimes with a divergent title (collab tagged on
508    // only one row, casing differences). The exact-title dedup above misses those,
509    // so collapse any remaining same-instant rows, keeping the most informative
510    // (longest) title.
511    let mut i = 0;
512    while i + 1 < out.len() {
513        if out[i].start_time == out[i + 1].start_time {
514            if out[i + 1].title.chars().count() > out[i].title.chars().count() {
515                out[i].title = std::mem::take(&mut out[i + 1].title);
516            }
517            out.remove(i + 1);
518        } else {
519            i += 1;
520        }
521    }
522    out
523}

525/// Resolve an event's start time (unix seconds): prefer the ISO `datetime`, else
526/// combine `date` + `time` + the assumed offset. `None` when no exact time.
527fn event_start(e: &OcrEvent, opts: &OcrOpts) -> Option<i64> {
528    if let Some(dt) = e.datetime.as_deref().map(str::trim).filter(|s| !s.is_empty()) {
529        if let Ok(parsed) = chrono::DateTime::parse_from_rfc3339(dt) {
530            return Some(parsed.timestamp());
531        }
532    }
533    // Fallback: date (YYYY-MM-DD) + time (HH:MM) + assumed offset → RFC3339.
534    let date = e.date.as_deref().map(str::trim).filter(|s| !s.is_empty())?;
535    let time = e.time.as_deref().map(str::trim).filter(|s| !s.is_empty())?;
536    let offset = if opts.offset.trim().is_empty() {
537        "+00:00"
538    } else {
539        opts.offset.trim()
540    };
541    let composed = format!("{date}T{time}:00{offset}");
542    chrono::DateTime::parse_from_rfc3339(&composed)
543        .ok()
544        .map(|d| d.timestamp())
545}

590    #[test]
591    fn multi_timezone_one_stream_collapses_to_single_event() {
592        // girl_dm_-style row: the SAME 1:00 PM PDT stream printed in three zones —
593        // PDT, GMT, and JST (next calendar day, the '*' marker). They denote one
594        // moment, so map_events must collapse them to a single event by instant —
595        // never one event per timezone label.
596        let raw = r#"[
597          {"title":"Art stream","collab":null,"datetime":"2026-06-21T13:00:00-07:00"},
598          {"title":"Art stream","collab":null,"datetime":"2026-06-21T20:00:00+00:00"},

600        ]"#;

605        // 2026-06-21T13:00-07:00 == 2026-06-21T20:00Z == unix 1782072000.
606        assert_eq!(segs[0].start_time, 1782072000);
