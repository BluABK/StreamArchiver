1//! UI-freeze watchdog + panic dialog (Windows-focused, compile-safe elsewhere).
2//!
3//! Two safety nets that turn a *silent* GUI hang or a process-killing panic into
4//! a *visible* native dialog:
5//!
6//! 1. [`install_panic_dialog`] — a `std::panic::set_hook` that pops a native
7//!    `TaskDialog` with the panic message + location + scrollable stack trace /
8//!    environment info before the default hook runs.
9//!
10//! 2. [`Heartbeat`] + [`start_watchdog`] — the UI thread stamps a monotonic
11//!    timestamp every frame ([`Heartbeat::beat`]) and an optional coarse activity
12//!    label ([`Heartbeat::set_activity`]) before risky sections. A dedicated
13//!    background thread wakes every second; if the last beat is older than the
14//!    threshold while the app is meant to be rendering, it shows a native dialog
15//!    **off the UI thread**. Debounced so it fires once per hang.
16
17use std::sync::{Arc, Mutex};
18use std::sync::atomic::{AtomicBool, AtomicU8, AtomicU64, Ordering};
19use std::time::{Duration, Instant};
20
21/// Coarse "what was the UI thread doing" phases.
22#[derive(Clone, Copy, Debug, PartialEq, Eq)]
23#[repr(u8)]
24pub enum Activity {
25    Idle = 0,
26    Frame = 1,
27    Properties = 2,
28    EmoteViewerGrid = 3,
29    EmoteDecodePump = 4,
30    Chat = 5,
31    PropertiesLoad = 6,
32}
33
34impl Activity {
35    fn from_u8(v: u8) -> Activity {
36        match v {
37            1 => Activity::Frame,
38            2 => Activity::Properties,
39            3 => Activity::EmoteViewerGrid,
40            4 => Activity::EmoteDecodePump,
41            5 => Activity::Chat,
42            6 => Activity::PropertiesLoad,
43            _ => Activity::Idle,
44        }
45    }
46    fn label(self) -> &'static str {
47        match self {
48            Activity::Idle => "idle",
49            Activity::Frame => "rendering a frame",
50            Activity::Properties => "drawing the channel properties window",
51            Activity::EmoteViewerGrid => "drawing the emote viewer",
52            Activity::EmoteDecodePump => "decoding / uploading emote images",
53            Activity::Chat => "drawing the chat replay popup",
54            Activity::PropertiesLoad => "loading the channel properties assets",
55        }
56    }
57}

59/// Shared, cheap, lock-free heartbeat. Clone freely (it's an `Arc` inside).
60#[derive(Clone)]
61pub struct Heartbeat {
62    inner: Arc<Inner>,
63}
64
65struct Inner {
66    last_beat_ms: AtomicU64,
67    activity: AtomicU8,
68    active: AtomicBool,
69    /// Free-form context label set by the UI thread alongside the activity
70    /// (e.g. "Channel: MoriCalliope"). Shown in freeze reports.
71    context: Mutex<String>,
72    /// Thread ID of the UI thread captured at heartbeat creation. Used by the
73    /// watchdog to suspend and read the UI thread's stack on freeze.
74    ui_tid: u32,
75    start: Instant,
76}
77
78impl Heartbeat {
79    pub fn new() -> Heartbeat {
80        let start = Instant::now();
81        let ui_tid = current_thread_id();
82        let hb = Heartbeat {
83            inner: Arc::new(Inner {
84                last_beat_ms: AtomicU64::new(0),
85                activity: AtomicU8::new(Activity::Idle as u8),
86                active: AtomicBool::new(true),
87                context: Mutex::new(String::new()),
88                ui_tid,
89                start,
90            }),
91        };
92        hb.beat();
93        hb
94    }
95
96    #[inline]
97    pub fn beat(&self) {
98        let ms = self.inner.start.elapsed().as_millis() as u64;
99        self.inner.last_beat_ms.store(ms, Ordering::Relaxed);
100    }
101
102    #[inline]
103    pub fn set_activity(&self, a: Activity) {
104        self.inner.activity.store(a as u8, Ordering::Relaxed);
105    }
106
107    /// Set a human-readable context label alongside the activity (e.g. which
108    /// channel window is open). Shown in freeze reports to identify the instance.
109    pub fn set_context(&self, ctx: impl Into<String>) {
110        if let Ok(mut g) = self.inner.context.lock() {
111            *g = ctx.into();
112        }
113    }
114
115    /// Clear the context label (call when leaving a context-tagged section).
116    #[allow(dead_code)]
117    pub fn clear_context(&self) {
118        if let Ok(mut g) = self.inner.context.lock() {
119            g.clear();
120        }
121    }
122
123    #[inline]
124    pub fn set_active(&self, active: bool) {
125        self.inner.active.store(active, Ordering::Relaxed);
126    }
127
128    fn age(&self) -> Duration {
129        let now = self.inner.start.elapsed().as_millis() as u64;
130        let last = self.inner.last_beat_ms.load(Ordering::Relaxed);
131        Duration::from_millis(now.saturating_sub(last))
132    }
133
134    fn is_active(&self) -> bool {
135        self.inner.active.load(Ordering::Relaxed)
136    }
137
138    fn activity(&self) -> Activity {
139        Activity::from_u8(self.inner.activity.load(Ordering::Relaxed))
140    }
141
142    fn context(&self) -> String {
143        self.inner.context.lock().map(|g| g.clone()).unwrap_or_default()
144    }
145
146    fn ui_tid(&self) -> u32 {
147        self.inner.ui_tid
148    }
149}
150
151impl Default for Heartbeat {
152    fn default() -> Self {
153        Self::new()
154    }
155}

160    #[cfg(not(target_os = "windows"))]
161    { 0u32 }

164/// Spawn the watchdog thread (call ONCE at startup).
165pub fn start_watchdog(hb: Heartbeat, threshold: Duration, exit_after_dialog: bool) {
166    const REPROMPT_COOLDOWN: Duration = Duration::from_secs(20);
167    std::thread::Builder::new()
168        .name("ui-watchdog".into())
169        .spawn(move || {
170            let mut last_dialog: Option<Instant> = None;
171            loop {
172                std::thread::sleep(Duration::from_secs(1));
173                let age = hb.age();
174                let active = hb.is_active();
175
176                if active && age >= threshold {
177                    let cooled = last_dialog.is_none_or(|t| t.elapsed() >= REPROMPT_COOLDOWN);
178                    if cooled {
179                        let secs = age.as_secs();
180                        let doing = hb.activity().label();
181                        let ctx = hb.context();
182                        let summary = format!(
183                            "StreamArchiver's UI thread has stopped responding.\n\n\
184                             Last UI heartbeat:   {secs}s ago\n\
185                             Last known activity: {doing}{}\n\n\
186                             Choose \"Keep waiting\" to give it more time, or \
187                             \"Force quit\" to close the app now.\n\n\
188                             Background recordings are NOT affected — they run \
189                             in separate processes.",
190                            if ctx.is_empty() {
191                                String::new()
192                            } else {
193                                format!("\n  Context:             {ctx}")
194                            },
195                        );
196                        let detail = build_hang_detail(&hb, secs, doing);
197                        let log_dir = crate::app_paths::logs_dir()
198                            .to_string_lossy()
199                            .into_owned();
200                        let force_quit = show_detail_dialog(
201                            "StreamArchiver — UI frozen",
202                            "StreamArchiver — UI frozen",
203                            &summary,
204                            &detail,
205                            &log_dir,
206                            true,
207                            Some((&hb, threshold)),
208                        );
209                        if force_quit || exit_after_dialog {
210                            std::process::exit(101);
211                        }
212                        last_dialog = Some(Instant::now());
213                    }
214                } else if age < threshold {
215                    last_dialog = None;
216                }
217            }
218        })
219        .expect("spawn ui-watchdog thread");
220}

222/// Install a panic hook that shows a native TaskDialog with scrollable details
223/// (panic message, location, thread, stack trace, environment), then chains to
224/// the previous hook so the backtrace still hits stderr / the log.
225pub fn install_panic_dialog() {
226    let previous = std::panic::take_hook();
227    std::panic::set_hook(Box::new(move |info| {
228        let payload = info.payload();
229        let msg: &str = if let Some(s) = payload.downcast_ref::<&str>() {
230            s
231        } else if let Some(s) = payload.downcast_ref::<String>() {
232            s.as_str()
233        } else {
234            "Box<dyn Any>"
235        };
236        let location = info
237            .location()
238            .map(|l| format!("{}:{}:{}", l.file(), l.line(), l.column()))
239            .unwrap_or_else(|| "unknown location".to_string());
240        let thread = std::thread::current();
241        let tname = thread.name().unwrap_or("<unnamed>");
242
243        let summary = format!(
244            "StreamArchiver crashed (panic).\n\n\
245             Thread:   {tname}\n\
246             Location: {location}\n\n\
247             {msg}\n\n\
248             The application will now close.",
249        );
250        let detail = build_panic_detail(msg, tname, &location);
251        let log_dir = crate::app_paths::logs_dir()
252            .to_string_lossy()
253            .into_owned();
254
255        show_detail_dialog(
256            "StreamArchiver crashed",
257            "StreamArchiver crashed (panic)",
258            &summary,
259            &detail,
260            &log_dir,
261            false,
262            None,
263        );
264
265        previous(info);
266    }));
267}

271fn build_panic_detail(msg: &str, thread: &str, location: &str) -> String {
272    let backtrace = std::backtrace::Backtrace::force_capture();
273    let backtrace_str = backtrace.to_string();
274    let backtrace_note = if backtrace_str.contains("<unknown>") || backtrace_str.trim().is_empty() {
275        " (symbols stripped — release build shows raw addresses only)"
276    } else {
277        ""
278    };
279
280    let pid = std::process::id();
281    let version = env!("CARGO_PKG_VERSION");
282    let os = os_info_string();
283    let log_dir = crate::app_paths::logs_dir().to_string_lossy().into_owned();
284    let (thread_count, named_threads) = enumerate_threads_text(pid);
285
286    format!(
287        "=== Panic ===\n\
288         Message:  {msg}\n\
289         Thread:   {thread}\n\
290         Location: {location}\n\
291         PID:      {pid}\n\n\
292         === Environment ===\n\
293         App version: {version}\n\
294         OS:          {os}\n\
295         Log dir:     {log_dir}\n\n\
296         === Live threads ({thread_count} total; named) ===\n\
297         {named_threads}\n\n\
298         === Stack trace{backtrace_note} ===\n\
299         {backtrace_str}"
300    )
301}

303fn build_hang_detail(hb: &Heartbeat, frozen_secs: u64, activity: &str) -> String {
304    let pid = std::process::id();
305    let ui_tid = hb.ui_tid();
306    let ctx = hb.context();
307    let version = env!("CARGO_PKG_VERSION");
308    let os = os_info_string();
309    let log_dir = crate::app_paths::logs_dir().to_string_lossy().into_owned();
310    let (thread_count, named_threads) = enumerate_threads_text(pid);
311    let stack = capture_ui_thread_stack(ui_tid);
312
313    let context_line = if ctx.is_empty() {
314        String::new()
315    } else {
316        format!("Context:       {ctx}\n")
317    };
318
319    format!(
320        "=== Freeze Diagnostics ===\n\
321         Frozen for:    {frozen_secs}s\n\
322         Last activity: {activity}\n\
323         {context_line}\
324         PID:           {pid}\n\
325         UI thread TID: {ui_tid}\n\n\
326         === Environment ===\n\
327         App version: {version}\n\
328         OS:          {os}\n\
329         Log dir:     {log_dir}\n\n\
330         === Live threads ({thread_count} total; named) ===\n\
331         {named_threads}\n\n\
332         === UI thread stack (TID {ui_tid}) ===\n\
333         {stack}"
334    )
335}

337fn os_info_string() -> String {
338    let out = std::process::Command::new("powershell")
339        .args([
340            "-NoProfile",
341            "-NonInteractive",
342            "-Command",
343            "(Get-CimInstance Win32_OperatingSystem).Caption + ' ' + \
344             (Get-CimInstance Win32_OperatingSystem).OSArchitecture + \
345             ' build ' + (Get-CimInstance Win32_OperatingSystem).BuildNumber",
346        ])
347        .output();
348    if let Ok(o) = out {
349        let s = String::from_utf8_lossy(&o.stdout).trim().to_owned();
350        if !s.is_empty() {
351            return s;
352        }
353    }
354    format!("Windows ({})", std::env::consts::ARCH)
355}

357/// Enumerate threads in this process. Returns (total_count, named_threads_text).
358/// Uses GetThreadDescription (Win 10 1607+) to show only threads with names,
359/// which is far more useful than a wall of TID+priority numbers.
360fn enumerate_threads_text(pid: u32) -> (usize, String) {
361    #[cfg(target_os = "windows")]
362    {
363        use windows::Win32::Foundation::CloseHandle;
364        use windows::Win32::System::Diagnostics::ToolHelp::{
365            CreateToolhelp32Snapshot, Thread32First, Thread32Next,
366            THREADENTRY32, TH32CS_SNAPTHREAD,
367        };
368        use windows::Win32::System::Threading::{
369            GetThreadDescription, OpenThread, THREAD_QUERY_INFORMATION,
370        };
371
372        unsafe {
373            let Ok(snap) = CreateToolhelp32Snapshot(TH32CS_SNAPTHREAD, 0) else {
374                return (0, "(could not snapshot threads)".to_owned());
375            };
376            let mut te = THREADENTRY32 {
377                dwSize: std::mem::size_of::<THREADENTRY32>() as u32,
378                ..Default::default()
379            };
380            let mut total = 0usize;
381            let mut named: Vec<String> = Vec::new();
382
383            if Thread32First(snap, &mut te).is_ok() {
384                loop {
385                    if te.th32OwnerProcessID == pid {
386                        total += 1;
387                        // Try to get the thread's name via Win10 1607+ API.
388                        if let Ok(thandle) = OpenThread(
389                            THREAD_QUERY_INFORMATION,
390                            false,
391                            te.th32ThreadID,
392                        ) {
393                            // GetThreadDescription returns Result<PWSTR> in windows 0.62.
394                            if let Ok(desc) = GetThreadDescription(thandle) {
395                                if !desc.is_null() {
396                                    let ptr = desc.0;
397                                    let len = (0..)
398                                        .take_while(|&i| *ptr.add(i) != 0)
399                                        .count();
400                                    if len > 0 {
401                                        let slice = std::slice::from_raw_parts(ptr, len);
402                                        let name = String::from_utf16_lossy(slice);
403                                        named.push(format!(
404                                            "  TID {:6}  {}",
405                                            te.th32ThreadID, name
406                                        ));
407                                    }
408                                    // Intentionally not freeing desc — acceptable in
409                                    // a crash/freeze handler that runs at most once.
410                                }
411                            }
412                            let _ = CloseHandle(thandle);
413                        }
414                    }
415                    te.dwSize = std::mem::size_of::<THREADENTRY32>() as u32;
416                    if Thread32Next(snap, &mut te).is_err() {
417                        break;
418                    }
419                }
420            }
421            let _ = CloseHandle(snap);
422
423            let text = if named.is_empty() {
424                "(no named threads — set thread names via Builder::name())".to_owned()
425            } else {
426                named.join("\n")
427            };
428            (total, text)
429        }
430    }
431    #[cfg(not(target_os = "windows"))]
432    {
433        let _ = pid;
434        (0, "(not available on this platform)".to_owned())
435    }
436}

438/// Capture the UI thread's call stack by suspending it, reading its CONTEXT,
439/// then walking frames with StackWalk64 and resolving symbols with SymFromAddr.
440/// Returns a formatted multi-line string. Fails gracefully on any error.
441fn capture_ui_thread_stack(tid: u32) -> String {
442    #[cfg(target_os = "windows")]
443    {
444        use std::sync::atomic::AtomicBool;
445        use windows::Win32::Foundation::CloseHandle;
446        use windows::Win32::System::Diagnostics::Debug::{
447            GetThreadContext, StackWalk64, SymFromAddr, SymFunctionTableAccess64,
448            SymGetModuleBase64, SymInitialize, SymSetOptions, ADDRESS_MODE,
449            CONTEXT, CONTEXT_FLAGS, STACKFRAME64, SYMBOL_INFO,
450            SYMOPT_DEFERRED_LOADS, SYMOPT_UNDNAME,
451        };
452        use windows::Win32::System::Threading::{
453            GetCurrentProcess, OpenThread, ResumeThread, SuspendThread,
454            THREAD_GET_CONTEXT, THREAD_QUERY_INFORMATION, THREAD_SUSPEND_RESUME,
455        };
456
457        // AMD64 CONTEXT flags (raw values; avoids chasing crate constant names).
458        const CTX_AMD64: u32 = 0x0010_0000;
459        const CTX_FLAGS: u32 = CTX_AMD64 | 0x3; // CONTEXT_CONTROL | CONTEXT_INTEGER
460        const MACHINE_AMD64: u32 = 0x8664;
461
462        static SYM_INIT: AtomicBool = AtomicBool::new(false);
463
464        unsafe {
465            let Ok(thread) = OpenThread(
466                THREAD_GET_CONTEXT | THREAD_SUSPEND_RESUME | THREAD_QUERY_INFORMATION,
467                false,
468                tid,
469            ) else {
470                return format!("(could not open UI thread {tid})");
471            };
472
473            SuspendThread(thread);
474
475            let mut ctx = std::mem::MaybeUninit::<CONTEXT>::zeroed().assume_init();
476            ctx.ContextFlags = CONTEXT_FLAGS(CTX_FLAGS);
477            let got_ctx = GetThreadContext(thread, &mut ctx).is_ok();
478
479            if !got_ctx {
480                ResumeThread(thread);
481                let _ = CloseHandle(thread);
482                return "(GetThreadContext failed)".to_string();
483            }
484
485            let proc = GetCurrentProcess();
486
487            // SymInitialize is idempotent but call it once to avoid races.
488            if !SYM_INIT.swap(true, Ordering::SeqCst) {
489                SymSetOptions(SYMOPT_UNDNAME | SYMOPT_DEFERRED_LOADS);
490                let _ = SymInitialize(proc, None, true);
491            }
492
493            let mut frame = std::mem::MaybeUninit::<STACKFRAME64>::zeroed().assume_init();
494            frame.AddrPC.Offset = ctx.Rip;
495            frame.AddrPC.Mode = ADDRESS_MODE(3); // AddrModeFlat
496            frame.AddrFrame.Offset = ctx.Rbp;
497            frame.AddrFrame.Mode = ADDRESS_MODE(3);
498            frame.AddrStack.Offset = ctx.Rsp;
499            frame.AddrStack.Mode = ADDRESS_MODE(3);
500
501            let mut ctx_walk = ctx;
502            let mut lines: Vec<String> = Vec::new();
503
504            for i in 0..48 {
505                // The windows crate declares these helpers as `unsafe fn` (cdecl)
506            // but StackWalk64 wants `extern "system"` fn pointers. Transmute the
507            // calling convention — on x64 Windows they are identical in practice.
508            use windows::Win32::Foundation::HANDLE;
509            type FtaFn = unsafe extern "system" fn(HANDLE, u64) -> *mut std::ffi::c_void;
510            type MbFn  = unsafe extern "system" fn(HANDLE, u64) -> u64;
511            let fta: FtaFn = std::mem::transmute(SymFunctionTableAccess64 as *const ());
512            let mb:  MbFn  = std::mem::transmute(SymGetModuleBase64 as *const ());
513
514            let ok = StackWalk64(
515                    MACHINE_AMD64,
516                    proc,
517                    thread,
518                    &mut frame,
519                    &mut ctx_walk as *mut CONTEXT as *mut _,
520                    None,
521                    Some(fta),
522                    Some(mb),
523                    None,
524                );
525                if !ok.as_bool() || frame.AddrPC.Offset == 0 {
526                    break;
527                }
528
529                let addr = frame.AddrPC.Offset;
530
531                // Resolve symbol. SYMBOL_INFO has a variable-length Name tail.
532                const MAX_NAME: usize = 512;
533                let sym_total = std::mem::size_of::<SYMBOL_INFO>() + MAX_NAME;
534                let mut sym_buf: Vec<u8> = vec![0u8; sym_total];
535                let sym = sym_buf.as_mut_ptr() as *mut SYMBOL_INFO;
536                (*sym).SizeOfStruct = std::mem::size_of::<SYMBOL_INFO>() as u32;
537                (*sym).MaxNameLen = MAX_NAME as u32;
538
539                let mut disp = 0u64;
540                let label = if SymFromAddr(proc, addr, Some(&mut disp), sym).is_ok() {
541                    let name_ptr = std::ptr::addr_of!((*sym).Name) as *const u8;
542                    let name_len = ((*sym).NameLen as usize).min(MAX_NAME);
543                    let bytes = std::slice::from_raw_parts(name_ptr, name_len);
544                    let name = String::from_utf8_lossy(bytes);
545                    format!("{name}+{disp}")
546                } else {
547                    format!("0x{addr:016x}")
548                };
549                lines.push(format!("  {i:2}: {label}"));
550            }
551
552            // Resume AFTER the full stack walk — thread must stay suspended
553            // while StackWalk64 reads its stack memory.
554            ResumeThread(thread);
555            let _ = CloseHandle(thread);
556
557            if lines.is_empty() {
558                "(no frames captured — debug symbols may be stripped)".to_string()
559            } else {
560                lines.join("\n")
561            }
562        }
563    }
564    #[cfg(not(target_os = "windows"))]
565    {
566        let _ = tid;
567        "(stack capture not available on this platform)".to_string()
568    }
569}

573/// Show a rich crash/hang dialog with a scrollable detail pane.
574///
575/// `warning` selects the warning vs error icon and adjusts the button set:
576/// - panic (false): "Close" + "Copy details" + "Open log folder"
577/// - hang  (true):  "Force quit" + "Keep waiting" + "Copy details" + "Open log folder"
578///
579/// `hb_for_timer`: when `Some`, the dialog polls the heartbeat every ~200ms
580/// and auto-dismisses with "Keep waiting" if the UI recovers. Pass `None` for
581/// crash/panic dialogs where no heartbeat exists.
582///
583/// Returns `true` only when the user chose **Force quit** (hang path).
584fn show_detail_dialog(
585    title: &str,
586    heading: &str,
587    summary: &str,
588    detail: &str,
589    log_dir: &str,
590    warning: bool,
591    hb_for_timer: Option<(&Heartbeat, Duration)>,
592) -> bool {
593    #[cfg(target_os = "windows")]
594    return show_task_dialog_win32(title, heading, summary, detail, log_dir, warning, hb_for_timer);
595
596    #[cfg(not(target_os = "windows"))]
597    {
598        let _ = (heading, detail, log_dir, warning, hb_for_timer);
599        let _ = rfd::MessageDialog::new()
600            .set_level(rfd::MessageLevel::Error)
601            .set_title(title)
602            .set_buttons(rfd::MessageButtons::Ok)
603            .set_description(summary)
604            .show();
605        false
606    }
607}

632/// `LoadImageW` so Windows handles pixel layout correctly.
633/// Call once at startup after the settings store is opened.
634/// A missing file or an unloadable PNG silently falls back to the standard icon.
635pub fn set_dialog_icon(path: Option<std::path::PathBuf>) {
636    #[cfg(target_os = "windows")]
637    if let Some(p) = path {
638        if let Some(hicon) = prepare_and_load_icon(&p) {
639            let _ = DIALOG_ICON_HANDLE.set(hicon.0 as usize);
640        }
641    }
642    #[cfg(not(target_os = "windows"))]
643    let _ = path;
644}

646/// Downscale `src` to 48×48 with Lanczos3, write an ICO to appdata, and load
647/// it via `LoadImageW`. Returns `None` on any failure (bad PNG, save error, …).
648/// Using LoadImageW with an ICO file avoids the manual BGRA flip and premultiply
649/// dance required by CreateBitmap/CreateIconIndirect.
650#[cfg(target_os = "windows")]
651fn prepare_and_load_icon(
652    src: &std::path::Path,
653) -> Option<windows::Win32::UI::WindowsAndMessaging::HICON> {
654    use image::imageops::FilterType;
655    use windows::Win32::UI::WindowsAndMessaging::{
656        LoadImageW, HICON, IMAGE_ICON, LR_LOADFROMFILE,
657    };
658    use windows::core::PCWSTR;
659
660    const ICON_SIZE: u32 = 48;
661
662    // High-quality downscale; avoids the aliasing Windows produces when given a
663    // large PNG and asked to scale it to a 48×48 icon on its own.
664    let img = image::open(src).ok()?.into_rgba8();
665    let scaled = image::imageops::resize(&img, ICON_SIZE, ICON_SIZE, FilterType::Lanczos3);
666
667    // Persist to appdata as an ICO so the result can be inspected and reused.
668    let ico_path = crate::app_paths::data_dir().join("dialog_icon.ico");
669    image::DynamicImage::ImageRgba8(scaled).save(&ico_path).ok()?;
670
671    // LoadImageW handles pixel layout (orientation, alpha) correctly for ICO files.
672    let path_w: Vec<u16> = ico_path
673        .to_string_lossy()
674        .encode_utf16()
675        .chain(std::iter::once(0))
676        .collect();
677    unsafe {
678        let handle = LoadImageW(
679            None,
680            PCWSTR(path_w.as_ptr()),
681            IMAGE_ICON,
682            ICON_SIZE as i32,
683            ICON_SIZE as i32,
684            LR_LOADFROMFILE,
685        )
686        .ok()?;
687        Some(HICON(handle.0))
688    }
689}

691// TDM_CLICK_BUTTON = WM_USER + 102 = 0x0466: simulates a button click on the
692// task dialog. Used to auto-dismiss the hang dialog when the UI recovers.
693const TDM_CLICK_BUTTON: u32 = 0x0466;
694
695/// TaskDialog callback. Intercepts button clicks that should NOT close the
696/// dialog (Copy details, Open log folder) and returns S_FALSE to keep it open.
697/// For hang dialogs, also polls the heartbeat on every timer tick (≈200ms) and
698/// auto-dismisses with "Keep waiting" if the UI thread becomes responsive again.
699#[cfg(target_os = "windows")]
700unsafe extern "system" fn task_dialog_cb(
701    hwnd: windows::Win32::Foundation::HWND,
702    msg: windows::Win32::UI::Controls::TASKDIALOG_NOTIFICATIONS,
703    wparam: windows::Win32::Foundation::WPARAM,
704    _lparam: windows::Win32::Foundation::LPARAM,
705    _lp_ref: isize,
706) -> windows::core::HRESULT {
707    use windows::Win32::Foundation::{LPARAM, WPARAM};
708    use windows::Win32::UI::Controls::{TDN_BUTTON_CLICKED, TASKDIALOG_NOTIFICATIONS};
709    use windows::Win32::UI::WindowsAndMessaging::PostMessageW;
710
711    // TDN_TIMER fires every ~200ms when TDF_CALLBACK_TIMER is set.
712    if msg == TASKDIALOG_NOTIFICATIONS(4) {
713        let hb_ptr = CB_HB_PTR.with(|c| c.get());
714        if hb_ptr != 0 {
715            let threshold_ms = CB_THRESHOLD_MS.with(|c| c.get());
716            // SAFETY: CB_HB_PTR holds a raw pointer to Inner from an Arc that lives
717            // for the program's lifetime (owned by start_watchdog's closure).
718            let inner = unsafe { &*(hb_ptr as *const Inner) };
719            let now = inner.start.elapsed().as_millis() as u64;
720            let last = inner.last_beat_ms.load(Ordering::Relaxed);
721            if now.saturating_sub(last) < threshold_ms {
722                // UI recovered — queue a "Keep waiting" click to dismiss.
723                // SAFETY: hwnd is valid for the lifetime of the dialog callback.
724                unsafe { let _ = PostMessageW(Some(hwnd), TDM_CLICK_BUTTON, WPARAM(2), LPARAM(0)); }
725            }
726        }
727        return windows::core::HRESULT(0);
728    }
729
730    if msg == TDN_BUTTON_CLICKED {
731        let btn = wparam.0 as i32;
732        if btn == BTN_COPY {
733            CB_DETAIL.with(|d| copy_to_clipboard(&d.borrow()));
734            return windows::core::HRESULT(1); // S_FALSE — keep dialog open
735        }
736        if btn == BTN_OPEN_LOGS {
737            CB_LOG_DIR.with(|d| {
738                let _ = std::process::Command::new("explorer")
739                    .arg(d.borrow().as_str())
740                    .spawn();
741            });
742            return windows::core::HRESULT(1); // S_FALSE — keep dialog open
743        }
744    }
745    windows::core::HRESULT(0) // S_OK — allow dialog to close
746}

748#[cfg(target_os = "windows")]
749fn show_task_dialog_win32(
750    title: &str,
751    heading: &str,
752    summary: &str,
753    detail: &str,
754    log_dir: &str,
755    warning: bool,
756    hb_for_timer: Option<(&Heartbeat, Duration)>,
757) -> bool {
758    use windows::Win32::Foundation::{HWND, HINSTANCE};
759    use windows::Win32::UI::Controls::{
760        TaskDialogIndirect, TASKDIALOG_BUTTON, TASKDIALOGCONFIG, TASKDIALOG_FLAGS,
761        TASKDIALOGCONFIG_0, TASKDIALOGCONFIG_1,
762        TDF_ALLOW_DIALOG_CANCELLATION, TDF_SIZE_TO_CONTENT, TDF_USE_HICON_MAIN,
763        TDCBF_CLOSE_BUTTON, TD_ERROR_ICON, TD_WARNING_ICON,
764    };
765    use windows::Win32::UI::WindowsAndMessaging::{HICON, IDCANCEL};
766    use windows::core::PCWSTR;
767
768    // TDF_CALLBACK_TIMER (0x0040): fires TDN_TIMER every ~200ms so we can poll
769    // the heartbeat and auto-dismiss when the UI thread becomes responsive again.
770    const TDF_CALLBACK_TIMER: TASKDIALOG_FLAGS = TASKDIALOG_FLAGS(0x0040);
771
772    // Make detail and log_dir available to the callback via thread-locals.
773    CB_DETAIL.with(|d| *d.borrow_mut() = detail.to_string());
774    CB_LOG_DIR.with(|d| *d.borrow_mut() = log_dir.to_string());
775    // For hang dialogs: expose the heartbeat so the timer callback can auto-close.
776    if let Some((hb, threshold)) = hb_for_timer {
777        CB_HB_PTR.with(|c| c.set(std::sync::Arc::as_ptr(&hb.inner) as usize));
778        CB_THRESHOLD_MS.with(|c| c.set(threshold.as_millis() as u64));
779    }
780
781    let w = |s: &str| -> Vec<u16> { s.encode_utf16().chain(std::iter::once(0)).collect() };
782
783    let title_w = w(title);
784    let heading_w = w(heading);
785    let summary_w = w(summary);
786    let detail_w = w(detail);
787    let footer_w = w(&format!("Log directory: {log_dir}"));
788    let expand_w = w("Show details");
789    let collapse_w = w("Hide details");
790
791    let lbl_force = w("Force quit");
792    let lbl_wait = w("Keep waiting");
793    let lbl_close = w("Close");
794    let lbl_copy = w("Copy details");
795    let lbl_logs = w("Open log folder");
796
797    // Try custom icon first; fall back to the built-in TD_WARNING/ERROR constants.
798    let custom_hicon: Option<HICON> = DIALOG_ICON_HANDLE
799        .get()
800        .copied()
801        .map(|raw| HICON(raw as *mut _));
802
803    let (extra_flags, anon1) = match custom_hicon {
804        Some(hicon) => (
805            TDF_USE_HICON_MAIN,
806            TASKDIALOGCONFIG_0 { hMainIcon: hicon },
807        ),
808        None => {
809            let icon = if warning { TD_WARNING_ICON } else { TD_ERROR_ICON };
810            (
811                windows::Win32::UI::Controls::TASKDIALOG_FLAGS(0),
812                TASKDIALOGCONFIG_0 { pszMainIcon: icon },
813            )
814        }
815    };
816
817    let mut buttons: Vec<TASKDIALOG_BUTTON> = Vec::new();
818    if warning {
819        buttons.push(TASKDIALOG_BUTTON {
820            nButtonID: BTN_FORCE_QUIT,
821            pszButtonText: PCWSTR(lbl_force.as_ptr()),
822        });
823        buttons.push(TASKDIALOG_BUTTON {
824            nButtonID: IDCANCEL.0 as i32,
825            pszButtonText: PCWSTR(lbl_wait.as_ptr()),
826        });
827    } else {
828        buttons.push(TASKDIALOG_BUTTON {
829            nButtonID: IDCANCEL.0 as i32,
830            pszButtonText: PCWSTR(lbl_close.as_ptr()),
831        });
832    }
833    buttons.push(TASKDIALOG_BUTTON {
834        nButtonID: BTN_COPY,
835        pszButtonText: PCWSTR(lbl_copy.as_ptr()),
836    });
837    buttons.push(TASKDIALOG_BUTTON {
838        nButtonID: BTN_OPEN_LOGS,
839        pszButtonText: PCWSTR(lbl_logs.as_ptr()),
840    });
841
842    let config = TASKDIALOGCONFIG {
843        cbSize: std::mem::size_of::<TASKDIALOGCONFIG>() as u32,
844        hwndParent: HWND(std::ptr::null_mut()),
845        hInstance: HINSTANCE(std::ptr::null_mut()),
846        dwFlags: TDF_SIZE_TO_CONTENT | TDF_ALLOW_DIALOG_CANCELLATION | extra_flags
847            | if hb_for_timer.is_some() { TDF_CALLBACK_TIMER } else { TASKDIALOG_FLAGS(0) },
848        dwCommonButtons: TDCBF_CLOSE_BUTTON,
849        pszWindowTitle: PCWSTR(title_w.as_ptr()),
850        Anonymous1: anon1,
851        pszMainInstruction: PCWSTR(heading_w.as_ptr()),
852        pszContent: PCWSTR(summary_w.as_ptr()),
853        cButtons: buttons.len() as u32,
854        pButtons: buttons.as_ptr(),
855        nDefaultButton: IDCANCEL.0 as i32,
856        cRadioButtons: 0,
857        pRadioButtons: std::ptr::null(),
858        nDefaultRadioButton: 0,
859        pszVerificationText: PCWSTR::null(),
860        pszExpandedInformation: PCWSTR(detail_w.as_ptr()),
861        pszExpandedControlText: PCWSTR(collapse_w.as_ptr()),
862        pszCollapsedControlText: PCWSTR(expand_w.as_ptr()),
863        Anonymous2: TASKDIALOGCONFIG_1 {
864            hFooterIcon: HICON(std::ptr::null_mut()),
865        },
866        pszFooter: PCWSTR(footer_w.as_ptr()),
867        pfCallback: Some(task_dialog_cb),
868        lpCallbackData: 0,
869        cxWidth: 0,
870    };
871
872    let mut btn: i32 = IDCANCEL.0 as i32;
873    let ok = unsafe { TaskDialogIndirect(&config, Some(&mut btn), None, None) };
874    // Clear heartbeat pointer so the callback never sees a dangling reference.
875    CB_HB_PTR.with(|c| c.set(0));
876    CB_THRESHOLD_MS.with(|c| c.set(0));
877
878    if ok.is_err() {
879        let level = if warning {
880            rfd::MessageLevel::Warning
881        } else {
882            rfd::MessageLevel::Error
883        };
884        let _ = rfd::MessageDialog::new()
885            .set_level(level)
886            .set_title(title)
887            .set_buttons(rfd::MessageButtons::Ok)
888            .set_description(summary)
889            .show();
890        return false;
891    }
892
893    // BTN_COPY and BTN_OPEN_LOGS are handled by the callback (S_FALSE keeps the
894    // dialog open so they never appear here). Only BTN_FORCE_QUIT reaches this.
895    matches!(btn, BTN_FORCE_QUIT)
896}

898/// Copy `text` to the system clipboard via PowerShell (avoids wrestling with
899/// Win32 memory handle types across crate versions).
900fn copy_to_clipboard(text: &str) {
901    let tmp = std::env::temp_dir().join("streamarchiver_crash_detail.txt");
902    if std::fs::write(&tmp, text).is_ok() {
903        let _ = std::process::Command::new("powershell")
904            .args([
905                "-NoProfile",
906                "-NonInteractive",
907                "-Command",
908                &format!(
909                    "Get-Content -Raw '{}' | Set-Clipboard",
910                    tmp.display()
911                ),
912            ])
913            .output();
914    }
