14/// True if the channel asset directory has not been fetched in the last 24 hours.
15pub fn should_refetch_assets(asset_dir: &Path) -> bool {
16    let stamp = asset_dir.join(".assets_fetched_at");
17    match std::fs::read_to_string(&stamp) {
18        Ok(s) => {
19            let fetched: i64 = s.trim().parse().unwrap_or(0);
20            now_unix() - fetched > 86_400
21        }
22        Err(_) => true,
23    }
24}

66/// First file in `dir` whose name starts with `prefix` (e.g. `"banner."`). Used to
67/// locate a canonical channel asset (`icon.png`, `banner.jpg`) without knowing its
68/// extension. Skips the `history/` subdir and archived `{stem}_{ts}.ext` variants
69/// since those don't start with `{stem}.`.
70pub(crate) fn find_asset(dir: &Path, prefix: &str) -> Option<PathBuf> {
71    std::fs::read_dir(dir)
72        .ok()?
73        .flatten()
74        .map(|e| e.path())
75        .find(|p| {
76            p.file_name()
77                .and_then(|n| n.to_str())
78                .is_some_and(|n| n.starts_with(prefix))
79        })
80}

82/// Derive a file extension from a URL path (before `?` query string).
83fn ext_from_url(url: &str) -> Option<&str> {
84    let path = url.split('?').next()?;
85    let ext = path.rsplit('.').next()?;
86    if ext.len() <= 5 && ext.chars().all(|c| c.is_ascii_alphanumeric()) {
87        Some(ext)
88    } else {
89        None
90    }

93/// Ensure a `px × px` downscaled copy of the channel icon exists at
94/// `asset_dir/icon_{px}.png`. Generated with Lanczos3 on first call; the cached
95/// file is reused on subsequent calls unless the source icon is newer (mtime
96/// check). Returns the path to the scaled file, or `None` when no source icon
97/// is present or image processing fails. When the source is already ≤ `px` the
98/// source path is returned as-is (no unnecessary upscaling).
99pub fn ensure_scaled_icon(asset_dir: &Path, px: u32) -> Option<PathBuf> {
100    let out = asset_dir.join(format!("icon_{px}.png"));
101    let src = find_asset(asset_dir, "icon.")?;
102
103    if out.exists() {
104        // Regenerate only if the source icon was updated after the last scale.
105        let src_mtime = std::fs::metadata(&src).ok()?.modified().ok()?;
106        let out_mtime = std::fs::metadata(&out).ok()?.modified().ok()?;
107        if out_mtime >= src_mtime {
108            return Some(out);
109        }
110    }
111
112    let bytes = std::fs::read(&src).ok()?;
113    let img = image::load_from_memory(&bytes).ok()?.to_rgba8();
114    if img.width() <= px && img.height() <= px {
115        return Some(src);
116    }
117    let scaled = image::imageops::resize(&img, px, px, image::imageops::FilterType::Lanczos3);
118    scaled.save(&out).ok()?;
119    Some(out)
120}

122/// Download a URL to a file path; creates parent directories as needed.
123pub(crate) async fn download_image(client: &Client, url: &str, dest: &Path) -> Result<()> {
124    let url = if url.starts_with("//") {
125        format!("https:{url}")
126    } else {
127        url.to_string()
128    };
129    if let Some(parent) = dest.parent() {
130        tokio::fs::create_dir_all(parent).await?;
131    }
132    let resp = client.get(&url).send().await?;
133    if !resp.status().is_success() {
134        bail!("HTTP {} for {}", resp.status(), url);
135    }
136    let bytes = resp.bytes().await?;
137    tokio::fs::write(dest, bytes).await?;
138    Ok(())
139}

141/// The current canonical asset file `dir/{stem}.<ext>` (any extension), if one
142/// exists. Matches `icon.png` but never the `history/` dir or an archived
143/// `icon_<ts>.png` (those use `{stem}_`, not `{stem}.`).
144async fn current_asset(dir: &Path, stem: &str) -> Option<PathBuf> {
145    let prefix = format!("{stem}.");
146    let mut rd = tokio::fs::read_dir(dir).await.ok()?;
147    while let Ok(Some(entry)) = rd.next_entry().await {
148        if entry.file_name().to_string_lossy().starts_with(&prefix) {
149            return Some(entry.path());
150        }
151    }
152    None
153}

155/// Download a per-channel **singular** asset (icon / banner) into `dir`,
156/// preserving history — this is an archiver, so a profile pic / banner the
157/// channel later changes must not be lost.
158///
159/// `dir/{stem}.{ext}` always holds the latest version. When the freshly fetched
160/// image differs (byte-for-byte) from the current canonical file, the old one is
161/// moved into `dir/history/{stem}_{retired_at}.{old_ext}` before being replaced;
162/// `retired_at` is the unix time it was supplanted, so the history reads as a
163/// timeline. An identical re-download is a no-op (no spurious history entry).
164async fn download_image_archival(
165    client: &Client,
166    url: &str,
167    dir: &Path,
168    stem: &str,
169    ext: &str,
170) -> Result<()> {
171    let url = if url.starts_with("//") {
172        format!("https:{url}")
173    } else {
174        url.to_string()
175    };
176    tokio::fs::create_dir_all(dir).await?;
177    let resp = client.get(&url).send().await?;
178    if !resp.status().is_success() {
179        bail!("HTTP {} for {}", resp.status(), url);
180    }
181    let bytes = resp.bytes().await?;
182    archive_and_write(dir, stem, ext, &bytes).await
183}

185/// Place `bytes` as the canonical `dir/{stem}.{ext}`, archiving any differing
186/// current version into `dir/history/` first. Network-free (the testable core of
187/// [`download_image_archival`]). A byte-identical current file is left untouched
188/// (no spurious history entry); a differing one is moved to
189/// `history/{stem}_{retired_at}.{old_ext}` so it is never lost.
190async fn archive_and_write(dir: &Path, stem: &str, ext: &str, bytes: &[u8]) -> Result<()> {
191    if let Some(cur_path) = current_asset(dir, stem).await {
192        match tokio::fs::read(&cur_path).await {
193            // Unchanged since last fetch — leave everything as-is.
194            Ok(cur) if cur == bytes => return Ok(()),
195            // Changed — archive the old version before it's overwritten.
196            Ok(_) => {
197                let hist = dir.join("history");
198                tokio::fs::create_dir_all(&hist).await?;
199                let cur_ext = cur_path
200                    .extension()
201                    .and_then(|e| e.to_str())
202                    .unwrap_or("img");
203                // Name by retirement time, but never collide with an existing
204                // archived version (two changes in the same second, or a clock
205                // that didn't advance) — append a counter so nothing is lost.
206                let ts = now_unix();
207                let mut archived = hist.join(format!("{stem}_{ts}.{cur_ext}"));
208                let mut n = 1;
209                while tokio::fs::try_exists(&archived).await.unwrap_or(false) {
210                    n += 1;
211                    archived = hist.join(format!("{stem}_{ts}_{n}.{cur_ext}"));
212                }
213                // Move the old canonical into history (rename; fall back to
214                // copy+remove if the move fails). This also clears a stale
215                // canonical whose extension differs from the new one.
216                if tokio::fs::rename(&cur_path, &archived).await.is_err() {
217                    tokio::fs::copy(&cur_path, &archived).await?;
218                    let _ = tokio::fs::remove_file(&cur_path).await;
219                }
220                // Log the replacement so the change-history can show it. `stem` is
221                // "icon"/"banner"; `id` points at the archived previous version.
222                let archived_name = archived
223                    .file_name()
224                    .and_then(|n| n.to_str())
225                    .unwrap_or_default()
226                    .to_string();
227                append_asset_changes(
228                    dir,
229                    &[AssetChange {
230                        at: ts,
231                        kind: stem.to_string(),
232                        provider: String::new(),
233                        action: "changed".to_string(),
234                        name: String::new(),
235                        id: archived_name,
236                        old: String::new(),
237                        new: String::new(),
238                    }],
239                )
240                .await;
241            }
242            // Unreadable current file — just overwrite it.
243            Err(_) => {}
244        }
245    }
246
247    tokio::fs::write(dir.join(format!("{stem}.{ext}")), bytes).await?;
248    Ok(())
249}

265async fn fetch_twitch_channel_assets(

275        #[serde(default)]

281    }

296    };
297

299
300    let icon_ext = ext_from_url(&user.profile_image_url).unwrap_or("jpg");

305    }
306
307    if !user.offline_image_url.is_empty() {
308        let banner_ext = ext_from_url(&user.offline_image_url).unwrap_or("jpg");
309        if let Err(e) =
310            download_image_archival(client, &user.offline_image_url, asset_dir, "banner", banner_ext)
311                .await
312        {
313            warn!("twitch banner: {e}");
314        }
315    }
316    Ok(())

441/// Download Twitch channel emotes into `asset_dir/emotes/twitch/` and write a
442/// per-channel manifest `asset_dir/emotes/twitch.json`. Mirrors the BTTV/FFZ/7TV
443/// pattern so Twitch emotes also have named files and diff/history tracking.
444async fn fetch_twitch_emotes(
445    client: &Client,
446    client_id: &str,
447    token: &str,
448    broadcaster_id: &str,
449    asset_dir: &Path,
450) -> Result<()> {
451    if broadcaster_id.is_empty() {
452        return Ok(());
453    }
454    let emote_dir = asset_dir.join("emotes").join("twitch");
455    tokio::fs::create_dir_all(&emote_dir).await?;
456
457    let url = format!(
458        "https://api.twitch.tv/helix/chat/emotes?broadcaster_id={broadcaster_id}"
459    );
460    let resp = client
461        .get(&url)
462        .header("Client-Id", client_id)
463        .bearer_auth(token)
464        .send()
465        .await?;
466    if !resp.status().is_success() {
467        bail!("Helix emotes: {}", resp.status());
468    }
469    let r: HelixEmotesResp = resp.json().await?;
470
471    let mut manifest: Vec<EmoteManifestEntry> = Vec::new();
472
473    for emote in &r.data {
474        let animated = emote.format.iter().any(|f| f == "animated");
475        let (src_url, ext) = if animated {
476            (
477                format!(
478                    "https://static-cdn.jtvnw.net/emoticons/v2/{}/animated/dark/3.0",
479                    emote.id
480                ),
481                "gif",
482            )
483        } else {
484            (emote.images.url_4x.clone(), "png")
485        };
486        manifest.push(EmoteManifestEntry {
487            name: emote.name.clone(),
488            id: emote.id.clone(),
489            ext: ext.to_string(),
490            shared: false,
491        });
492        // New downloads get `{id}_{name}.{ext}`; old `{id}.{ext}` files are kept
493        // as-is (the viewer resolver falls back to them).
494        let new_dest = emote_dir.join(format!(
495            "{}_{}.{ext}",
496            emote.id,
497            sanitize_emote_name(&emote.name)
498        ));
499        let old_dest = emote_dir.join(format!("{}.{ext}", emote.id));
500        if asset_present(&new_dest) || asset_present(&old_dest) {
501            continue;
502        }
503        if let Err(e) = download_image(client, &src_url, &new_dest).await {
504            warn!("Twitch emote {}: {e}", emote.id);
505        }
506    }
507
508    if !manifest.is_empty() {
509        record_manifest_change(asset_dir, "twitch", &manifest).await;
510        if let Ok(json) = serde_json::to_string(&manifest) {
511            let _ = tokio::fs::write(asset_dir.join("emotes").join("twitch.json"), json).await;
512        }
513    }
514
515    Ok(())
516}

539/// A previously-downloaded asset is "present" only if it exists AND is non-empty.
540/// `download_image` writes non-atomically (truncate-then-write), so an interrupted
541/// fetch can leave a 0-byte file; treating that as absent lets a later pass repair
542/// it instead of the `exists()` guard pinning the corrupt file forever.
543fn asset_present(path: &Path) -> bool {
544    std::fs::metadata(path).map(|m| m.len() > 0).unwrap_or(false)
545}

547/// Sanitize an emote code for use as a filename component. Keeps alphanumerics,
548/// underscores, and hyphens; replaces anything else with `_`.
549pub(crate) fn sanitize_emote_name(name: &str) -> String {
550    name.chars()
551        .map(|c| if c.is_alphanumeric() || c == '_' || c == '-' { c } else { '_' })
552        .collect()
553}

555// ---------- Asset change history ----------
556
557/// One recorded change to a channel's assets, appended as a JSON line to the
558/// per-channel-platform `asset_changes.jsonl`. This is the queryable companion to
559/// the filesystem `history/` archives: the images/manifests preserve the *bytes*,
560/// this log preserves *what changed and when* so the UI can present a timeline.
561///
562/// Why a log at all: emote manifests are overwritten wholesale on every refetch,
563/// so a code the streamer later removes would otherwise vanish without a trace
564/// (it's not even caught as "deprecated", since it's no longer in the manifest).
565/// Diffing the old manifest against the new before the overwrite records the
566/// removal (and additions) here, permanently.
567///
568/// The `kind`/`action` pair is a small open vocabulary the UI maps to display:
569/// - `kind = "emote"`  → `action` `"added"`/`"removed"`, with `provider`
570///   (`"7tv"`/`"bttv"`/`"ffz"`), `name` (the code) and `id`.
571/// - `kind = "icon"`/`"banner"` → `action` `"changed"`; `id` is the archived
572///   filename kept under `history/`.
573/// - `kind = "name_color"` → `action` `"added"`/`"removed"`/`"changed"` with the
574///   `old`/`new` hex strings.
575#[derive(Clone, Debug, serde::Serialize, serde::Deserialize)]
576pub struct AssetChange {
577    /// Unix seconds the change was recorded (when the refetch saw it).
578    pub at: i64,
579    pub kind: String,
580    /// Emote provider stem for `kind = "emote"` (`"7tv"`/`"bttv"`/`"ffz"`); empty otherwise.
581    #[serde(default, skip_serializing_if = "String::is_empty")]
582    pub provider: String,
583    pub action: String,
584    /// Emote code (for `kind = "emote"`); empty otherwise.
585    #[serde(default, skip_serializing_if = "String::is_empty")]
586    pub name: String,
587    /// Emote id, or the archived `history/` filename for icon/banner; empty otherwise.

592    pub old: String,

594    #[serde(default, skip_serializing_if = "String::is_empty")]
595    pub new: String,

598impl AssetChange {
599    fn emote(at: i64, provider: &str, action: &str, name: &str, id: &str) -> AssetChange {

601            at,
602            kind: "emote".to_string(),
603            provider: provider.to_string(),
604            action: action.to_string(),
605            name: name.to_string(),
606            id: id.to_string(),
607            old: String::new(),
608            new: String::new(),
609        }
610    }

613/// Diff two emote manifests by **code** (case-sensitive, as typed in chat) and
614/// return one [`AssetChange`] per added/removed code. Empty/whitespace codes are
615/// ignored (legacy name-less entries can never match chat anyway). An id-only
616/// change to an existing code yields nothing — only the code set matters here, so
617/// churn in ids/urls doesn't spam the history. Output is sorted by code so the log
618/// (and the unit test) is deterministic.
619fn diff_emote_manifest(
620    old: &[EmoteManifestEntry],
621    new: &[EmoteManifestEntry],
622    provider: &str,
623    at: i64,
624) -> Vec<AssetChange> {
625    use std::collections::HashMap;
626    let index = |m: &[EmoteManifestEntry]| -> HashMap<String, String> {
627        m.iter()
628            .filter(|e| !e.name.trim().is_empty())
629            .map(|e| (e.name.clone(), e.id.clone()))
630            .collect()
631    };
632    let old_idx = index(old);
633    let new_idx = index(new);
634    let mut out: Vec<AssetChange> = Vec::new();
635    for (name, id) in &old_idx {
636        if !new_idx.contains_key(name) {
637            out.push(AssetChange::emote(at, provider, "removed", name, id));
638        }
639    }
640    for (name, id) in &new_idx {
641        if !old_idx.contains_key(name) {
642            out.push(AssetChange::emote(at, provider, "added", name, id));
643        }
644    }
645    // A code is in at most one of {added, removed}, so sorting by code alone is a
646    // total, stable order.
647    out.sort_by_key(|c| c.name.to_lowercase());
648    out
649}

651/// Append change records to the channel-platform `asset_changes.jsonl` (one JSON
652/// object per line, append-only). Best-effort: a write failure is swallowed — the
653/// history is a convenience layer and must never abort or fail an asset fetch.
654async fn append_asset_changes(asset_dir: &Path, changes: &[AssetChange]) {
655    if changes.is_empty() {
656        return;
657    }
658    let mut buf = String::new();
659    for c in changes {
660        if let Ok(line) = serde_json::to_string(c) {
661            buf.push_str(&line);
662            buf.push('\n');
663        }
664    }
665    if buf.is_empty() || tokio::fs::create_dir_all(asset_dir).await.is_err() {
666        return;
667    }
668    use tokio::io::AsyncWriteExt;
669    if let Ok(mut f) = tokio::fs::OpenOptions::new()
670        .create(true)
671        .append(true)
672        .open(asset_dir.join("asset_changes.jsonl"))
673        .await
674    {
675        let _ = f.write_all(buf.as_bytes()).await;
676    }
677}

679/// Read a channel-platform's recorded asset changes (`asset_changes.jsonl`) in
680/// chronological append order (oldest first). Malformed/blank lines are skipped;
681/// a missing file yields an empty vec. Synchronous — the UI calls it directly on
682/// popup-open (the file is tiny: a handful of lines per refetch).
683pub fn read_asset_changes(asset_dir: &Path) -> Vec<AssetChange> {
684    let Ok(s) = std::fs::read_to_string(asset_dir.join("asset_changes.jsonl")) else {
685        return Vec::new();
686    };
687    s.lines()
688        .filter(|l| !l.trim().is_empty())
689        .filter_map(|l| serde_json::from_str::<AssetChange>(l).ok())
690        .collect()
691}

693/// Record what a freshly-fetched emote manifest changed, *before* it overwrites the
694/// previous one at `asset_dir/emotes/{provider}.json`. The prior manifest is read,
695/// diffed by code against `new`, and on any add/remove: the old manifest is
696/// snapshotted to `emotes/history/{provider}_{ts}.json` (full archival, mirroring
697/// the icon/banner `history/`) and the per-emote changes are appended to
698/// `asset_changes.jsonl`. A no-op on the first fetch (no prior manifest = baseline,
699/// not a change) or when the code set is unchanged. `provider` is the manifest stem
700/// (`"7tv"`/`"bttv"`/`"ffz"`). The current manifest file is left in place — the
701/// caller writes the new one right after, so the canonical manifest is never
702/// missing even if this snapshot write fails.
703async fn record_manifest_change(asset_dir: &Path, provider: &str, new: &[EmoteManifestEntry]) {
704    let emotes_dir = asset_dir.join("emotes");
705    let manifest_path = emotes_dir.join(format!("{provider}.json"));
706    // Retry on transient lock errors (e.g. Windows Defender scanning a newly written
707    // file on CI). Return immediately on NotFound — that's the expected first-fetch
708    // baseline case, not an error.
709    let old_json = {
710        let mut outcome = None;
711        for attempt in 0..4u32 {
712            if attempt > 0 {
713                tokio::time::sleep(tokio::time::Duration::from_millis(50)).await;
714            }
715            match tokio::fs::read_to_string(&manifest_path).await {
716                Ok(s) => {
717                    outcome = Some(s);
718                    break;
719                }
720                Err(e) if e.kind() == std::io::ErrorKind::NotFound => return,
721                Err(_) => {}
722            }
723        }
724        match outcome {
725            Some(s) => s,
726            None => return,
727        }
728    };
729    // Treat a corrupt/truncated prior manifest as "unknown" and bail, mirroring the
730    // missing-file early return above. Defaulting to an empty Vec would diff every
731    // current emote as a fresh "add" and snapshot a manifest we couldn't even parse.
732    let Ok(old) = serde_json::from_str::<Vec<EmoteManifestEntry>>(&old_json) else {
733        return;
734    };
735    let at = now_unix();
736    let changes = diff_emote_manifest(&old, new, provider, at);
737    if changes.is_empty() {
738        return;
739    }
740    // Snapshot the prior manifest (full archival) before it's overwritten. Written
741    // from the in-memory bytes, not a rename, so the canonical path stays valid.
742    let hist = emotes_dir.join("history");
743    if tokio::fs::create_dir_all(&hist).await.is_ok() {
744        let mut dest = hist.join(format!("{provider}_{at}.json"));
745        let mut n = 1;
746        while tokio::fs::try_exists(&dest).await.unwrap_or(false) {
747            n += 1;
748            dest = hist.join(format!("{provider}_{at}_{n}.json"));
749        }
750        let _ = tokio::fs::write(&dest, old_json.as_bytes()).await;
751    }
752    append_asset_changes(asset_dir, &changes).await;
753}

755/// Download BTTV emotes:
756/// - Channel emotes → `asset_dir/emotes/bttv/{id}.ext` (per-channel, unchanged)
757/// - Shared emotes  → `platform_dir/bttv/emotes/{id}.ext` (global dedup, skip if present)
758/// Writes a manifest `asset_dir/emotes/bttv.json` listing all active emote IDs for this channel.
759async fn fetch_bttv_emotes(
760    client: &Client,
761    broadcaster_id: &str,
762    asset_dir: &Path,
763    platform_dir: &Path,
764) -> Result<()> {
765    if broadcaster_id.is_empty() {
766        return Ok(());
767    }
768    #[derive(Deserialize)]
769    struct BttvEmote {
770        id: String,
771        /// The emote word as typed in chat (e.g. `modCheck`). `#[serde(default)]`
772        /// so one malformed emote can't abort the whole channel's BTTV fetch; an
773        /// empty code just yields an unmatchable manifest entry (reader skips it).
774        #[serde(default)]
775        code: String,
776        #[serde(rename = "imageType")]
777        image_type: String,
778    }
779    #[derive(Deserialize)]
780    struct BttvResp {
781        #[serde(rename = "channelEmotes", default)]
782        channel_emotes: Vec<BttvEmote>,
783        #[serde(rename = "sharedEmotes", default)]
784        shared_emotes: Vec<BttvEmote>,
785    }
786
787    let url = format!(
788        "https://api.betterttv.net/3/cached/users/twitch/{broadcaster_id}"
789    );
790    let resp = client.get(&url).send().await?;
791    // 404 = channel has no BTTV emotes; that's normal
792    if resp.status() == reqwest::StatusCode::NOT_FOUND {
793        return Ok(());
794    }
795    if !resp.status().is_success() {
796        bail!("BTTV: {}", resp.status());
797    }
798    let r: BttvResp = resp.json().await?;
799
800    let mut manifest: Vec<EmoteManifestEntry> = Vec::new();
801
802    // Channel emotes — per-channel directory
803    if !r.channel_emotes.is_empty() {
804        let dir = asset_dir.join("emotes").join("bttv");
805        tokio::fs::create_dir_all(&dir).await?;
806        for emote in &r.channel_emotes {
807            manifest.push(EmoteManifestEntry {
808                name: emote.code.clone(),
809                id: emote.id.clone(),
810                ext: emote.image_type.clone(),
811                shared: false,
812            });
813            let new_dest = dir.join(format!(
814                "{}_{}.{}",
815                emote.id,
816                sanitize_emote_name(&emote.code),
817                emote.image_type
818            ));
819            let old_dest = dir.join(format!("{}.{}", emote.id, emote.image_type));
820            if asset_present(&new_dest) || asset_present(&old_dest) {
821                continue;
822            }
823            let url = format!(
824                "https://cdn.betterttv.net/emote/{}/3x.{}",
825                emote.id, emote.image_type
826            );
827            if let Err(e) = download_image(client, &url, &new_dest).await {
828                warn!("BTTV channel emote {}: {e}", emote.id);
829            } else {
830                tokio::time::sleep(Duration::from_millis(150)).await;
831            }
832        }
833    }
834
835    // Shared emotes — global dedup cache
836    if !r.shared_emotes.is_empty() {
837        let global_dir = platform_dir.join("bttv").join("emotes");
838        tokio::fs::create_dir_all(&global_dir).await?;
839        for emote in &r.shared_emotes {
840            manifest.push(EmoteManifestEntry {
841                name: emote.code.clone(),
842                id: emote.id.clone(),
843                ext: emote.image_type.clone(),
844                shared: true,
845            });
846            let new_dest = global_dir.join(format!(
847                "{}_{}.{}",
848                emote.id,
849                sanitize_emote_name(&emote.code),
850                emote.image_type
851            ));
852            let old_dest = global_dir.join(format!("{}.{}", emote.id, emote.image_type));
853            if asset_present(&new_dest) || asset_present(&old_dest) {
854                continue;
855            }
856            let url = format!(
857                "https://cdn.betterttv.net/emote/{}/3x.{}",
858                emote.id, emote.image_type
859            );
860            if let Err(e) = download_image(client, &url, &new_dest).await {
861                warn!("BTTV shared emote {}: {e}", emote.id);
862            } else {
863                tokio::time::sleep(Duration::from_millis(150)).await;
864            }
865        }
866    }
867
868    // Write manifest listing all active emote IDs for this channel
869    if !manifest.is_empty() {
870        let manifest_dir = asset_dir.join("emotes");
871        tokio::fs::create_dir_all(&manifest_dir).await?;
872        // Record added/removed codes against the previous manifest before overwriting.
873        record_manifest_change(asset_dir, "bttv", &manifest).await;
874        if let Ok(json) = serde_json::to_string(&manifest) {
875            let _ = tokio::fs::write(manifest_dir.join("bttv.json"), json).await;
876        }
877    }
878
879    Ok(())
880}

1152/// - FFZ emotes → `platform_dir/ffz/emotes/` + manifest `asset_dir/emotes/ffz.json`
1153/// - 7TV emotes → `platform_dir/7tv/emotes/` + manifest `asset_dir/emotes/7tv.json`
1154/// - Broadcaster name colour → `asset_dir/name_color.txt` (Helix `chat/color`)
1155/// Returns `true` if the channel icon/banner fetch succeeded (badges/emotes/colour
1156/// are best-effort and don't affect the result). The 24h "fetched" stamp is written
1157/// **only on success**, so a failed fetch (e.g. empty/invalid `broadcaster_id`,
1158/// API error) is retried instead of being blocked for a day.
1159pub async fn run_twitch_assets(
1160    client: &Client,
1161    client_id: &str,
1162    token: &str,
1163    broadcaster_id: &str,
1164    asset_dir: &Path,
1165    platform_dir: &Path,
1166) -> bool {
1167    let ok = match fetch_twitch_channel_assets(client, client_id, token, broadcaster_id, asset_dir)
1168        .await
1169    {
1170        Ok(()) => true,
1171        Err(e) => {
1172            warn!("Twitch channel assets ({broadcaster_id}): {e}");
1173            false
1174        }
1175    };
1176    tokio::time::sleep(Duration::from_millis(300)).await;
1177    if let Err(e) =
1178        fetch_twitch_badges(client, client_id, token, broadcaster_id, asset_dir, platform_dir).await
1179    {
1180        warn!("Twitch badges ({broadcaster_id}): {e}");
1181    }
1182    tokio::time::sleep(Duration::from_millis(300)).await;
1183    if let Err(e) =
1184        fetch_twitch_emotes(client, client_id, token, broadcaster_id, asset_dir).await
1185    {
1186        warn!("Twitch emotes ({broadcaster_id}): {e}");
1187    }
1188    tokio::time::sleep(Duration::from_millis(300)).await;
1189    if let Err(e) = fetch_bttv_emotes(client, broadcaster_id, asset_dir, platform_dir).await {
1190        warn!("BTTV ({broadcaster_id}): {e}");
1191    }
1192    if let Err(e) = fetch_ffz_emotes(client, broadcaster_id, asset_dir, platform_dir).await {
1193        warn!("FFZ ({broadcaster_id}): {e}");
1194    }
1195    if let Err(e) = fetch_7tv_emotes(client, broadcaster_id, asset_dir, platform_dir).await {
1196        warn!("7TV ({broadcaster_id}): {e}");
1197    }
1198    tokio::time::sleep(Duration::from_millis(300)).await;
1199    if let Err(e) =
1200        fetch_twitch_name_color(client, client_id, token, broadcaster_id, asset_dir).await
1201    {
1202        warn!("Twitch name color ({broadcaster_id}): {e}");
1203    }
1204    if ok {
1205        write_fetched_stamp(asset_dir);
1206    }
1207    ok
1208}

1210/// Fetch the broadcaster's chosen Twitch chat name colour (Helix `chat/color`) and
1211/// cache it as `asset_dir/name_color.txt` (e.g. `#9146FF`). The chat replay uses
1212/// the IRC `color` tag directly, but this lets the Streams list tint a Twitch
1213/// channel's name with the streamer's own colour. No file is written when the user
1214/// hasn't set a colour (Helix returns an empty string), so the UI falls back to its
1215/// automatic palette.
1216async fn fetch_twitch_name_color(
1217    client: &Client,
1218    client_id: &str,
1219    token: &str,
1220    broadcaster_id: &str,
1221    asset_dir: &Path,
1222) -> Result<()> {
1223    if broadcaster_id.is_empty() {
1224        return Ok(());
1225    }
1226    let url = format!("https://api.twitch.tv/helix/chat/color?user_id={broadcaster_id}");
1227    let resp = client
1228        .get(&url)
1229        .header("Client-Id", client_id)
1230        .bearer_auth(token)
1231        .send()
1232        .await?;
1233    if !resp.status().is_success() {
1234        bail!("Helix chat color: {}", resp.status());
1235    }
1236    let v: serde_json::Value = resp.json().await?;
1237    let color = v["data"][0]["color"].as_str().unwrap_or("").trim().to_string();
1238    let dest = asset_dir.join("name_color.txt");
1239    // Read the previous colour first so we can log a transition (and only a real one).
1240    let old_color = std::fs::read_to_string(&dest)
1241        .map(|s| s.trim().to_string())
1242        .unwrap_or_default();
1243    if color.is_empty() {
1244        // Broadcaster cleared their colour — drop any stale cache so the UI reverts
1245        // to the automatic palette instead of tinting with a colour no longer used.
1246        let _ = tokio::fs::remove_file(&dest).await;
1247    } else {
1248        tokio::fs::create_dir_all(asset_dir).await?;
1249        let _ = tokio::fs::write(&dest, &color).await;
1250    }
1251    // Only log a transition once a baseline exists. On the first-ever fetch the
1252    // stamp is absent, so a name colour appearing for the first time is the baseline
1253    // (silent), not an "added" change — consistent with emote/icon/banner baselines.
1254    if old_color != color && assets_ever_fetched(asset_dir) {
1255        let action = if old_color.is_empty() {
1256            "added"
1257        } else if color.is_empty() {
1258            "removed"
1259        } else {
1260            "changed"
1261        };
1262        append_asset_changes(
1263            asset_dir,
1264            &[AssetChange {
1265                at: now_unix(),
1266                kind: "name_color".to_string(),
1267                provider: String::new(),
1268                action: action.to_string(),
1269                name: String::new(),
1270                id: String::new(),
1271                old: old_color,
1272                new: color,
1273            }],
1274        )
1275        .await;
1276    }
1277    Ok(())
1278}

1280/// Extract the channel banner URL from a parsed `ytInitialData` blob, trying the
1281/// newer `pageHeaderRenderer` path first, then the classic `c4TabbedHeaderRenderer`
1282/// path. Returns `None` when no banner is found (channel has no art set).
1283fn youtube_banner_from_page_data(data: &serde_json::Value) -> Option<String> {
1284    // New format (2024+): pageHeaderRenderer → imageBannerViewModel
1285    if let Some(sources) = data["header"]["pageHeaderRenderer"]["banner"]
1286        ["imageBannerViewModel"]["image"]["sources"]
1287        .as_array()
1288    {
1289        if let Some(url) = sources.last().and_then(|s| s["url"].as_str()) {
1290            return Some(normalize_yt_banner_url(url));
1291        }
1292    }
1293    // Legacy format: c4TabbedHeaderRenderer → banner → thumbnails
1294    if let Some(thumbs) = data["header"]["c4TabbedHeaderRenderer"]["banner"]["thumbnails"]
1295        .as_array()
1296    {
1297        if let Some(url) = thumbs.last().and_then(|t| t["url"].as_str()) {
1298            return Some(normalize_yt_banner_url(url));
1299        }
1300    }
1301    None
1302}

1304/// Request the widest available crop of a YouTube banner URL. YouTube banner URLs
1305/// on googleusercontent.com carry a `=w<N>-fcrop64=…` suffix; stripping it and
1306/// appending `=w2560` gives the full-width version (2560 px, the maximum YouTube
1307/// serves). Non-Google URLs are returned unchanged.
1308fn normalize_yt_banner_url(url: &str) -> String {
1309    if url.contains("googleusercontent.com") || url.contains("ggpht.com") {
1310        if let Some((base, _)) = url.split_once('=') {
1311            return format!("{base}=w2560");
1312        }
1313    }
1314    url.to_string()
1315}

1317/// Fetch the YouTube channel page and download the page-header banner from
1318/// `ytInitialData`. Works without a YouTube API key — just scrapes the channel
1319/// home page. Saves as `banner.<ext>` in `asset_dir/` with history preservation.
1320async fn fetch_youtube_page_banner(
1321    client: &Client,
1322    channel_url: &str,
1323    asset_dir: &Path,
1324    fingerprint: Option<&BrowserFingerprint>,
1325) -> Result<()> {
1326    let base = {
1327        let t = channel_url.trim().trim_end_matches('/');
1328        t.strip_suffix("/live")
1329            .or_else(|| t.strip_suffix("/streams"))
1330            .or_else(|| t.strip_suffix("/community"))
1331            .unwrap_or(t)
1332            .to_string()
1333    };
1334    let rb = client
1335        .get(&base)
1336        .query(&[("hl", "en"), ("gl", "US")])
1337        .header("Accept-Language", "en-US,en;q=0.9")
1338        .header("Cookie", "CONSENT=YES+1; SOCS=CAI");
1339    let rb = if let Some(fp) = fingerprint {
1340        fp.apply_yt_nav_headers(rb)
1341    } else {
1342        rb
1343    };
1344    let resp = rb.send().await?;
1345    if !resp.status().is_success() {
1346        bail!("YouTube channel page: {}", resp.status());
1347    }
1348    let body = resp.text().await?;
1349    let data = crate::detectors::extract_json_after(&body, "ytInitialData")
1350        .ok_or_else(|| anyhow::anyhow!("ytInitialData not found"))?;
1351    let banner_url = youtube_banner_from_page_data(&data)
1352        .ok_or_else(|| anyhow::anyhow!("no banner found in ytInitialData"))?;
1353    let ext = ext_from_url(&banner_url).unwrap_or("jpg");
1354    tokio::fs::create_dir_all(asset_dir).await?;
1355    download_image_archival(client, &banner_url, asset_dir, "banner", ext).await
1356}

1358/// Run YouTube channel asset fetches. Tries two approaches:
1359/// 1. YouTube Data API (icon + banner + branding) when `api_key` and `channel_id`
1360///    are both non-empty.
1361/// 2. Page-scrape banner (fetches the channel home page, extracts the wide
1362///    page-header banner from `ytInitialData`) — works without an API key.
1363///
1364/// The page-scrape banner is a **fallback**: it runs only when the API path did
1365/// not already write a banner. The two sources expose different banner images
1366/// (the API's `bannerExternalUrl` vs the page-header banner), so writing both on
1367/// every fetch made them overwrite each other and spam the change history with
1368/// phantom "banner replaced" entries. The 24 h stamp is written only when at
1369/// least one approach succeeds.
1370pub async fn run_youtube_assets(
1371    client: &Client,
1372    api_key: &str,
1373    channel_id: &str,
1374    channel_url: &str,
1375    asset_dir: &Path,
1376    fingerprint: Option<&BrowserFingerprint>,
1377) -> bool {
1378    let mut any_ok = false;
1379    let mut api_set_banner = false;
1380
1381    if !api_key.is_empty() && !channel_id.is_empty() {
1382        match fetch_youtube_channel_assets(client, api_key, channel_id, asset_dir).await {
1383            Ok(banner_set) => {
1384                any_ok = true;
1385                api_set_banner = banner_set;
1386            }
1387            Err(e) => warn!("YouTube channel assets ({channel_id}): {e}"),
1388        }
1389    }
1390
1391    // Fallback only: skip the page scrape entirely when the API already supplied a
1392    // banner, so a single channel never alternates between two banner sources.
1393    if !api_set_banner && !channel_url.is_empty() {
1394        match fetch_youtube_page_banner(client, channel_url, asset_dir, fingerprint).await {
1395            Ok(()) => any_ok = true,
1396            Err(e) if !any_ok => warn!("YouTube page banner ({channel_url}): {e}"),
1397            Err(_) => {}
1398        }
1399    }
1400
1401    if any_ok {
1402        write_fetched_stamp(asset_dir);
1403    }
1404    any_ok
1405}

1425    /// A fresh, unique temp directory for a test. Combines the pid, a
1426    /// process-lifetime counter, and a nanosecond timestamp so a directory left
1427    /// behind by a *panicking* run (whose end-of-test cleanup never runs) can
1428    /// never be reused — even if the OS recycles the pid — which would otherwise
1429    /// let stale `asset_changes.jsonl` lines leak into a later run's assertions.
1430    fn unique_test_dir(prefix: &str) -> std::path::PathBuf {
1431        use std::sync::atomic::{AtomicU64, Ordering};
1432        static SEQ: AtomicU64 = AtomicU64::new(0);
1433        let n = SEQ.fetch_add(1, Ordering::Relaxed);
1434        let nanos = std::time::SystemTime::now()
1435            .duration_since(std::time::UNIX_EPOCH)
1436            .map(|d| d.as_nanos())
1437            .unwrap_or(0);
1438        let dir = std::env::temp_dir()
1439            .join(format!("{prefix}-{}-{n}-{nanos}", std::process::id()));
1440        let _ = std::fs::remove_dir_all(&dir);
1441        dir
1442    }
