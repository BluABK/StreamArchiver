1//! Unicode-emoji detection + Twemoji asset mapping for the chat replay.
2//!
3//! egui's renderer draws glyph outlines only (no colour fonts), so colour emoji
4//! must be rendered as images. We segment chat text into emoji vs non-emoji runs,
5//! map each emoji grapheme cluster to a [Twemoji](https://github.com/jdecked/twemoji)
6//! PNG filename, and the UI downloads/caches those on demand (CC-BY 4.0 — attributed
7//! in Settings). Detection is deliberately *liberal*: a false positive just triggers
8//! a Twemoji fetch that 404s and falls back to the Unicode glyph, while a false
9//! negative leaves the glyph as today — both degrade gracefully.
10
11/// Pinned Twemoji asset version (the maintained jdecked fork; Twitter's original
12/// repo is archived). Stable URLs for offline-cacheable downloads.
13const TWEMOJI_VER: &str = "15.1.0";
14
15const ZWJ: u32 = 0x200D;
16const VS16: u32 = 0xFE0F;
17
18fn is_skin_tone(u: u32) -> bool {
19    (0x1F3FB..=0x1F3FF).contains(&u)
20}
21
22fn is_modifier(u: u32) -> bool {
23    u == VS16 || is_skin_tone(u) // VS16 or skin-tone modifier (extend, never start)
24}
25
26fn is_regional(u: u32) -> bool {
27    (0x1F1E6..=0x1F1FF).contains(&u)
28}
29
30/// Whether `c` can *begin* an emoji cluster. Liberal over the emoji planes/blocks
31/// but excludes the joiners/modifiers/regional indicators (those only ever *extend*
32/// or *pair*, never start) so a lone skin-tone swatch or half a flag isn't coloured.
33fn is_emoji_base(c: char) -> bool {
34    let u = c as u32;
35    if is_modifier(u) || is_regional(u) || u == ZWJ {
36        return false;
37    }
38    matches!(u,
39        0x1F000..=0x1FAFF        // main emoji planes (faces, hands, hearts, …)
40        | 0x2600..=0x27BF        // misc symbols + dingbats (☀ ★ ✈ ✊ ✨ ❤ …)
41        | 0x2B00..=0x2BFF        // misc symbols & arrows (⭐ ⬛ …)
42        | 0x2300..=0x23FF        // misc technical (⌚ ⏰ ⏳ …)
43        | 0x2122 | 0x2139        // ™ ℹ
44        | 0x203C | 0x2049        // ‼ ⁉
45        | 0x24C2                 // Ⓜ
46        | 0x3030 | 0x303D | 0x3297 | 0x3299
47        | 0x00A9 | 0x00AE        // © ®
48        | 0x2934 | 0x2935
49    )
50}

52/// Split `text` into `(slice, is_emoji)` runs, coalescing each emoji grapheme
53/// cluster (base + VS16/skin-tone modifiers + ZWJ-joined continuations, or a
54/// regional-indicator pair = flag) into ONE emoji slice. Non-emoji text accumulates
55/// between them. Never panics: all slicing uses recorded char-boundary byte offsets.
56pub fn segment(text: &str) -> Vec<(&str, bool)> {
57    let mut out: Vec<(&str, bool)> = Vec::new();
58    let cs: Vec<(usize, char)> = text.char_indices().collect();
59    let mut i = 0usize;
60    let mut text_start = 0usize; // byte index where the current non-emoji run began
61    while i < cs.len() {
62        let (byte_i, c) = cs[i];
63        let u = c as u32;
64        // A regional indicator only starts a cluster as a complete pair (flag).
65        let regional_pair =
66            is_regional(u) && cs.get(i + 1).is_some_and(|&(_, n)| is_regional(n as u32));
67        if !is_emoji_base(c) && !regional_pair {
68            i += 1;
69            continue;
70        }
71        // Flush any pending non-emoji text before this cluster.
72        if byte_i > text_start {
73            out.push((&text[text_start..byte_i], false));
74        }
75        let cluster_start = byte_i;
76        let mut j = i + 1;
77        if regional_pair {
78            j = i + 2; // exactly two regional indicators
79        } else {
80            loop {
81                let Some(&(_, nc)) = cs.get(j) else { break };
82                let nu = nc as u32;
83                if is_modifier(nu) {
84                    j += 1;
85                } else if nu == ZWJ {
86                    // Join the next pictographic (+ its modifiers) if present.
87                    if cs.get(j + 1).is_some_and(|&(_, a)| is_emoji_base(a)) {
88                        j += 2;
89                        continue;
90                    }
91                    j += 1; // dangling joiner: absorb it so it doesn't leak into text
92                    break;
93                } else {
94                    break;
95                }
96            }
97        }
98        let cluster_end = cs.get(j).map(|&(b, _)| b).unwrap_or(text.len());
99        out.push((&text[cluster_start..cluster_end], true));
100        i = j;
101        text_start = cluster_end;
102    }
103    if text_start < text.len() {
104        out.push((&text[text_start..], false));
105    }
106    out
107}

109/// Lowercase-hex codepoints of a cluster joined by `-`, optionally dropping the
110/// VS16 (`U+FE0F`) presentation selector.
111fn codepoints(cluster: &str, drop_vs16: bool) -> String {
112    cluster
113        .chars()
114        .map(|c| c as u32)
115        .filter(|&u| !(drop_vs16 && u == VS16))
116        .map(|u| format!("{u:x}"))
117        .collect::<Vec<_>>()
118        .join("-")
119}
120
121/// Stable cache key (filename stem) for a cluster — the raw codepoints, so two
122/// distinct clusters never collide and the key is independent of Twemoji's
123/// irregular FE0F-dropping rules.
124pub fn cache_key(cluster: &str) -> String {
125    codepoints(cluster, false)
126}
127
128/// The Twemoji 72px PNG CDN URL for a codepoint filename stem.
129pub fn twemoji_url(stem: &str) -> String {
130    format!("https://cdn.jsdelivr.net/gh/jdecked/twemoji@{TWEMOJI_VER}/assets/72x72/{stem}.png")
131}
132
133/// Candidate Twemoji URLs to try for a cluster, in order. Twemoji's FE0F-drop rule
134/// is irregular (depends on ZWJ presence), so we try the FE0F-kept name then the
135/// FE0F-stripped name; the first that exists wins.
136pub fn twemoji_url_candidates(cluster: &str) -> Vec<String> {
137    let kept = codepoints(cluster, false);
138    let stripped = codepoints(cluster, true);
139    let mut v = vec![twemoji_url(&kept)];
140    if stripped != kept {
141        v.push(twemoji_url(&stripped));
142    }
143    v
144}

146#[cfg(test)]
147mod tests {
148    use super::*;
149
150    #[test]
151    fn segments_plain_text_as_one_run() {
152        assert_eq!(segment("hello world"), vec![("hello world", false)]);
153    }
154
155    #[test]
156    fn isolates_a_simple_emoji() {
157        // 😀 = U+1F600 (one codepoint, 4 bytes).
158        let segs = segment("hi 😀 yo");
159        assert_eq!(segs, vec![("hi ", false), ("😀", true), (" yo", false)]);
160    }
161
162    #[test]
163    fn coalesces_modifiers_zwj_and_flags() {
164        // Waving hand + skin tone = one cluster.
165        let wave = "\u{1F44B}\u{1F3FB}";
166        assert_eq!(segment(wave), vec![(wave, true)]);
167        // ZWJ sequence (man + ZWJ + computer) = one cluster.
168        let zwj = "\u{1F468}\u{200D}\u{1F4BB}";
169        assert_eq!(segment(zwj), vec![(zwj, true)]);
170        // Flag = two regional indicators = one cluster.
171        let flag = "\u{1F1FA}\u{1F1F8}";
172        assert_eq!(segment(flag), vec![(flag, true)]);
173    }
174
175    #[test]
176    fn url_candidates_try_fe0f_kept_then_stripped() {
177        // ZWJ sequence with FE0F (rainbow flag): the FE0F-kept name is the real
178        // Twemoji asset, so it must be tried first; stripped is the fallback.
179        let c = twemoji_url_candidates("\u{1F3F3}\u{FE0F}\u{200D}\u{1F308}");
180        assert!(c[0].ends_with("1f3f3-fe0f-200d-1f308.png"));
181        assert!(c[1].ends_with("1f3f3-200d-1f308.png"));
182        // Lone heart with VS16: both kept/stripped present as candidates.
183        let h = twemoji_url_candidates("\u{2764}\u{FE0F}");
184        assert!(h[0].ends_with("2764-fe0f.png"));
185        assert!(h[1].ends_with("2764.png"));
186        // Bare emoji: a single candidate.
187        assert_eq!(twemoji_url_candidates("\u{1F600}").len(), 1);
188    }
189
190    #[test]
191    fn cache_key_is_raw_codepoints() {
192        assert_eq!(cache_key("\u{1F44B}\u{1F3FB}"), "1f44b-1f3fb");
193        assert_eq!(cache_key("\u{2764}\u{FE0F}"), "2764-fe0f");
194    }
195
196    #[test]
197    fn lone_modifier_or_half_flag_is_not_an_emoji_cluster() {
198        // A bare skin-tone modifier is plain text, not its own emoji.
199        assert_eq!(segment("\u{1F3FB}"), vec![("\u{1F3FB}", false)]);
200        // A single trailing regional indicator after a flag stays text.
201        let segs = segment("\u{1F1FA}\u{1F1F8}\u{1F1FA}");
202        assert_eq!(segs[0], ("\u{1F1FA}\u{1F1F8}", true));
203        assert_eq!(segs[1], ("\u{1F1FA}", false));
204    }
205
206    #[test]
207    fn dangling_zwj_does_not_leak_into_text() {
208        // man + ZWJ + non-emoji text: the ZWJ is absorbed into the emoji cluster.
209        let segs = segment("\u{1F468}\u{200D}hi");
210        assert_eq!(segs[0], ("\u{1F468}\u{200D}", true));
211        assert_eq!(segs[1], ("hi", false));
212    }
