1//! Shared poll scheduler.
2//!
3//! A single task loops over all enabled monitors, batches the due ones by
4//! detection method, runs detection (Twitch Helix in one batched call;
5//! scrape/generic probes concurrently with a cap), writes results back to the
6//! store, and emits an [`AppEvent::MonitorState`] on any state change. This is
7//! the low-idle-footprint design: one timer, batched work, no thread/process
8//! per channel.
9

24/// Max concurrent scrape/probe checks per tick.
25const MAX_CONCURRENCY: usize = 8;
26/// Upper bound on idle sleep so config changes from the UI are picked up.
27const MAX_SLEEP_SECS: i64 = 30;
28
29#[derive(Clone, Copy)]
30enum PerItemMode {
31    Scrape,
32    Generic,
33    YouTubeApi,
34    KickApi,
35}
36
37/// Run the scheduler until shutdown is signaled.
38pub async fn run(
39    ctx: Arc<DetectContext>,
40    events: EventTx,
41    live_tx: mpsc::UnboundedSender<LiveSignal>,
42    active: ActiveSet,
43    shutdown: Arc<AtomicBool>,
44    jobs: crate::events::JobRegistry,
45) {
46    while !shutdown.load(Ordering::SeqCst) {
47        // Live poll can be disabled from the Background view (a global pause of
48        // detection/recording); idle-check for re-enable without polling.
49        if !ctx.store.job_enabled("job_live_poll") {
50            crate::app_core::sleep_cancellable(Duration::from_secs(10), &shutdown).await;
51            continue;
52        }
53        let wait = tick(&ctx, &events, &live_tx, &active).await;
54        crate::events::mark_job(&jobs, "Live poll", wait as i64);
55        crate::app_core::sleep_cancellable(Duration::from_secs(wait), &shutdown).await;
56    }
57}

59async fn tick(
60    ctx: &Arc<DetectContext>,
61    events: &EventTx,
62    live_tx: &mpsc::UnboundedSender<LiveSignal>,
63    active: &ActiveSet,
64) -> u64 {
65    let rows = match ctx.store.list_monitors_with_channels() {
66        Ok(rows) => rows,
67        Err(e) => {
68            warn!("scheduler: failed to load monitors: {e:#}");
69            return 5;
70        }
71    };
72
73    let now = now_unix();
74    let mut min_wait = MAX_SLEEP_SECS;
75    let mut twitch_items: Vec<DetectItem> = Vec::new();
76    let mut scrape_items: Vec<DetectItem> = Vec::new();
77    let mut generic_items: Vec<DetectItem> = Vec::new();
78    let mut youtube_api_items: Vec<DetectItem> = Vec::new();
79    let mut kick_api_items: Vec<DetectItem> = Vec::new();
80    let mut prev_state: HashMap<i64, String> = HashMap::new();
81    // monitor id -> (channel name, detection short label) for readable logs.
82    let mut meta: HashMap<i64, (String, &'static str)> = HashMap::new();
83
84    let recording: std::collections::HashSet<i64> =
85        active.lock().unwrap().keys().copied().collect();
86
87    for row in &rows {
88        let m = &row.monitor;
89        prev_state.insert(m.id, m.last_state.clone());
90        if !row.channel.enabled || !m.enabled {
91            continue;
92        }
93        // Don't poll a monitor that's currently being recorded — the supervisor
94        // owns its state until the tool exits.
95        if recording.contains(&m.id) {
96            continue;
97        }
98        // Methods handled by the scheduler today; others are driven elsewhere
99        // (CLI self-poll/EventSub in later phases).
100        // EventSubHelix is polled here (Helix) *and* pushed by the EventSub task;
101        // whichever sees live first wins (the supervisor dedupes). WebSub is the
102        // same idea for YouTube: scrape-polled here as a fallback, and pushed by
103        // the websub task (which triggers an on-demand liveness check).
104        // WebSubOnly is push-only — no poll fallback, so it is not in this list.
105        let handled = matches!(
106            m.detection_method,
107            DetectionMethod::TwitchApi
108                | DetectionMethod::EventSubHelix
109                | DetectionMethod::Scrape
110                | DetectionMethod::GenericProbe
111                | DetectionMethod::YouTubeApi
112                | DetectionMethod::KickApi
113                | DetectionMethod::WebSub
114        );
115        if !handled {
116            continue;
117        }
118
119        let interval = m.poll_interval_secs.max(5);
120        let due_at = m.last_checked_at.unwrap_or(0) + interval;
121        if now >= due_at {
122            meta.insert(
123                m.id,
124                (row.channel.name.clone(), m.detection_method.short_label()),
125            );
126            let item = DetectItem {
127                monitor_id: m.id,
128                url: m.url.clone(),
129                platform: m.platform(),
130            };
131            match m.detection_method {
132                DetectionMethod::TwitchApi | DetectionMethod::EventSubHelix => {
133                    twitch_items.push(item)
134                }
135                DetectionMethod::Scrape | DetectionMethod::WebSub => scrape_items.push(item),
136                DetectionMethod::GenericProbe => generic_items.push(item),
137                DetectionMethod::YouTubeApi => youtube_api_items.push(item),
138                DetectionMethod::KickApi => kick_api_items.push(item),
139                _ => {}
140            }
141            min_wait = min_wait.min(interval);
142        } else {
143            min_wait = min_wait.min(due_at - now);
144        }
145    }
146
147    let due = twitch_items.len()
148        + scrape_items.len()
149        + generic_items.len()
150        + youtube_api_items.len()
151        + kick_api_items.len();
152    if due > 0 {
153        debug!(
154            "scheduler: polling {due} monitor(s) due [twitch={} scrape={} generic={} yt-api={} kick={}]",
155            twitch_items.len(),
156            scrape_items.len(),
157            generic_items.len(),
158            youtube_api_items.len(),
159            kick_api_items.len(),
160        );
161    }
162
163    let mut outcomes: Vec<DetectOutcome> = Vec::new();
164    if !twitch_items.is_empty() {
165        outcomes.extend(ctx.detect_twitch(&twitch_items).await);
166    }
167    if !scrape_items.is_empty() {
168        outcomes.extend(run_per_item(ctx, scrape_items, PerItemMode::Scrape).await);
169    }
170    if !generic_items.is_empty() {
171        outcomes.extend(run_per_item(ctx, generic_items, PerItemMode::Generic).await);
172    }
173    if !youtube_api_items.is_empty() {
174        outcomes.extend(run_per_item(ctx, youtube_api_items, PerItemMode::YouTubeApi).await);
175    }
176    if !kick_api_items.is_empty() {
177        outcomes.extend(run_per_item(ctx, kick_api_items, PerItemMode::KickApi).await);
178    }
179
180    let checked_at = now_unix();
181    for o in &outcomes {
182        let new_state = if o.error {
183            "error"
184        } else if o.live {
185            "live"
186        } else {
187            "offline"
188        };
189        if let Err(e) = ctx
190            .store
191            .set_monitor_check_result(o.monitor_id, new_state, checked_at)
192        {
193            warn!(
194                "scheduler: failed to persist state for {}: {e:#}",
195                o.monitor_id
196            );
197        }
198        let old_state = prev_state.get(&o.monitor_id).map(String::as_str);
199        let changed = old_state != Some(new_state);
200
201        // Readable per-poll logging: name [method] result (+ go-live / error
202        // detail). A state change is INFO; a routine poll is DEBUG.
203        let (name, method) = meta
204            .get(&o.monitor_id)
205            .map(|(n, m)| (n.as_str(), *m))
206            .unwrap_or(("?", "?"));
207        let extra = if o.error {
208            format!(" — {}", o.detail)
209        } else if o.live {
210            match o.went_live_at {
211                Some(t) => format!(" (live since {})", fmt_log_time(t)),
212                None => String::new(),
213            }
214        } else {
215            String::new()
216        };
217        if changed {
218            info!(
219                "poll: {name} [{method}] {} -> {new_state}{extra}",
220                old_state.unwrap_or("?")
221            );
222            let _ = events.send(AppEvent::MonitorState {
223                monitor_id: o.monitor_id,
224                state: new_state.to_string(),
225            });
226        } else {
227            debug!("poll: {name} [{method}] {new_state}{extra}");
228        }
229        // Signal the supervisor to (consider) starting a recording. Use the
230        // platform-reported go-live time when available, else approximate it
231        // with our detection time.
232        if o.live && !o.error {
233            let signal = match o.went_live_at {
234                Some(t) => LiveSignal::new(o.monitor_id, Some(t), false),
235                None => LiveSignal::new(o.monitor_id, Some(checked_at), true),
236            }
237            .with_stream_id(o.stream_id.clone())
238            .with_thumbnail_url(o.thumbnail_url.clone())
239            .with_broadcaster_id(o.broadcaster_id.clone())
240            .with_stream_title(o.stream_title.clone());
241            let _ = live_tx.send(signal);
242        }
243    }
244
245    min_wait.clamp(1, MAX_SLEEP_SECS) as u64
246}

248/// Local `HH:MM:SS` for a unix timestamp (log-friendly).
249fn fmt_log_time(t: i64) -> String {
250    chrono::DateTime::from_timestamp(t, 0)
251        .map(|dt| dt.with_timezone(&chrono::Local).format("%H:%M:%S").to_string())
252        .unwrap_or_default()
253}

255async fn run_per_item(
256    ctx: &Arc<DetectContext>,
257    items: Vec<DetectItem>,
258    mode: PerItemMode,
259) -> Vec<DetectOutcome> {
260    let sem = Arc::new(Semaphore::new(MAX_CONCURRENCY));
261    let mut set: JoinSet<DetectOutcome> = JoinSet::new();
262    for item in items {
263        let ctx = ctx.clone();
264        let sem = sem.clone();
265        set.spawn(async move {
266            let _permit = sem.acquire().await.expect("semaphore");
267            match mode {
268                PerItemMode::Scrape => ctx.detect_scrape(&item).await,
269                PerItemMode::Generic => ctx.detect_generic(&item).await,
270                PerItemMode::YouTubeApi => ctx.detect_youtube_api(&item).await,
271                PerItemMode::KickApi => ctx.detect_kick_api(&item).await,
272            }
273        });
274    }
275    let mut out = Vec::new();
276    while let Some(res) = set.join_next().await {
277        if let Ok(o) = res {
278            out.push(o);
279        }
280    }
281    out
282}
